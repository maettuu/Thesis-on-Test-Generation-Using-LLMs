it("should render Japanese vertical punctuation correctly in a missing vertical font", async () => {
  const { PDFDocument } = await import("../../src/display/api.js");
  const { getVerticalPresentationForm } = await import("../../src/core/fonts_utils.js");

  const pdfUrl = "https://example.com/ncode.syosetu.com.pdf";
  const pdfDoc = await PDFDocument.load(await fetch(pdfUrl).then((res) => res.arrayBuffer()));
  const page = await pdfDoc.getPage(1);
  const operatorList = await page.getOperatorList();
  const textContent = await page.getTextContent();

  const expectedVerticalPunctuation = getVerticalPresentationForm()[0x3001]; // IDEOGRAPHIC COMMA
  const actualVerticalPunctuation = textContent.items[0].str;

  expect(actualVerticalPunctuation.charCodeAt(0)).toBe(expectedVerticalPunctuation);
});
