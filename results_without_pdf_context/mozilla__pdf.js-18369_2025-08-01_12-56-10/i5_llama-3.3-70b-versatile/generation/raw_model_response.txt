```javascript
it("should render Japanese vertical punctuation correctly in Firefox", async () => {
  const { PDFDocument } = await import("../../src/display/api.js");
  const { getVerticalPresentationForm } = await import("../../src/core/fonts_utils.js");
  const pdfUrl = `${await import("./test_utils.js").then(m => m.TEST_PDFS_PATH)}/ncode.syosetu.com.pdf`;
  const pdfDoc = await PDFDocument.load(pdfUrl);
  const page = await pdfDoc.getPage(1);
  const textContent = await page.getTextContent();
  const expectedVerticalPunctuation = getVerticalPresentationForm()[0x3001];
  const actualVerticalPunctuation = textContent.items[0].str.charCodeAt(0);
  expect(actualVerticalPunctuation).toBe(expectedVerticalPunctuation);
});
```
