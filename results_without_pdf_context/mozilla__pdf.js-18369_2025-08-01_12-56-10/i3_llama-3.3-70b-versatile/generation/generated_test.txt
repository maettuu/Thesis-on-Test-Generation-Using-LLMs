it("should render Japanese vertical punctuation correctly", async () => {
  const { PDFDocumentProxy } = await import("../../src/display/api.js");
  const { DefaultFileReaderFactory } = await import("./test_utils.js");
  const pdfUrl = `${TEST_PDFS_PATH}ncode.syosetu.com.pdf`;
  const pdfDocument = await PDFDocumentProxy.load({
    url: pdfUrl,
    cMapUrl: CMAP_URL,
    cMapPacked: true,
    pdfWorkerSrc: WORKER_SRC,
  }, {
    fetch: DefaultFileReaderFactory.fetch,
  });
  const pdfPage = await pdfDocument.getPage(1);
  const viewport = new PageViewport({
    width: 800,
  });
  const renderTask = pdfPage.render({
    viewport,
    intent: "display",
  });
  await renderTask.promise;
  const textContent = await pdfPage.getTextContent();
  const expectedText = "、。";
  const actualText = textContent.items[0].str;
  expect(actualText).toContain(expectedText);
});
