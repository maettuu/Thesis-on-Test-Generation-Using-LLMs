it("should not find obsolete files", async () => {
  const fs = await import("fs");
  const path = await import("path");

  const obsoleteFiles = [
    path.resolve(__dirname, "../../src/shared/cffStandardStrings.js"),
    path.resolve(__dirname, "../../src/shared/fonts_utils.js"),
  ];

  const filesExist = obsoleteFiles.some((file) => fs.existsSync(file));
  const expected = false;

  expect(filesExist).toBe(expected);
});
