it("should redo an added imagetextdrawing to the PDF file using keyboard shortcuts", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { PDFDocumentProxy } = await import("../../src/display/api.js");

  const pdfDocument = new PDFDocumentProxy();
  const editorUIManager = new AnnotationEditorUIManager(pdfDocument);

  editorUIManager.addAnnotation({ type: "text", content: "Test" });
  editorUIManager.undo();

  const redoShortcutEvent = new KeyboardEvent("keydown", {
    key: "Z",
    ctrlKey: true,
    shiftKey: true,
  });

  document.dispatchEvent(redoShortcutEvent);

  const annotations = editorUIManager.getAnnotations();
  const expected = [{ type: "text", content: "Test" }];
  expect(annotations).toEqual(expected);
});
