it("should redo an added imagetextdrawing to the PDF file using Keyboard shortcuts", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { PDFDocument } = await import("../../src/core/document.js");
  const { PDFFetchStream } = await import("../../src/display/fetch_stream.js");
  const { PDFPageProxy } = await import("../../src/display/api.js");

  const pdfDoc = await PDFDocument.load(await PDFFetchStream.fetch("path_to_test_pdf"));
  const pageProxy = pdfDoc.getPage(1);
  const annotationEditorUIManager = new AnnotationEditorUIManager(pageProxy);

  annotationEditorUIManager.addEventListener("annotationAdded", () => {
    annotationEditorUIManager.undo();
    annotationEditorUIManager.redo();
  });

  annotationEditorUIManager.addAnnotation("testAnnotation");

  const expectedAnnotations = 1;
  const actualAnnotations = annotationEditorUIManager.annotations.length;

  expect(actualAnnotations).toBe(expectedAnnotations);
});
