it("should redo an added image/text/drawing to the PDF file using Keyboard shortcuts", async () => {
  const { AnnotationEditorUIManager, CommandManager, KeyboardManager } = await import("../../src/display/editor/tools.js");
  const { AnnotationEditorType } = await import("../../src/shared/util.js");
  const annotationEditorUIManager = new AnnotationEditorUIManager(null, null, null, null);
  annotationEditorUIManager.updateMode(AnnotationEditorType.FREE_TEXT);
  const editor = { id: 'test', pageIndex: 0, isEmpty: () => false, remove: () => {}, commitOrRemove: () => {} };
  annotationEditorUIManager.addEditorToLayer(editor);
  annotationEditorUIManager.setSelected(editor);
  annotationEditorUIManager.addCommands({ cmd: () => {}, undo: () => {}, mustExec: true });
  annotationEditorUIManager.undo();
  const keyboardManager = new KeyboardManager([
    [["ctrl+y", "ctrl+shift+Z", "mac+meta+shift+Z", "ctrl+shift+z", "mac+meta+shift+Z", "mac+meta+shift+z"], annotationEditorUIManager.redo],
  ]);
  keyboardManager.exec(annotationEditorUIManager, { key: 'y', ctrlKey: true });
  expect(annotationEditorUIManager.getMode()).toBe(AnnotationEditorType.FREE_TEXT);
});
