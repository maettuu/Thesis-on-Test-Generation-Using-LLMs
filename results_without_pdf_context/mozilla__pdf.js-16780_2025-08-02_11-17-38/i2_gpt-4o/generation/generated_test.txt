it("should resize SVG images without preserving aspect ratio", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const svgData = `data:image/svg+xml;charset=UTF-8,<svg viewBox="0 0 100 100" xmlns="http://www.w3.org/2000/svg"><rect width="100" height="100" style="fill:red;"/></svg>`;
  const imageManager = new ImageManager();
  const data = await imageManager.#get("testSvg", svgData);
  const canvas = new OffscreenCanvas(100, 200);
  const ctx = canvas.getContext("2d");
  ctx.drawImage(data.bitmap, 0, 0, 100, 100, 0, 0, 100, 200);
  const imageData = ctx.getImageData(0, 150, 1, 1).data;
  const isRed = imageData[0] === 255 && imageData[1] === 0 && imageData[2] === 0 && imageData[3] === 255;
  expect(isRed).toBe(true);
});
