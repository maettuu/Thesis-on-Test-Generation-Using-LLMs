it("should allow resizing of SVG images horizontallyvertically in pdfjs", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const imageManager = new ImageManager();
  const svg = `data:image/svg+xml;charset=UTF-8,<svg viewBox="0 0 1 1" width="1" height="1" xmlns="http://www.w3.org/2000/svg"><rect width="1" height="1" style="fill:red;"/></svg>`;
  const image = await imageManager.#get(svg, svg);
  const canvas = new OffscreenCanvas(10, 10);
  const ctx = canvas.getContext("2d");
  ctx.drawImage(image.bitmap, 0, 0, 10, 10);
  const imageData = ctx.getImageData(0, 0, 10, 10).data;
  const expectedColor = [255, 0, 0, 255]; // Red color
  const actualColor = [
    imageData[0],
    imageData[1],
    imageData[2],
    imageData[3],
  ];
  expect(actualColor).toEqual(expectedColor);
});
