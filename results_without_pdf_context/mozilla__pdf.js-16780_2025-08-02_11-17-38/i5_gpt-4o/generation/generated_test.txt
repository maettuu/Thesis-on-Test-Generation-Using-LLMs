it("should resize SVG images horizontally and vertically", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const svgData = `data:image/svg+xml;charset=UTF-8,<svg viewBox="0 0 100 100" xmlns="http://www.w3.org/2000/svg"><rect width="100" height="100" style="fill:red;"/></svg>`;
  const canvas = new OffscreenCanvas(200, 200);
  const ctx = canvas.getContext("2d");
  const imageElement = new Image();
  imageElement.src = svgData;
  await imageElement.decode();
  ctx.drawImage(imageElement, 0, 0, 100, 100, 0, 0, 200, 200);
  const imageData = ctx.getImageData(0, 0, 200, 200).data;
  const isResized = Array.from(imageData).some((value, index) => index % 4 === 0 && value === 255);
  expect(isResized).toBe(true);
});
