it("should resize SVG images without preserving aspect ratio", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const svgData = `data:image/svg+xml;charset=UTF-8,<svg viewBox="0 0 1 1" width="1" height="1" xmlns="http://www.w3.org/2000/svg"><rect width="1" height="1" style="fill:red;"/></svg>`;
  const imageManager = new ImageManager();
  const data = await imageManager.#get("testSvg", svgData);
  const canvas = new OffscreenCanvas(1, 3);
  const ctx = canvas.getContext("2d");
  ctx.drawImage(data.bitmap, 0, 0, 1, 1, 0, 0, 1, 3);
  const pixelData = new Uint32Array(ctx.getImageData(0, 0, 1, 1).data.buffer)[0];
  expect(pixelData).toBe(0);
});
