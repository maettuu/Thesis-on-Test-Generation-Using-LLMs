it("should resize SVG images horizontallyvertically without keeping the aspect ratio", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const { Util } = await import("../../shared/util.js");
  const imageManager = new ImageManager();
  const svgImage = new Blob([`<svg width="100" height="100"><rect width="100" height="100" style="fill:red;"/></svg>`], { type: 'image/svg+xml' });
  const image = await imageManager.#get('svgImage', svgImage);
  const imageElement = image.bitmap;
  imageElement.width = 50;
  imageElement.height = 200;
  const canvas = new OffscreenCanvas(50, 200);
  const ctx = canvas.getContext("2d");
  ctx.drawImage(imageElement, 0, 0, 50, 200);
  const pixels = ctx.getImageData(0, 0, 50, 200).data;
  let isResized = false;
  for (let i = 0; i < pixels.length; i += 4) {
    if (pixels[i] === 255 && pixels[i + 1] === 0 && pixels[i + 2] === 0) {
      isResized = true;
      break;
    }
  }
  expect(isResized).toBe(true);
});
