it("should not block UI when changing font size in editor", async () => {
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const editor = new AnnotationEditor({
    parent: { moveEditorInDOM: jest.fn() },
    uiManager: {},
    id: "test-editor",
    name: "test-name",
    pageIndex: 0,
    isCentered: false,
  });

  // Simulate font size change
  editor.fontSize = 24;

  // Verify that moveInDOM is called after a timeout without blocking
  expect(editor.moveInDOM).toHaveBeenCalled();
});
