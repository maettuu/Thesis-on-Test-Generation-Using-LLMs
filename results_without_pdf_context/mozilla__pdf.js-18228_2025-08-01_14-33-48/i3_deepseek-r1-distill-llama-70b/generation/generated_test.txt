it("should hide barcode annotations when PMD is present", () => {
  const dummyDict = {
    has: key => key === "PMD",
    get: () => undefined,
  };
  const params = {
    subtype: "Widget",
    fieldType: "Tx",
    dict: dummyDict,
    data: {
      readOnly: true,
      noHTML: false,
      fieldValue: "Test",
    },
  };
  const annotation = AnnotationFactory.createAnnotation(params);
  if (!(annotation.flags & AnnotationFlag.HIDDEN)) {
    throw new Error("Expected annotation to have the HIDDEN flag set when PMD is present");
  }
  if (annotation.data.hidden !== true) {
    throw new Error("Expected annotation.data.hidden to be true when PMD is present");
  }
});
