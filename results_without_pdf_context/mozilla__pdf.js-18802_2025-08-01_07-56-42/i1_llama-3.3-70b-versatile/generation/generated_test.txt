it("should render texts correctly in PDF", async () => {
  const { PDFDocumentProxy } = await import("../../src/display/api.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");
  const params = buildGetDocumentParams(`${TEST_PDFS_PATH}issue.pdf`);
  const pdfDoc = await PDFDocumentProxy.load(params);
  const page = await pdfDoc.getPage(1);
  const textContent = await page.getTextContent();
  const expectedText = "Expected text content";
  expect(textContent.items[0].str).toBe(expectedText);
});
