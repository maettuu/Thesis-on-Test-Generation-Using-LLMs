it("should render texts correctly in PDF", async () => {
  const { PDFDocument } = await import("../../src/display/api.js");
  const { getStdFontMap } = await import("./standard_fonts.js");
  const fontName = "GillSansMT";
  const expectedFontName = "Helvetica";
  const actualFontName = getStdFontMap()[fontName];
  expect(actualFontName).toBe(expectedFontName);
});
