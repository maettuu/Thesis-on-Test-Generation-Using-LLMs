it("should render PDF text fields correctly after applying the patch", async () => {
  const { PDFDocumentLoadingTask, PDFWorker } = await import("../../src/display/api.js");
  const { DefaultFileReaderFactory } = await import("./test_utils.js");
  const { GlobalWorkerOptions } = await import("../../src/display/worker_options.js");
  const worker = new PDFWorker({
    workerSrc: "../../build/generic/build/pdf.worker.mjs",
  });
  const pdfFile = `${TEST_PDFS_PATH}/example.pdf`;
  const loadingTask = PDFDocumentLoadingTask.fromUrl(pdfFile, {
    cMapUrl: CMAP_URL,
    cMapPacked: true,
  });
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const viewport = pdfPage.getViewport({ scale: 1 });
  const renderTask = pdfPage.render({
    viewport,
    canvasContext: null,
    renderInteractive: false,
    annotationMode: AnnotationMode.NONE,
  });
  await renderTask.promise;
  const textContent = await pdfPage.getTextContent();
  const expectedText = "Hello World";
  const actualText = textContent.items[0].str;
  expect(actualText).toBe(expectedText);
});
