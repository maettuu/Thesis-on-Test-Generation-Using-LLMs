it("should render GillSansMT fonts correctly", async () => {
  const { PDFDocument } = await import("../../src/display/api.js");
  const { getFontSubstitution } = await import("../../src/core/font_substitutions.js");
  const fontName = "GillSansMT";
  const expectedFontName = "Helvetica";
  const actualFontName = getFontSubstitution(fontName);
  expect(actualFontName).toBe(expectedFontName);
});
