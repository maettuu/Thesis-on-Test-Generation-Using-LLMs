```javascript
it("should prevent circular references in Form XObjects", async () => {
  const pdfData = await fetch('https://github.com/user-attachments/files/19691094/evil_xobjects.pdf').then(response => response.arrayBuffer());
  const pdf = await import("../../src/display/api.js");
  const { PDFDocument } = pdf;
  const doc = await PDFDocument.load(pdfData);
  const pages = await doc.getPages();
  const page = pages[0];
  const renderedPage = await page.render({ intent: 'display' });
  expect(renderedPage).not.toBeNull();
});
```
