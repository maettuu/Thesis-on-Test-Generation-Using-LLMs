it("should throw an error when encountering a circular reference in Form XObjects", async () => {
  const { PDFDocument } = await import("../../src/display/api.js");
  const pdfData = await import("../../web/annotation_layer_builder.js");
  const pdfBytes = await fetch('https://github.com/user-attachments/files/19691094/evil_xobjects.pdf').then(res => res.arrayBuffer());
  const pdf = await PDFDocument.load(pdfBytes);
  const page = await pdf.getPage(1);
  await expect(page.getTextContent()).rejects.toThrowError('getOperatorList - ignoring circular reference');
});
