it("should prevent endless loop when loading PDF with circular Form XObject references", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { fetch } = await import("../../src/display/network.js");
  const pdfUrl = "https://github.com/user-attachments/files/19691094/evil_xobjects.pdf";
  const pdfResponse = await fetch(pdfUrl);
  const pdfBuffer = await pdfResponse.arrayBuffer();
  const pdfDocument = await PDFDocument.load(pdfBuffer);
  const firstPage = await pdfDocument.getPage(1);
  const operatorListPromise = firstPage.getOperatorList();
  await expect(operatorListPromise).not.toThrow();
});
