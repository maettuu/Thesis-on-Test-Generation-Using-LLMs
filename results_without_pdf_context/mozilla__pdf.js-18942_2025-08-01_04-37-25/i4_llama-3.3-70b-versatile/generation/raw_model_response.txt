it("should translate font correctly when fontFile is a Dict", async () => {
  const { PartialEvaluator, Dict, Name, NullStream } = await import("../../src/core/evaluator.js");
  const evaluator = new PartialEvaluator({
    xref: { fetchIfRef: () => new Dict() },
    handler: { send: () => {} },
    pageIndex: 1,
    idFactory: { createObjId: () => 1 },
    fontCache: new Map(),
    builtInCMapCache: new Map(),
    standardFontDataCache: new Map(),
    globalImageCache: new Map(),
    systemFontCache: new Map(),
    options: {},
  });
  const fontFile = new Dict();
  fontFile.set("Subtype", Name.get("Type1"));
  const descriptor = new Dict();
  descriptor.set("FontFile", fontFile);
  const font = new Dict();
  font.set("Descriptor", descriptor);
  const loadedName = await evaluator.loadFont("fontName", null, new Dict(), null, null);
  expect(loadedName.loadedName).not.toBeNull();
});
