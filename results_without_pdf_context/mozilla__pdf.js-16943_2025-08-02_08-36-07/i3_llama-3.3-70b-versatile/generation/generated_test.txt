it("should change the style of the line and the resizers around a selected editor", async () => {
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const editor = new AnnotationEditor({
    parent: { pageIndex: 1, pageDimensions: [100, 100], div: document.createElement("div") },
    uiManager: { addCommands: () => {} },
    id: "editor",
    name: "editor",
  });
  editor.select();
  const resizersDiv = editor.#resizersDiv;
  expect(resizersDiv.classList.contains("hidden")).toBe(false);
  expect(editor.div.classList.contains("selectedEditor")).toBe(true);
});
