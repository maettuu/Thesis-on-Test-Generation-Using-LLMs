it("should change the style of the line and the resizers around a selected editor", async () => {
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const { getComputedStyle } = await import("../../src/shared/util.js");
  const editor = new AnnotationEditor({ parent: { viewport: { rotation: 0 } }, id: 1, x: 10, y: 10, uiManager: {} });
  const style = getComputedStyle(document.documentElement);
  AnnotationEditor.initialize({ get: () => {} });
  editor.div = { style: {} };
  editor.fixAndSetPosition();
  const left = editor.div.style.left;
  const top = editor.div.style.top;
  expect(left).not.toBe(`${(100 * editor.x).toFixed(2)}%`);
  expect(top).not.toBe(`${(100 * editor.y).toFixed(2)}%`);
});
