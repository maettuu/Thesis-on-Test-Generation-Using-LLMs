```javascript
it("should change the style of the line and the resizers around a selected editor", async () => {
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const { AnnotationEditorLayer } = await import("../../src/display/editor/annotation_editor_layer.js");
  const { PDFPageView } = await import("../../web/pdf_page_view.js");

  const pageView = new PDFPageView();
  const annotationEditorLayer = new AnnotationEditorLayer(pageView);
  const annotationEditor = new AnnotationEditor(annotationEditorLayer);

  const initialBorderWidth = annotationEditor.div.style.borderStyle;
  const initialResizerStyle = annotationEditor.div.children[0].style.borderStyle;

  annotationEditor.initialize();

  const updatedBorderWidth = annotationEditor.div.style.borderStyle;
  const updatedResizerStyle = annotationEditor.div.children[0].style.borderStyle;

  expect(updatedBorderWidth).not.toBe(initialBorderWidth);
  expect(updatedResizerStyle).not.toBe(initialResizerStyle);
});
```
