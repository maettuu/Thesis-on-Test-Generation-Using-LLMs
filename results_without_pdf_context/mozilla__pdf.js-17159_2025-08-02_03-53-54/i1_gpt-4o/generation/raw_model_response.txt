```javascript
it("should reflect the correct state of radio buttons on the pdf form", async () => {
  const { ButtonWidgetAnnotation } = await import("../../src/core/annotation.js");
  const { Dict, Name } = await import("../../src/core/primitives.js");

  const params = {
    dict: new Dict({
      Parent: new Dict({
        V: Name.get("Radio1"),
      }),
      AP: new Dict({
        N: new Dict({
          Radio1: new StringStream("checked appearance"),
          Off: new StringStream("unchecked appearance"),
        }),
      }),
    }),
  };

  const radioButton = new ButtonWidgetAnnotation(params);
  const expectedFieldValue = "Radio1";
  const expectedButtonValue = "Radio1";

  expect(radioButton.data.fieldValue).toEqual(expectedFieldValue);
  expect(radioButton.data.buttonValue).toEqual(expectedButtonValue);
});
```
