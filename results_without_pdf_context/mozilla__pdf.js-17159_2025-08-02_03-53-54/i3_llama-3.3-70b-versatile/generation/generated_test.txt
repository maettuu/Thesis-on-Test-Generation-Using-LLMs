it("should correctly set the buttonValue in ButtonWidgetAnnotation for radio buttons", async () => {
  const xref = new XRefMock();
  const annotationGlobals = await AnnotationFactory.createGlobals(xref);
  const idFactory = createIdFactory();
  const dict = new Dict(xref);
  dict.set("FT", Name.get("Btn"));
  dict.set("Ff", AnnotationFieldFlag.RADIO);
  dict.set("AP", new Dict(xref));
  const ap = dict.get("AP");
  ap.set("N", new Dict(xref));
  const normalAppearance = ap.get("N");
  normalAppearance.set("Yes", new StringStream(""));
  normalAppearance.set("Off", new StringStream(""));
  const params = {
    xref,
    ref: new Ref(xref, 1),
    dict,
    subtype: "Widget",
    id: idFactory.createObjId(),
    annotationGlobals,
    collectFields: true,
  };
  const annotation = await AnnotationFactory._create(params);
  expect(annotation.data.buttonValue).toBe("Yes");
});
