it("should reflect the correct state of radio buttons on the pdf form", async () => {
  const { ButtonWidgetAnnotation } = await import("../../src/core/annotation.js");
  const { Dict, Name } = await import("../../src/core/primitives.js");

  const params = {
    dict: new Dict(null),
    xref: null,
    annotationGlobals: { xfaDatasets: null, acroForm: new Dict(null) },
  };

  params.dict.set("Parent", new Dict(null));
  params.dict.get("Parent").set("V", Name.get("On"));
  params.dict.set("AP", new Dict(null));
  params.dict.get("AP").set("N", new Dict(null));
  params.dict.get("AP").get("N").set("On", new Name("On"));
  params.dict.get("AP").get("N").set("Off", new Name("Off"));

  const annotation = new ButtonWidgetAnnotation(params);
  const expected = "On";
  const actual = annotation.data.fieldValue;

  expect(actual).toBe(expected);
});
