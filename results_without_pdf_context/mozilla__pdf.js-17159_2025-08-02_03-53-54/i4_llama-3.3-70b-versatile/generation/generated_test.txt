it("should correctly set the buttonValue in ButtonWidgetAnnotation for radio buttons", async () => {
  const { ButtonWidgetAnnotation, AnnotationFactory } = await import("./src/core/annotation.js");
  const annotation = new ButtonWidgetAnnotation({
    dict: {
      FT: "Btn",
      Ff: 0,
      AP: {
        N: {
          Off: {},
        },
      },
    },
    xref: {},
    annotationGlobals: {},
    collectFields: true,
    pageIndex: 0,
    pageRef: {},
  });
  annotation._processRadioButton({
    params: {
      dict: {
        FT: "Btn",
        Ff: 0,
        AP: {
          N: {
            Off: {},
          },
        },
      },
      xref: {},
      annotationGlobals: {},
      collectFields: true,
      pageIndex: 0,
      pageRef: {},
    },
  });
  const expectedButtonValue = null;
  const actualButtonValue = annotation.data.buttonValue;
  expect(actualButtonValue).toBe(expectedButtonValue);
});
