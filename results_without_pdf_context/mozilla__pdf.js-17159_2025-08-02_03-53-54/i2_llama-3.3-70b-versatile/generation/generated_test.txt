it("should reflect the real state of radio buttons in print view", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { AnnotationFactory } = await import("../../src/core/annotation.js");
  const { createDefaultAppearance } = await import("../../src/core/default_appearance.js");
  const { Dict, Name, Ref } = await import("../../src/core/primitives.js");

  const pdfDoc = await PDFDocument.load(await fetch("path_to_pdf_with_radio_buttons.pdf").then(res => res.arrayBuffer()));
  const annotationFactory = new AnnotationFactory();
  const page = pdfDoc.getPage(1);
  const annotations = await page.getAnnotations();

  const radioButtonAnnotation = annotations.find(annotation => annotation instanceof annotationFactory.constructor.ButtonWidgetAnnotation && annotation.data.radioButton);

  const fieldValue = radioButtonAnnotation.data.fieldValue;
  const buttonValue = radioButtonAnnotation.data.buttonValue;

  expect(fieldValue).not.toBeNull();
  expect(buttonValue).not.toBeNull();
});
