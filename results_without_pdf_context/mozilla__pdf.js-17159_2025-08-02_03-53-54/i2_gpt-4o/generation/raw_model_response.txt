```javascript
it("should reflect the correct state of radio buttons on the pdf form", async () => {
  const { ButtonWidgetAnnotation } = await import("../../src/core/annotation.js");
  const { Dict, Name } = await import("../../src/core/primitives.js");

  const params = {
    dict: new Dict(),
  };

  const parentDict = new Dict();
  parentDict.set("V", Name.get("RadioValue"));
  params.dict.set("Parent", parentDict);

  const appearanceDict = new Dict();
  const normalAppearanceDict = new Dict();
  normalAppearanceDict.set("RadioValue", null);
  normalAppearanceDict.set("Off", null);
  appearanceDict.set("N", normalAppearanceDict);
  params.dict.set("AP", appearanceDict);

  const radioButton = new ButtonWidgetAnnotation(params);
  const expectedFieldValue = "RadioValue";
  const expectedButtonValue = "RadioValue";

  const actualFieldValue = radioButton.data.fieldValue;
  const actualButtonValue = radioButton.data.buttonValue;

  expect(actualFieldValue).toEqual(expectedFieldValue);
  expect(actualButtonValue).toEqual(expectedButtonValue);
});
```
