it("should reflect the correct state of radio buttons on the pdf form", async () => {
  const { AnnotationFactory } = await import("../../src/core/annotation.js");
  const { PDFDocument } = await import("../../src/core/document.js");

  const pdfDoc = await PDFDocument.load(/* PDF data with radio buttons */);
  const page = await pdfDoc.getPage(0);
  const annotations = await page.getAnnotations();

  const radioButtonAnnotation = annotations.find(
    (annotation) => annotation.fieldType === "RadioButton"
  );

  const expectedState = "On"; // Assuming "On" is the expected state for the test
  const actualState = radioButtonAnnotation.data.fieldValue;

  expect(actualState).toBe(expectedState);
});
