it("should correctly reflect radio button state in print view", async () => {
  const { ButtonWidgetAnnotation, Dict, Name, Ref } = await import("../../src/core/annotation.js");
  const { AnnotationFactory } = await import("../../src/core/annotation.js");
  const { createIdFactory } = await import("./test_utils.js");
  const idFactory = createIdFactory();
  const annotationFactory = new AnnotationFactory(idFactory);
  const params = {
    dict: new Dict({
      FT: new Name("Btn"),
      T: new Name("Radio Button"),
      Parent: new Ref(1),
      AP: new Dict({
        N: new Dict({
          On: new StringStream("Checked Appearance"),
          Off: new StringStream("Unchecked Appearance"),
        }),
      }),
    }),
  };
  const annotation = annotationFactory.createAnnotation(params);
  const buttonWidgetAnnotation = new ButtonWidgetAnnotation(params);
  const fieldValue = buttonWidgetAnnotation.data.fieldValue;
  const buttonValue = buttonWidgetAnnotation.data.buttonValue;
  assert.strictEqual(fieldValue, null);
  assert.strictEqual(buttonValue, null);
});
