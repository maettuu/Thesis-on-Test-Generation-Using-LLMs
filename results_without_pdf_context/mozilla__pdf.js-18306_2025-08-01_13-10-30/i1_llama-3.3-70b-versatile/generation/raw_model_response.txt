it("should render PDF with non-integer DW correctly", async () => {
  const { PartialEvaluator, Dict, Name, Stream, StringStream, XRefMock } = await import("../../src/core/evaluator.js");
  const { createIdFactory } = await import("./test_utils.js");
  const xref = new XRefMock();
  const idFactory = createIdFactory();
  const dict = new Dict();
  dict.set("DW", 100.5);
  dict.set("W", [1, 2, 3]);
  const evaluator = new PartialEvaluator({
    xref,
    handler: null,
    pageIndex: 0,
    idFactory,
    fontCache: {},
    builtInCMapCache: {},
    standardFontDataCache: {},
    globalImageCache: {},
    systemFontCache: {},
  });
  const properties = { composite: true };
  evaluator.extractWidths(dict, {}, properties);
  const defaultWidth = properties.defaultWidth;
  expect(defaultWidth).toBeCloseTo(101);
});
