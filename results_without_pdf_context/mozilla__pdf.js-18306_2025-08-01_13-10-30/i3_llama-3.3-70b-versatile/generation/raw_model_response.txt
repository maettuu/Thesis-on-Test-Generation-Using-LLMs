it("should render PDF with non-integer DW correctly", async () => {
  const xref = new XRefMock();
  const idFactory = createIdFactory();
  const handler = {
    send: jest.fn(),
    sendWithPromise: jest.fn(),
  };
  const evaluator = new PartialEvaluator({
    xref,
    handler,
    pageIndex: 0,
    idFactory,
  });
  const dict = new Dict();
  dict.set("DW", 1000.5);
  const properties = {
    composite: true,
  };
  const defaultWidth = evaluator.extractWidths(dict, null, properties).defaultWidth;
  expect(defaultWidth).toBeCloseTo(1000);
});
