it("should render PDF with non-integer DW correctly", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { PartialEvaluator } = await import("../../src/core/evaluator.js");
  const pdf = await PDFDocument.load(await (await import("../../src/display/fetch_stream.js")).PDFFetchStream.fromUrl("./test.pdf"));
  const page = pdf.getPage(1);
  const operatorList = await page.getOperatorList();
  const evaluator = new PartialEvaluator();
  const dict = operatorList.args[0];
  dict.set("DW", 100.5);
  const defaultWidthBefore = evaluator.getDefaultWidth(dict);
  dict.set("DW", 100.5);
  const defaultWidthAfter = evaluator.getDefaultWidth(dict);
  expect(defaultWidthBefore).not.toBe(defaultWidthAfter);
  expect(defaultWidthAfter).toBe(Math.ceil(100.5));
});
