it("should generate a text layer for PDFs with broken font names", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const url = "https://github.com/user-attachments/files/20281525/Capital.Bancorp.Inc.Corporate.Responsibility.2025509.SD000000003063823863-1-2.pdf";
  const loadingTask = getDocument(url);
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const textContent = await pdfPage.getTextContent();
  const hasTextLayer = textContent.items.length > 0;
  expect(hasTextLayer).toBe(true);
});
