it("should generate a text layer for PDFs with Type3 fonts and missing FontDescriptor", async () => {
  const { PartialEvaluator, Dict, Name, Stream, StringStream, OperatorList, WorkerTask } = await import("../../src/core/evaluator.js");
  const { createIdFactory, XRefMock } = await import("./test_utils.js");
  const xref = new XRefMock();
  const idFactory = createIdFactory();
  const evaluator = new PartialEvaluator({
    xref,
    handler: {},
    pageIndex: 0,
    idFactory,
    fontCache: {},
    builtInCMapCache: {},
    standardFontDataCache: {},
    globalColorSpaceCache: {},
    globalImageCache: {},
    systemFontCache: {},
  });
  const dict = new Dict();
  dict.set("Type", Name.get("Type3"));
  dict.set("FontBBox", [0, 0, 100, 100]);
  const font = await evaluator.translateFont({
    descriptor: null,
    dict,
    baseDict: dict,
    composite: false,
    type: "Type3",
    firstChar: 0,
    lastChar: 100,
    toUnicode: null,
    cssFontInfo: null,
  });
  expect(font).toBeInstanceOf(Object);
});
