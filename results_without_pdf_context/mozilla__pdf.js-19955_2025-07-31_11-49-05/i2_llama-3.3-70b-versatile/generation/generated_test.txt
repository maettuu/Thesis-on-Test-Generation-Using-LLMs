it("should generate a text layer for PDFs with Type fonts and incomplete FontDescriptor", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { PDFFetchStream } = await import("../../src/display/fetch_stream.js");
  const pdfUrl = "https://github.com/user-attachments/files/20281525/Capital.Bancorp.Inc.Corporate.Responsibility.2025509.SD000000003063823863-1-2.pdf";
  const pdf = await PDFDocument.load({ url: pdfUrl, fetch: PDFFetchStream.fetch });
  const page = await pdf.getPage(1);
  const textContent = await page.getTextContent();
  const expectedText = "Capital";
  const actualText = textContent.items[0].str;
  expect(actualText).toContain(expectedText);
});
