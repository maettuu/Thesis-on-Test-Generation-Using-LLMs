it("should filter out nonBaseStream instances from the streams array in StreamsSequenceStream", async () => {
  const { BaseStream, StreamsSequenceStream } = await import("../../src/core/decode_stream.js");
  const streams = [new BaseStream(), 'non-stream', new BaseStream()];
  const streamsSequenceStream = new StreamsSequenceStream(streams);
  const expectedStreams = streams.filter(stream => stream instanceof BaseStream);
  expect(streamsSequenceStream.streams).toEqual(expectedStreams);
});
