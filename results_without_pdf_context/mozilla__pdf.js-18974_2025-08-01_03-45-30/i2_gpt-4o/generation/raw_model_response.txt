it("should skip non-Stream sub-streams in StreamsSequenceStream", async () => {
  const { StreamsSequenceStream } = await import("../../src/core/decode_stream.js");
  const { BaseStream } = await import("../../src/core/base_stream.js");
  const { Stream } = await import("../../src/core/stream.js");

  const validStream = new Stream(new Uint8Array([1, 2, 3]));
  const invalidStream = {}; // Not an instance of BaseStream
  const streams = [validStream, invalidStream];

  const sequenceStream = new StreamsSequenceStream(streams);
  const expectedLength = validStream.length;
  const actualLength = sequenceStream._rawMinBufferLength;

  expect(actualLength).toBe(expectedLength);
});
