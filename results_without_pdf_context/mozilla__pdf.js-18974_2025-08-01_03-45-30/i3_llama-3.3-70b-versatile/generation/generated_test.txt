it("should filter out nonBaseStream instances in StreamsSequenceStream", async () => {
  const { StreamsSequenceStream, DecodeStream, Stream } = await import("../../src/core/decode_stream.js");
  const streams = [new DecodeStream(), new Stream(), new DecodeStream()];
  const streamsSequenceStream = new StreamsSequenceStream(streams);
  const expectedStreams = streams.filter(s => s instanceof DecodeStream);
  const actualStreams = streamsSequenceStream.streams;
  expect(actualStreams).toEqual(expectedStreams);
});
