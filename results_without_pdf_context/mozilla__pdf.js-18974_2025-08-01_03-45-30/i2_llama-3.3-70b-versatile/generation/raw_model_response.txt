it("should render the signature page of a digitally signed PDF", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { PDFFetchStream } = await import("../../src/display/fetch_stream.js");
  const pdfUrl = "https://github.com/user-attachments/files/17550758/doc1520828609.pdf";
  const pdf = await PDFDocument.load({ url: pdfUrl, fetch: (url) => new PDFFetchStream(url) });
  const pageNumber = 45;
  const page = await pdf.getPage(pageNumber);
  const renderTask = page.render({ viewport: page.getViewport({ scale: 1 }) });
  const expectedError = null;
  const actualError = await new Promise((resolve) => {
    renderTask.promise.then(() => resolve(null), (error) => resolve(error));
  });
  expect(actualError).toBe(expectedError);
});
