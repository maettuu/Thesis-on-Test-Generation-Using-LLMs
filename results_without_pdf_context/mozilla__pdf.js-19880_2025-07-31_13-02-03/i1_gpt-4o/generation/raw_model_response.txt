```javascript
it("should throw an error when input is null", async () => {
  const { numberToString } = await import("../../src/core/core_utils.js");
  let error;
  try {
    numberToString(null);
  } catch (e) {
    error = e;
  }
  expect(error).toBeDefined();
  expect(error.message).toBe("numberToString - the value (null) should be a number.");
});
```
