it("should speed up decoding images in PDFs using ImageDecoder", async () => {
  const { JpegStream } = await import("../../src/core/jpeg_stream.js");
  const { JpegImage } = await import("../../src/core/jpg.js");
  const xref = new XRefMock();
  const dict = new Dict();
  const stream = new StringStream("FFD8FF");
  const jpegStream = new JpegStream(stream, null, dict);
  const getTransferableImage = jpegStream.getTransferableImage;
  const imageDecoderSupported = await JpegStream.canUseImageDecoder;
  if (imageDecoderSupported) {
    const image = await getTransferableImage.call(jpegStream);
    expect(image).not.toBeNull();
  } else {
    expect(getTransferableImage).toThrowError();
  }
});
