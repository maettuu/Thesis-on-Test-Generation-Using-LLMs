it("should use ImageDecoder to decode JPEG images", async () => {
  const { JpegStream } = await import("../../src/core/jpeg_stream.js");
  const { BaseStream } = await import("../../src/core/base_stream.js");
  const { shadow } = await import("../../src/shared/util.js");

  const mockStream = new BaseStream();
  const jpegStream = new JpegStream(mockStream, null, null);

  const mockImageDecoder = {
    decode: async () => ({ image: "decodedImage" }),
    close: () => {},
  };

  global.ImageDecoder = class {
    constructor() {
      return mockImageDecoder;
    }
    static isTypeSupported() {
      return true;
    }
  };

  const expected = "decodedImage";
  const actual = await jpegStream.getTransferableImage();
  expect(actual).toEqual(expected);
});
