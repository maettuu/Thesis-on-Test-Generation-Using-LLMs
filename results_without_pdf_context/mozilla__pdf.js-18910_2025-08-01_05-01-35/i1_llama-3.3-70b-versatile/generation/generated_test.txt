it("should use ImageDecoder to speed up decoding images in PDFs", async () => {
  const { JpegStream } = await import("../../src/core/jpeg_stream.js");
  const { PDFImage } = await import("../../src/core/image.js");
  const { JpegImage } = await import("../../src/core/jpg.js");
  const image = new Uint8Array([0xff, 0xd8, 0xff, 0xe0, 0x00, 0x10, 0x4a, 0x46, 0x49, 0x46, 0x00, 0x01]);
  const stream = { getBytes: () => image };
  const jpegStream = new JpegStream(stream, null, null);
  const pdfImage = new PDFImage({ image: jpegStream });
  const transferableImage = await pdfImage.image.getTransferableImage();
  expect(transferableImage).not.toBeNull();
});
