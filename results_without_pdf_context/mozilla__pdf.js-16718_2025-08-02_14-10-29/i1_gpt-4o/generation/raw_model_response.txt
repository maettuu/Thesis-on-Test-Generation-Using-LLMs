it("should save PDF annotations correctly", async () => {
  const { updateAcroform } = await import("../../src/core/writer.js");
  const { XRefMock } = await import("./test_utils.js");

  const xref = new XRefMock();
  const acroForm = new Dict(xref);
  const acroFormRef = Ref.get(1, 0);
  const hasXfa = false;
  const hasXfaDatasetsEntry = false;
  const xfaDatasetsRef = null;
  const needAppearances = true;
  const newRefs = [];

  await updateAcroform({
    xref,
    acroForm,
    acroFormRef,
    hasXfa,
    hasXfaDatasetsEntry,
    xfaDatasetsRef,
    needAppearances,
    newRefs,
  });

  const expected = true;
  const actual = newRefs.some(ref => ref.data.includes("/NeedAppearances true"));
  expect(actual).toBe(expected);
});
