it("should render Russian characters correctly", async () => {
  const { PDFDocumentProxy } = await import("../../src/display/api.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");
  const pdfUrl = `${getCrossOriginHostname()}/${TEST_PDFS_PATH}/test.pdf`;
  const params = buildGetDocumentParams(pdfUrl);
  const pdfDoc = await PDFDocumentProxy.load(params);
  const firstPage = await pdfDoc.getPage(1);
  const viewport = firstPage.getViewport({ scale: 1 });
  const renderTask = firstPage.render({
    viewport,
    intent: "display",
  });
  await renderTask.promise;
  const textContent = await firstPage.getTextContent();
  const expectedText = "Some russian characters: а, б, в";
  const actualText = textContent.items.map((item) => item.str).join("");
  expect(actualText).toContain(expectedText);
});
