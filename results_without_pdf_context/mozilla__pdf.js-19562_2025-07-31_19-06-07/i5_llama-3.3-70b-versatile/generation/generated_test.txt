it("should render russian characters correctly", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { PDFFetchStream } = await import("../../src/display/fetch_stream.js");
  const { CMap } = await import("../../src/core/cmap.js");
  const { getGlyphMapForStandardFonts } = await import("../../src/core/standard_fonts.js");

  const pdfUrl = "https://github.com/user-attachments/files/18943576/test.pdf";
  const pdf = await PDFDocument.load({ url: pdfUrl, cMapUrl: "https://cdn.jsdelivr.net/npm/pdfjs-dist@2.16.105/build/generic/web/cmaps/", cMapPacked: true });
  const page = await pdf.getPage(1);
  const textContent = await page.getTextContent();
  const expectedText = "Русские символы";
  const actualText = textContent.items.map(item => item.str).join("");

  expect(actualText).toContain(expectedText);
});
