it("should render Russian characters correctly", async () => {
  const { PDFDocumentProxy } = await import("../../src/display/api.js");
  const { TestPdfsServer } = await import("./test_utils.js");
  const server = new TestPdfsServer();
  const pdfUrl = `${server.url}/test.pdf`;
  const pdfDocument = await PDFDocumentProxy.load({ url: pdfUrl });
  const page = await pdfDocument.getPage(1);
  const viewport = page.getViewport({ scale: 1 });
  const renderTask = page.render({
    viewport,
    intent: "display",
  });
  const renderedPage = await renderTask.promise;
  const textContent = await page.getTextContent();
  const expectedText = "Привет";
  const actualText = textContent.items[0].str;
  expect(actualText).toBe(expectedText);
});
