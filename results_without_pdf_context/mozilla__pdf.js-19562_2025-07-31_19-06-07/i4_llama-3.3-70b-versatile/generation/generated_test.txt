it("should render russian characters correctly", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { getGlyphMapForStandardFonts } = await import("../../src/core/standard_fonts.js");
  const { fetchData } = await import("../../src/display/display_utils.js");
  const pdfData = await fetchData("https://github.com/user-attachments/files/18943576/test.pdf");
  const pdfDoc = await PDFDocument.load(pdfData);
  const firstPage = await pdfDoc.getPage(1);
  const textContent = await firstPage.getTextContent();
  const expectedText = "some russian text";
  const actualText = textContent.items.map(item => item.str).join("");
  expect(actualText).toContain(expectedText);
});
