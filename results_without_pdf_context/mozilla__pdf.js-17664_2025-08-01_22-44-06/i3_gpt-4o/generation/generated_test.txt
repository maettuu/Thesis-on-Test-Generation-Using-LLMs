it("should update highlighting position without lag", async () => {
  const { FreeOutliner } = await import("../../src/display/editor/outliner.js");
  const initialPoint = { x: 10, y: 10 };
  const box = [0, 0, 100, 100];
  const scaleFactor = 1;
  const thickness = 1;
  const isLTR = true;
  const outliner = new FreeOutliner(initialPoint, box, scaleFactor, thickness, isLTR);

  const newPoint = { x: 50, y: 50 };
  outliner.add(newPoint);

  const expectedPath = `M0.1 0.1 L0.5 0.5 L0.5 0.5 L0.1 0.1 Z`;
  const actualPath = outliner.toSVGPath();

  expect(actualPath).toBe(expectedPath);
});
