it("should update highlighting position without lag", async () => {
  const { FreeOutliner } = await import("../../src/display/editor/outliner.js");
  const outliner = new FreeOutliner();
  outliner.#box = [0, 0, 100, 100];
  
  const initialPoint = { x: 10, y: 10 };
  outliner.add(initialPoint);
  
  const expectedCoords = [
    (initialPoint.x - outliner.#box[0]) / outliner.#box[2],
    (initialPoint.y - outliner.#box[1]) / outliner.#box[3],
  ];
  
  const actualCoords = outliner.#getLastCoords().slice(0, 2);
  
  expect(actualCoords).toEqual(expectedCoords);
});
