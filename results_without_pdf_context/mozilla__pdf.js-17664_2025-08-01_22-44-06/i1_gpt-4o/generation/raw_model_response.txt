it("should update highlighting coordinates correctly", async () => {
  const { FreeOutliner } = await import("../../src/display/editor/outliner.js");
  const initialPoint = { x: 10, y: 10 };
  const box = [0, 0, 100, 100];
  const scaleFactor = 1;
  const thickness = 1;
  const isLTR = true;
  const outliner = new FreeOutliner(initialPoint, box, scaleFactor, thickness, isLTR);

  outliner.add({ x: 20, y: 20 });
  outliner.add({ x: 30, y: 30 });

  const expectedSVGPath = "M0.1 0.1 L0.2 0.2 L0.3 0.3 L0.2 0.2 L0.1 0.1 Z";
  const actualSVGPath = outliner.toSVGPath();

  expect(actualSVGPath).toBe(expectedSVGPath);
});
