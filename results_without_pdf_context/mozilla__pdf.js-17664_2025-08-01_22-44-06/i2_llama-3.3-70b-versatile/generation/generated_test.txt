it("should end a free highlight where the mouse pointer is", async () => {
  const { FreeOutliner } = await import("../../src/display/editor/outliner.js");
  const outliner = new FreeOutliner({ x: 10, y: 10 }, [0, 0, 100, 100], 1, 1, true);
  outliner.add({ x: 20, y: 20 });
  outliner.add({ x: 30, y: 30 });
  const outline = outliner.getOutlines();
  const points = outline.points;
  const lastPoint = points[points.length - 1];
  const expectedLastPoint = { x: 30, y: 30 };
  expect(lastPoint).toEqual(expectedLastPoint);
});
