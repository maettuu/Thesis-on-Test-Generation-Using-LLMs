it("should ensure free highlighting ends where the mouse pointer is", async () => {
  const { FreeOutliner } = await import("../../src/display/editor/outliner.js");
  
  const box = [0, 0, 100, 100];
  const scaleFactor = 1;
  const thickness = 1;
  const isLTR = true;
  const initialPoint = { x: 10, y: 10 };
  const outliner = new FreeOutliner(initialPoint, box, scaleFactor, thickness, isLTR);

  outliner.add({ x: 20, y: 20 });
  outliner.add({ x: 30, y: 30 });
  outliner.add({ x: 40, y: 40 });

  const expectedLastCoords = [40, 40];
  const actualLastCoords = [outliner.#lastX, outliner.#lastY];

  expect(actualLastCoords).toEqual(expectedLastCoords);
});
