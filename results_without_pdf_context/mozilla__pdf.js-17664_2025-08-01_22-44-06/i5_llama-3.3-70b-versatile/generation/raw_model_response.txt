it("should not lag behind the mouse cursor when using free highlighting in PDF", async () => {
  const { PDFDocument, PDFPageProxy } = await import("../../src/display/api.js");
  const { Outliner } = await import("../../src/display/editor/outliner.js");
  const { PDFDataRangeTransport } = await import("../../src/display/api.js");

  const pdfDoc = await PDFDocument.load(await (await import("../../src/shared/util.js")).stringToBytes("your_pdf_file.pdf"));
  const pdfPage = await pdfDoc.getPage(1);
  const pdfPageProxy = new PDFPageProxy(pdfDoc, pdfPage);
  const outliner = new Outliner(pdfPageProxy);

  const lastX = 100;
  const lastY = 100;
  outliner.add({ x: lastX, y: lastY });

  const lastCoords = outliner.#getLastCoords();
  const expectedLastCoords = [(lastX + (outliner.#last.subarray(4, 6)[0] - outliner.#last.subarray(16, 18)[0]) / 2) / pdfPageProxy.getViewport(1).width, 
                                (lastY + (outliner.#last.subarray(4, 6)[1] - outliner.#last.subarray(16, 18)[1]) / 2) / pdfPageProxy.getViewport(1).height, 
                                (lastX + (outliner.#last.subarray(16, 18)[0] - outliner.#last.subarray(4, 6)[0]) / 2) / pdfPageProxy.getViewport(1).width, 
                                (lastY + (outliner.#last.subarray(16, 18)[1] - outliner.#last.subarray(4, 6)[1]) / 2) / pdfPageProxy.getViewport(1).height];

  expect(lastCoords).toEqual(expectedLastCoords);
});
