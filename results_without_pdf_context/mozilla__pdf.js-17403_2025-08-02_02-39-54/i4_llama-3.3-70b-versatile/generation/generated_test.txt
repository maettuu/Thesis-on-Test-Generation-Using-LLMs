it("should render a PDF with many pages without slowing down", async () => {
  const { PDFWorker } = await import("../../src/display/api.js");
  const worker = new PDFWorker();
  const pdf = await worker.loadPDF("path_to_large_pdf.pdf");
  const pages = pdf.numPages;
  const startTime = Date.now();
  for (let i = 1; i <= pages; i++) {
    await pdf.getPage(i);
  }
  const endTime = Date.now();
  const renderTime = endTime - startTime;
  expect(renderTime).toBeLessThan(5000); // 5 seconds
});
