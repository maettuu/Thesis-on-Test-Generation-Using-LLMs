it("should skip useless tasks in the worker to improve fast scrolling with scanned books", async () => {
  const { MessageHandler } = await import("../../src/shared/message_handler.js");
  const { PDFWorker } = await import("../../src/display/api.js");
  const worker = new PDFWorker();
  const messageHandler = new MessageHandler("source", "target", worker);
  const actionHandler = {
    renderPage: () => new Promise(resolve => {
      // Simulate a long-running task
      setTimeout(() => resolve(), 100);
    }),
  };
  messageHandler.actionHandler = actionHandler;
  const callbackId = 1;
  const streamId = 1;
  const data = {
    targetName: "source",
    action: "renderPage",
    callbackId,
    streamId,
  };
  const cancelData = {
    targetName: "source",
    stream: true,
    streamId,
  };
  messageHandler._onComObjOnMessage({ data });
  messageHandler._onComObjOnMessage({ data: cancelData });
  await new Promise(resolve => setTimeout(resolve, 150));
  const expected = 0;
  const actual = messageHandler.callbackCapabilities[callbackId];
  expect(actual).toBeUndefined();
});
