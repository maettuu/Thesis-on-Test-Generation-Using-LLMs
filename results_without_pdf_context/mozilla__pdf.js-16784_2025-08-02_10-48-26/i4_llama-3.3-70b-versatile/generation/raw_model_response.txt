it("should allow resizing of SVG images horizontally/vertically in pdf.js", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const imageManager = new ImageManager();
  const svgImage = new Blob([`
    <svg viewBox="0 0 100 100" width="100" height="100" xmlns="http://www.w3.org/2000/svg">
      <rect width="100" height="100" style="fill:red;"/>
    </svg>
  `], { type: "image/svg+xml" });
  const fileReader = new FileReader();
  const imagePromise = new Promise((resolve, reject) => {
    fileReader.onload = async () => {
      const url = fileReader.result;
      const imageElement = new Image();
      imageElement.src = url;
      await new Promise(resolve => imageElement.onload = resolve);
      const canvas = document.createElement("canvas");
      canvas.width = 50;
      canvas.height = 50;
      const ctx = canvas.getContext("2d");
      ctx.drawImage(imageElement, 0, 0, 50, 50);
      resolve(ctx.getImageData(0, 0, 1, 1).data);
    };
    fileReader.onerror = reject;
  });
  fileReader.readAsDataURL(svgImage);
  const imageData = await imagePromise;
  expect(imageData[0]).not.toBe(0);
});
