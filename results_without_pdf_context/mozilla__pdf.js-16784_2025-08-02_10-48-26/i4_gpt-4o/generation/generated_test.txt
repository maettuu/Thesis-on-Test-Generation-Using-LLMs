it("should allow SVG images to be resized horizontallyvertically", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const svgData = `data:image/svg+xml;charset=UTF-8,<svg viewBox="0 0 1 1" width="1" height="1" xmlns="http://www.w3.org/2000/svg"><rect width="1" height="1" style="fill:red;"/></svg>`;
  const imageManager = new ImageManager();
  const image = new Image();
  image.src = svgData;
  await image.decode();
  const canvas = new OffscreenCanvas(1, 3);
  const ctx = canvas.getContext("2d");
  ctx.drawImage(image, 0, 0, 1, 1, 0, 0, 1, 3);
  const isResized = new Uint32Array(ctx.getImageData(0, 0, 1, 1).data.buffer)[0] === 0;
  expect(isResized).toBe(true);
});
