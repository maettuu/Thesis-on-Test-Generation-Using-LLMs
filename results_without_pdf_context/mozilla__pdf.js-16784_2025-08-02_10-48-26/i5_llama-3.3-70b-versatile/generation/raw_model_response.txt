it("should resize SVG images horizontally/vertically in pdf.js", async () => {
  const { PDFDocument, PDFPageProxy } = await import("../../src/display/api.js");
  const { createValidAbsoluteUrl } = await import("../../src/shared/util.js");
  const url = createValidAbsoluteUrl("path/to/svg/image.svg");
  const pdfDoc = await PDFDocument.load(url);
  const page = await pdfDoc.getPage(1);
  const svgImage = await page.getOperatorList().then((operatorList) => {
    for (const op of operatorList) {
      if (op.type === "drawImage" && op.image.type === "image/svg+xml") {
        return op.image;
      }
    }
  });
  const originalWidth = svgImage.width;
  const originalHeight = svgImage.height;
  const newWidth = originalWidth * 2;
  const newHeight = originalHeight * 2;
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const imageManager = new ImageManager();
  await imageManager.resizeImage(svgImage, newWidth, newHeight);
  const resizedWidth = svgImage.width;
  const resizedHeight = svgImage.height;
  expect(resizedWidth).toBe(newWidth);
  expect(resizedHeight).toBe(newHeight);
});
