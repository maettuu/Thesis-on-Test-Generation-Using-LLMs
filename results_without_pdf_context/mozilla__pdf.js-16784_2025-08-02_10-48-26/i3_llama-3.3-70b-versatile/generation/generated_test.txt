it("should allow resizing of SVG images in pdfjs", async () => {
  const { AnnotationEditorUIManager, ImageManager } = await import("../../src/display/editor/tools.js");
  const { AnnotationEditorType } = await import("../../src/shared/util.js");
  const { createIdFactory } = await import("./test_utils.js");

  const idFactory = createIdFactory();
  const imageManager = new ImageManager();
  const annotationEditorUIManager = new AnnotationEditorUIManager(null, null, null, null);

  const svgImage = new Blob([`
    <svg xmlns="http://www.w3.org/2000/svg" width="100" height="100">
      <rect width="100" height="100" style="fill:rgb(0,0,255)" />
    </svg>
  `], { type: "image/svg+xml" });

  const image = await imageManager.getFromFile(svgImage);
  const svgUrl = image.svgUrl;

  const mustRemoveAspectRatioPromise = ImageManager._checkIfSVGFitsInCanvas();
  const mustRemoveAspectRatio = await mustRemoveAspectRatioPromise;

  if (mustRemoveAspectRatio) {
    const resizedSvgUrl = `${svgUrl}#svgView(preserveAspectRatio(none))`;
    expect(resizedSvgUrl).not.toBe(svgUrl);
  } else {
    expect(svgUrl).toBe(svgUrl);
  }
});
