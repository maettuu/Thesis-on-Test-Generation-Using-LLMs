```javascript
it("should allow SVG images to be resized horizontally/vertically", async () => {
  const { PDFDocumentProxy, getDocument } = await import("../../src/display/api.js");
  const { DOMCanvasFactory } = await import("../../src/display/display_utils.js");

  const loadingTask = getDocument({ url: "path/to/svg-containing-pdf.pdf" });
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);

  const canvasFactory = new DOMCanvasFactory();
  const viewport = page.getViewport({ scale: 1.0 });
  const canvasAndContext = canvasFactory.create(viewport.width, viewport.height);
  const renderContext = {
    canvasContext: canvasAndContext.context,
    viewport: viewport,
  };

  await page.render(renderContext).promise;

  const canvas = canvasAndContext.canvas;
  const initialWidth = canvas.width;
  const initialHeight = canvas.height;

  canvas.width = initialWidth * 2;
  canvas.height = initialHeight * 2;

  const resizedViewport = page.getViewport({ scale: 2.0 });
  const resizedRenderContext = {
    canvasContext: canvasAndContext.context,
    viewport: resizedViewport,
  };

  await page.render(resizedRenderContext).promise;

  const resizedWidth = canvas.width;
  const resizedHeight = canvas.height;

  expect(resizedWidth).toBeGreaterThan(initialWidth);
  expect(resizedHeight).toBeGreaterThan(initialHeight);
});
```
