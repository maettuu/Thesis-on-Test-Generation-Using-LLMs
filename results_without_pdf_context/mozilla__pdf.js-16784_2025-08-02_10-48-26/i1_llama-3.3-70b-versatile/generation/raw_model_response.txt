it("should allow resizing of SVG images horizontally/vertically in pdf.js", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const { Util } = await import("../../shared/util.js");
  const imageManager = new ImageManager();
  const svgImage = new Blob([`<svg width="100" height="100"><rect width="100" height="100" style="fill:red;"/></svg>`], { type: "image/svg+xml" });
  const imageData = await imageManager.#get("svgImage", svgImage);
  const imageElement = imageData.bitmap;
  imageElement.width = 200;
  imageElement.height = 200;
  const expectedWidth = 200;
  const expectedHeight = 200;
  const actualWidth = imageElement.width;
  const actualHeight = imageElement.height;
  Util.assert(actualWidth === expectedWidth, `Expected width to be ${expectedWidth}, but got ${actualWidth}`);
  Util.assert(actualHeight === expectedHeight, `Expected height to be ${expectedHeight}, but got ${actualHeight}`);
});
