it("should add a comment button to the floating toolbar", async () => {
  const { AnnotationEditorUIManager, HighlightToolbar } = await import("./toolbar.js");
  const { AnnotationEditorUIManager: Manager } = await import("../display/editor/tools.js");
  const uiManager = new Manager(null, null, null, null, null, null, null, null, null, null, null, null, null);
  const highlightToolbar = new HighlightToolbar(uiManager);
  const editToolbar = highlightToolbar.#render();
  const buttons = editToolbar.querySelector(".buttons");
  const commentButton = buttons.querySelector("[data-l10n-id='pdfjs-comment-floating-button-label']");
  expect(commentButton).not.toBeNull();
});
