it("should display highlight on cropped pdf when saving/printing", async () => {
  const { PDFDocument, PDFPageProxy } = await import("../../src/display/api.js");
  const { Util } = await import("../../shared/util.js");
  const { HighlightEditor } = await import("./highlight.js");

  const pdfDoc = await PDFDocument.load(await fetchData("path_to_cropped_pdf.pdf"));
  const page = await pdfDoc.getPage(1);
  const viewport = page.getViewport({ scale: 1 });
  const pdfPageProxy = new PDFPageProxy(page);

  const highlightEditor = new HighlightEditor({
    pageIndex: 0,
    page: pdfPageProxy,
    x: 0,
    y: 0,
    width: viewport.width,
    height: viewport.height,
    color: Util.makeHexColor(255, 240, 102),
    opacity: 1,
  });

  const serializedData = highlightEditor.serialize();
  const expectedQuadPoints = serializedData.quadPoints;

  const actualQuadPoints = highlightEditor.#serializeBoxes([0, 0]);
  expect(actualQuadPoints).toEqual(expectedQuadPoints);
});
