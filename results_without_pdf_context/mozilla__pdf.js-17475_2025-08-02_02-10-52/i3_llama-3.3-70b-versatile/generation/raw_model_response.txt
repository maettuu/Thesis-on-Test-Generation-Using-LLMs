it("should correctly serialize highlight annotations when saving or printing", async () => {
  const { HighlightEditor } = await import("../../src/display/editor/highlight.js");
  const { PDFWorker } = await import("../../src/display/api.js");
  const { Util } = await import("../../src/shared/util.js");
  const worker = new PDFWorker();
  const pdf = await worker.loadPDF("path/to/example.pdf");
  const page = pdf.getPage(1);
  const editor = new HighlightEditor({
    pageIndex: 0,
    page,
    boxes: [{ x: 0.1, y: 0.1, width: 0.2, height: 0.2 }],
    color: "#fff066",
    opacity: 1,
  });
  const serialized = editor.serialize();
  const expectedQuadPoints = [
    72, 648, 72, 648, 144, 648, 144, 648,
  ];
  expect(serialized.quadPoints).toEqual(expectedQuadPoints);
});
