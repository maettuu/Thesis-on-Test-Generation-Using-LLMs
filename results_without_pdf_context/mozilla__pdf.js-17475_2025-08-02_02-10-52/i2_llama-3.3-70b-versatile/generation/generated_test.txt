it("should include the crop box when savingprinting highlighted text", async () => {
  const { PDFDocument, PDFPage } = await import("../../src/core/document.js");
  const { AnnotationEditor, AnnotationEditorType } = await import("../../src/display/editor/editor.js");
  const { Util } = await import("../../shared/util.js");

  const pdfDoc = await PDFDocument.load(await (await import("../../src/display/api.js")).build());
  const page = pdfDoc.getPage(0);
  const viewport = page.getViewport({ scale: 1 });
  const annotationEditor = new (await import("../../src/display/editor/highlight.js")).HighlightEditor({
    pageIndex: 0,
    page,
    viewport,
    color: [1, 0, 0],
    opacity: 1,
    boxes: [{ x: 0.1, y: 0.1, width: 0.1, height: 0.1 }],
  });

  const serializedData = annotationEditor.serialize();
  const expectedQuadPoints = [10, 710, 20, 710, 20, 700, 10, 700];
  const actualQuadPoints = serializedData.quadPoints;

  expect(actualQuadPoints).toEqual(expectedQuadPoints);
});
