it("should not throw TypeError when touch ends", async () => {
  const { TouchManager } = await import("../../src/display/touch_manager.js");
  const container = document.createElement("div");
  const touchManager = new TouchManager({
    container,
    signal: new AbortController().signal,
  });

  const touchStartEvent = new TouchEvent("touchstart", {
    touches: [
      new Touch({ identifier: 1, target: container, screenX: 0, screenY: 0 }),
      new Touch({ identifier: 2, target: container, screenX: 10, screenY: 10 }),
    ],
  });

  const touchEndEvent = new TouchEvent("touchend", {
    touches: [new Touch({ identifier: 1, target: container, screenX: 0, screenY: 0 })],
  });

  container.dispatchEvent(touchStartEvent);
  let errorOccurred = false;
  try {
    container.dispatchEvent(touchEndEvent);
  } catch (e) {
    errorOccurred = true;
  }

  expect(errorOccurred).toBe(false);
});
