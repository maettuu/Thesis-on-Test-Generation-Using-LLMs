it("should not throw TypeError when drawing annotations", async () => {
  const { PDFDocument, PDFPageProxy } = await import("../../src/display/api.js");
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { TouchManager } = await import("../../src/display/touch_manager.js");

  const pdfDocument = await PDFDocument.load(await (await import("../../src/display/node_utils.js")).fetchData("path_to_your_pdf_file.pdf"));
  const pdfPage = pdfDocument.getPage(1);
  const pageProxy = new PDFPageProxy(pdfPage);
  const annotationEditor = new AnnotationEditorUIManager(pageProxy);
  const touchManager = new TouchManager();

  let error = null;
  try {
    annotationEditor.setMode("DRAW");
    touchManager.onTouchStart({ touches: [{ clientX: 10, clientY: 10 }] });
    touchManager.onTouchMove({ touches: [{ clientX: 20, clientY: 20 }] });
    touchManager.onTouchEnd({ touches: [] });
  } catch (e) {
    error = e;
  }

  expect(error).toBeNull();
});
