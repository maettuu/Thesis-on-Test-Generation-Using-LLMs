it("should prevent null pointer exception when thistouchMoveAC is null", async () => {
  const { TouchManager } = await import("../../src/display/touch_manager.js");
  const touchManager = new TouchManager({
    container: document.createElement("div"),
    isPinchingDisabled: () => false,
    isPinchingStopped: () => false,
    onPinchStart: () => {},
    onPinching: () => {},
    onPinchEnd: () => {},
    signal: new AbortController().signal,
  });
  touchManager.#touchMoveAC = null;
  expect(() => touchManager.#onTouchEnd({ touches: [] })).not.toThrow();
});
