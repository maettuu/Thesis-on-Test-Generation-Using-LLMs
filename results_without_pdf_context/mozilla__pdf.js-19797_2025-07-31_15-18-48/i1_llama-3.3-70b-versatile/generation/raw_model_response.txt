it("should not throw TypeError when updating mode and drawing", async () => {
  const { AnnotationEditorUIManager, TouchManager } = await import("../../src/display/editor/tools.js");
  const { PDFDocument, PDFPage } = await import("../../src/core/document.js");
  const pdfDocument = await PDFDocument.load(await (await import("../../src/display/api.js")).build());
  const pdfPage = pdfDocument.getPage(1);
  const container = { addEventListener: () => {}, removeEventListener: () => {} };
  const viewer = { getPages: () => [pdfPage] };
  const annotationEditorUIManager = new AnnotationEditorUIManager(container, viewer, null, null, null, pdfDocument, null, null, false, false, false, null, null, true);
  const touchManager = new TouchManager({ container, signal: new AbortController().signal });
  annotationEditorUIManager.updateMode(AnnotationEditorType.INK);
  touchManager.#onTouchEnd({ touches: [] });
  expect(() => annotationEditorUIManager.updateMode(AnnotationEditorType.NONE)).not.toThrow(TypeError);
});
