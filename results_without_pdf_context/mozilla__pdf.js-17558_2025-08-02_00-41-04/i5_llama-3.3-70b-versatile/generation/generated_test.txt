it("should render emojis in PDFs", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { createDefaultAppearance } = await import("../../src/core/default_appearance.js");
  const { stringToUTF16String } = await import("../../src/core/core_utils.js");

  const pdfDoc = await PDFDocument.load(await (await import("../../src/display/fetch_stream.js")).PDFFetchStream.fromUrl("path_to_pdf_with_emojis.pdf"));
  const page = pdfDoc.getPage(1);
  const annotations = page.getAnnotations();
  const freeTextAnnotation = annotations[0];
  const appearance = createDefaultAppearance(freeTextAnnotation);
  const text = stringToUTF16String(freeTextAnnotation.getContents());
  const expectedText = "üëç Hello World üëç";

  expect(text).toBe(expectedText);
});
