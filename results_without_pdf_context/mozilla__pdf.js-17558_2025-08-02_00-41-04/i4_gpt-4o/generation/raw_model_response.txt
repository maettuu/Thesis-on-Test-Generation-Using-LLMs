it("should render emojis in FreeTextAnnotations correctly", async () => {
  const { FreeTextAnnotation } = await import("../../src/core/annotation.js");
  const { Dict, Name } = await import("../../src/core/primitives.js");
  const { stringToUTF16String } = await import("../../src/core/core_utils.js");

  const xref = {};
  const dict = new Dict(xref);
  dict.set("Subtype", Name.get("FreeText"));
  dict.set("Contents", stringToUTF16String("Hello üåç"));

  const params = {
    dict,
    xref,
    evaluatorOptions: { isOffscreenCanvasSupported: true },
  };

  const annotation = new FreeTextAnnotation(params);
  const expected = "Hello üåç";
  const actual = annotation.data.textContent.join("");

  expect(actual).toBe(expected);
});
