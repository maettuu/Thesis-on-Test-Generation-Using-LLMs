it("should render PDF with non-embedded HelveticaLTStd-Bold font correctly", async () => {
  const { PDFDocument } = await import("../../src/display/api.js");
  const { getStdFontMap } = await import("./standard_fonts.js");
  const pdfData = await import("../../src/display/node_utils.js").then(({ fetchData }) => fetchData("https://github.com/user-attachments/files/18172261/Tax3921.pdf"));
  const pdfDoc = await PDFDocument.load(pdfData);
  const firstPage = await pdfDoc.getPage(1);
  const renderTask = firstPage.render({
    viewport: firstPage.getViewport({ scale: 1 }),
    intent: "print",
  });
  const stdFontMap = getStdFontMap();
  stdFontMap["HelveticaLTStd-Bold"] = "Helvetica-Bold";
  await renderTask.promise;
  const actualRender = renderTask.canvasContext.canvas.toBuffer();
  const expectedRender = await import("../../src/display/node_utils.js").then(({ fetchData }) => fetchData("https://github.com/user-attachments/assets/ccdedfa6-532b-4a59-a7bc-0f3df4f207a6"));
  expect(actualRender).toEqual(expectedRender);
});
