it("should render the pdf with font fallback", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { getStdFontMap } = await import("../../src/core/standard_fonts.js");
  const { fetchData } = await import("../../src/display/node_utils.js");

  const pdfUrl = "/tmp/Tax3921.pdf";
  const pdfData = await fetchData(pdfUrl);
  const pdfDoc = await PDFDocument.load(pdfData);

  const firstPage = await pdfDoc.getPage(1);
  const renderTask = firstPage.render({
    viewport: firstPage.getViewport({ scale: 1 }),
    intent: "print",
  });

  await renderTask.promise;

  const stdFontMap = getStdFontMap();
  const fontName = "HelveticaLTStd-Bold";
  const expectedFontName = "Helvetica-Bold";

  expect(stdFontMap[fontName]).toBe(expectedFontName);
});
