it("should render font correctly when font fallback is required", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const { fetchData } = await import("../../src/display/node_utils.js");
  const pdfUrl = "/tmp/Tax3921.pdf";
  const pdfDocument = await getDocument({ url: pdfUrl, cMapUrl: "https://cdn.jsdelivr.net/npm/pdfjs-dist@2.16.105/build/generic/web/cmaps/", cMapPacked: true }).promise;
  const pdfPage = await pdfDocument.getPage(1);
  const viewport = pdfPage.getViewport({ scale: 1 });
  const renderTask = pdfPage.render({
    viewport,
    intent: "display",
  });
  await renderTask.promise;
  const expectedText = "Blue text should be visible";
  const actualText = await pdfPage.getTextContent();
  expect(actualText.items.some(item => item.str.includes(expectedText))).toBe(true);
});
