it("should correctly fallback to HelveticaBold for HelveticaLTStdBold", async () => {
  const { getStandardFontName } = await import("../../src/core/standard_fonts.js");
  const fontName = "HelveticaLTStd-Bold";
  const expected = "Helvetica-Bold";
  const actual = getStandardFontName(fontName);
  expect(actual).toEqual(expected);
});
