it("should reset the canvas filter to none if its not none or empty", async () => {
  const { DOMCanvasFactory } = await import("../../src/display/display_utils.js");
  const canvasFactory = new DOMCanvasFactory();
  const canvasAndCtx = canvasFactory.create(100, 100);
  const ctx = canvasAndCtx.context;

  ctx.filter = "blur(5px)";
  const expectedFilter = "none";

  const { resetCtxToDefault } = await import("../../src/display/canvas.js");
  resetCtxToDefault(ctx);

  const actualFilter = ctx.filter;
  expect(actualFilter).toBe(expectedFilter);
});
