it("should render poster PDF without being slow on Firefox", async () => {
  const { PDFDocument } = await import("../../src/display/api.js");
  const { createIdFactory } = await import("./test_utils.js");
  const pdf = await PDFDocument.load(await fetch("path_to_poster_pdf.pdf").then(res => res.arrayBuffer()));
  const idFactory = createIdFactory();
  const canvas = document.createElement("canvas");
  const ctx = canvas.getContext("2d");
  const canvasGraphics = new (await import("../../src/display/canvas.js")).CanvasGraphics(ctx, {}, {}, {}, {}, {});
  const page = await pdf.getPage(1);
  await page.render({ canvasContext: ctx, viewport: page.getViewport(1) });
  expect(canvas.toDataURL()).not.toBeNull();
});
