it("should render poster PDF without significant delay", async () => {
  const { PDFDocumentProxy, PDFPageProxy, PDFWorkerUtil } = await import("../../src/display/api.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");
  const { PageViewport } = await import("../../src/display/display_utils.js");
  const params = buildGetDocumentParams(`${TEST_PDFS_PATH}/poster.pdf`);
  const pdfDoc = await PDFDocumentProxy.load(params);
  const pdfPage = await pdfDoc.getPage(1);
  const viewport = new PageViewport({ pdfPage, scale: 1 });
  const renderTask = pdfPage.render({
    viewport,
    canvasContext: null,
    renderInteractiveForms: false,
    continueCallback: null,
  });
  const startTime = performance.now();
  await renderTask.promise;
  const endTime = performance.now();
  const renderTime = endTime - startTime;
  expect(renderTime).toBeLessThan(500);
});
