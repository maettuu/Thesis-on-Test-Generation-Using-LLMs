it("should render poster PDF quickly in Firefox", async () => {
  const { PDFDocument, PDFPageProxy } = await import("../../src/display/api.js");
  const { PDFFetchStream } = await import("../../src/display/fetch_stream.js");
  const { getFilenameFromUrl } = await import("../../src/display/display_utils.js");
  const { TEST_PDFS_PATH } = await import("./test_utils.js");

  const pdfUrl = `${TEST_PDFS_PATH}poster.pdf`;
  const pdfFilename = getFilenameFromUrl(pdfUrl);
  const pdfDocument = await PDFDocument.load({ url: pdfUrl, cMapUrl: "", cMapPacked: true, fetch: PDFFetchStream });
  const pdfPage = await pdfDocument.getPage(1);
  const renderTask = pdfPage.render({ viewport: pdfPage.getViewport({ scale: 1 }) });
  const startTime = Date.now();
  await renderTask.promise;
  const endTime = Date.now();
  const renderTime = endTime - startTime;
  const expectedRenderTime = 1000; // 1 second
  if (renderTime > expectedRenderTime) {
    throw new Error(`Rendering poster PDF took too long: ${renderTime}ms`);
  }
});
