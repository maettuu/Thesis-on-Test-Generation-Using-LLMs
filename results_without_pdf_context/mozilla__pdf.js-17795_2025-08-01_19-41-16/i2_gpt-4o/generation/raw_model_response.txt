it("should not render corrupted inlined images", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");

  const loadingTask = getDocument(buildGetDocumentParams("image1.pdf"));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);

  const operatorList = await pdfPage.getOperatorList();
  const imageOps = operatorList.fnArray.filter(fn => fn === OPS.paintInlineImageXObject);

  const expectedImageOpsCount = 0;
  const actualImageOpsCount = imageOps.length;

  expect(actualImageOpsCount).toBe(expectedImageOpsCount);
});
