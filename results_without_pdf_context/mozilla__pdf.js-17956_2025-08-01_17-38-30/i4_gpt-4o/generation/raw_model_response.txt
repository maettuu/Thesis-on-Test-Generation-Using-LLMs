```javascript
it("should not log 'OpenJPEG: Unknown format' to the console", async () => {
  const { JpxImage } = await import("../../src/core/jpx.js");
  const { Console } = await import("console");
  const console = new Console(process.stdout, process.stderr);
  const originalError = console.error;
  let logOutput = "";
  console.error = (msg) => {
    logOutput += msg;
  };

  try {
    JpxImage.decode(new Uint8Array([0, 1, 2, 3]), false);
  } catch (e) {
    // Expected to throw JpxError
  }

  console.error = originalError;
  const expectedLogOutput = "";
  expect(logOutput).toBe(expectedLogOutput);
});
```
