it("should throw an exception with the error message returned by openjpeg", async () => {
  const { JpxImage } = await import("../../src/core/jpx.js");
  const data = new Uint8Array([0x00, 0x00, 0x00, 0x00]); // Invalid JPX data
  const ignoreColorSpace = false;
  let error;
  try {
    await JpxImage.decode(data, ignoreColorSpace);
  } catch (e) {
    error = e;
  }
  expect(error).toBeInstanceOf(Error);
  expect(error.message).not.toBe("JPX decode failed");
});
