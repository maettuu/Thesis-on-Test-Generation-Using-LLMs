it("should not throw JpxError with 'JPX decode failed' message when OpenJPEG decode returns null", async () => {
  const { JpxImage } = await import("../../src/core/jpx.js");
  const data = new Uint8Array();
  const ignoreColorSpace = false;
  const errorMessage = "JPX decode failed";
  try {
    JpxImage.decode(data, ignoreColorSpace);
  } catch (error) {
    if (error.message === errorMessage) {
      throw new Error("Test failed: JpxError was thrown with 'JPX decode failed' message");
    }
  }
});
