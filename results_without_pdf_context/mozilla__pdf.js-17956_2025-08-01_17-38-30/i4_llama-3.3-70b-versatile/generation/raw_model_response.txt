```javascript
it("should not log 'OpenJPEG: Unknown format' when decoding fails", async () => {
  const { JpxImage } = await import("../../src/core/jpx.js");
  const consoleSpy = jest.spyOn(console, 'log');
  const data = new Uint8Array([0xff, 0x4f, 0xff, 0x51]); // Invalid JPX data
  try {
    JpxImage.decode(data, false);
  } catch (error) {
    expect(consoleSpy).not.toHaveBeenCalledWith('OpenJPEG: Unknown format');
  }
});
```
