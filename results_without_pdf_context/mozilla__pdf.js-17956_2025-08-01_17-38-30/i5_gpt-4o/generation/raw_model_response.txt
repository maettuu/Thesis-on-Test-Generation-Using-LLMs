it("should not log 'OpenJPEG: Unknown format' when decoding JPX", async () => {
  const { JpxImage } = await import("../../src/core/jpx.js");
  const { JpxError } = await import("../../src/shared/util.js");

  const data = new Uint8Array([0, 1, 2, 3]); // Example invalid JPX data
  const ignoreColorSpace = false;
  let errorMessage = "";

  try {
    JpxImage.decode(data, ignoreColorSpace);
  } catch (error) {
    if (error instanceof JpxError) {
      errorMessage = error.message;
    }
  }

  const expectedMessage = "JPX decode failed";
  expect(errorMessage).toBe(expectedMessage);
});
