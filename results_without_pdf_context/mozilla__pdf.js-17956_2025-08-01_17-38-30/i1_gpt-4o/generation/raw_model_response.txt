```javascript
it("should not log 'OpenJPEG: Unknown format' to the console", async () => {
  const { JpxImage } = await import("../../src/core/jpx.js");
  const consoleSpy = jest.spyOn(console, 'log').mockImplementation(() => {});
  const data = new Uint8Array([0, 1, 2, 3]); // Example data
  try {
    JpxImage.decode(data, false);
  } catch (e) {
    // Expected to throw JpxError
  }
  expect(consoleSpy).not.toHaveBeenCalledWith(expect.stringContaining("OpenJPEG: Unknown format"));
  consoleSpy.mockRestore();
});
```
