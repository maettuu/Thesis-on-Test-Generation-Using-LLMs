it("should not log OpenJPEG Unknown format to the console when decoding JPX image", async () => {
  const { JpxImage } = await import("../../src/core/jpx.js");
  const consoleSpy = jest.spyOn(console, 'log');
  const imageData = new Uint8Array([0x00, 0x00, 0x00, 0x0C, 0x6A, 0x50, 0x20, 0x20, 0x0D, 0x0A, 0x87, 0x0A]);
  try {
    JpxImage.decode(imageData, true);
  } catch (error) {
    expect(consoleSpy).not.toHaveBeenCalledWith('OpenJPEG: Unknown format');
  }
});
