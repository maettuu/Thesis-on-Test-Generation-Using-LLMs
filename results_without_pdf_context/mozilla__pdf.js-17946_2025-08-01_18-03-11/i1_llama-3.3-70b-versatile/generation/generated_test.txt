it("should render large images in PDF", async () => {
  const pdfUrl = "https://github.com/mozilla/pdf.js/files/13391396/example.pdf";
  const pdfDocument = await import("../../src/display/api.js");
  const pdf = await pdfDocument.PDFDocumentLoadingTask.fromURL(pdfUrl, null);
  const page = await pdf.getPage(1);
  const image = await page.getImage(0);
  const imageData = await image.getImageBytes();
  expect(imageData).not.toBeNull();
});
