it("should render large images in PDF", async () => {
  const { PDFDocument } = await import("../../src/display/api.js");
  const { fetch } = await import("../../src/shared/util.js");
  const pdfUrl = "https://github.com/mozilla/pdf.js/files/13391396/example.pdf";
  const pdfDoc = await PDFDocument.load(await fetch(pdfUrl));
  const firstPage = await pdfDoc.getPage(1);
  const operatorList = await firstPage.getOperatorList();
  const renderTask = firstPage.render({
    viewport: firstPage.getViewport({ scale: 1 }),
    intent: "display",
  });
  const imageRendered = renderTask.promise.then(() => true).catch(() => false);
  expect(await imageRendered).toBe(true);
});
