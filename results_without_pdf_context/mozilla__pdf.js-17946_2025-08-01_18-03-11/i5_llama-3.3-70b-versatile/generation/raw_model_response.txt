it("should render large images in PDF", async () => {
  const { PDFDocument } = await import("../../src/display/api.js");
  const pdf = await PDFDocument.load(await (await import("../../src/display/fetch_stream.js")).PDFFetchStream.fetch("https://github.com/mozilla/pdf.js/files/13391396/example.pdf"));
  const page = await pdf.getPage(1);
  const image = await page.getImage(0);
  const expectedWidth = 1650;
  const expectedHeight = 2338;
  const actualWidth = image.width;
  const actualHeight = image.height;
  expect(actualWidth).toBe(expectedWidth);
  expect(actualHeight).toBe(expectedHeight);
});
