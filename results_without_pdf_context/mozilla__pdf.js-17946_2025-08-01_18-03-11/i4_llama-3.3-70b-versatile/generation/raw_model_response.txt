it("should render large images in PDF that other renderers do", async () => {
  const { PDFDocument } = await import("../../src/display/api.js");
  const { JpxImage } = await import("../../src/core/jpx.js");
  const pdf = await PDFDocument.load(await fetch("example.pdf").then(res => res.arrayBuffer()));
  const page = await pdf.getPage(1);
  const operatorList = await page.getOperatorList();
  const image = operatorList.items.find(item => item instanceof JpxImage);
  const imageData = await image.getImageBytes(1000000);
  const expectedWidth = 1654;
  const expectedHeight = 2339;
  const actualWidth = image.width;
  const actualHeight = image.height;
  expect(actualWidth).toBe(expectedWidth);
  expect(actualHeight).toBe(expectedHeight);
});
