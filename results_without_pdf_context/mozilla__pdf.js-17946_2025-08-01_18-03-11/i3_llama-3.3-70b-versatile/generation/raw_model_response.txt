it("should render large images in PDF that other renderers do", async () => {
  const pdfUrl = "https://github.com/mozilla/pdf.js/files/13391396/example.pdf";
  const pdfDocument = await fetch(pdfUrl).then((response) => response.arrayBuffer());
  const pdf = await import("../../src/display/api.js");
  const pdfDoc = await pdf.PDFDocumentLoadingTask.fromBytes(pdfDocument, null);
  const firstPage = await pdfDoc.getPage(1);
  const image = firstPage.getOperatorList().getItems().find((item) => item.op === "Do");
  const imageStream = await firstPage.getOperatorList().getContext().streamForImage(image.args[0]);
  const imageData = await imageStream.getBytes();
  const jpxImage = await import("../../src/core/jpx.js").then((module) => module.JpxImage);
  const decodedImage = jpxImage.decode(imageData, false);
  expect(decodedImage.width).toBeGreaterThan(0);
  expect(decodedImage.height).toBeGreaterThan(0);
});
