it("should render large images in PDF using JPXDecode", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");
  const pdfUrl = "https://github.com/mozilla/pdf.js/files/13391396/example.pdf";
  const loadingTask = PDFDocument.load(buildGetDocumentParams(pdfUrl));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const operatorList = await page.getOperatorList();
  const jpxImages = operatorList.fnArray.filter(
    (fn) => fn === OPS.paintJpegXObject
  );
  expect(jpxImages.length).toBeGreaterThan(0);
});
