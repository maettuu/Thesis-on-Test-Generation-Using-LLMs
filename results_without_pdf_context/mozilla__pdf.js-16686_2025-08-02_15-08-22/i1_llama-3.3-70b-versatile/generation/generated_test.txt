it("should add a SVG image with a blob url", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const imageManager = new ImageManager();
  const svgBlob = new Blob(['<svg xmlns="http://www.w3.org/2000/svg" width="100" height="100"><rect width="100" height="100" style="fill:rgb(0,0,255)" /></svg>'], {type : 'image/svg+xml'});
  const key = await imageManager.#get('svg-image', svgBlob);
  expect(key).not.toBeNull();
  expect(key.bitmap).not.toBeNull();
  expect(key.isSvg).toBe(true);
});
