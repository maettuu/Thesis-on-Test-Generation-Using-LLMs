it("should apply transform correctly to points and bezier curves", async () => {
  const { Util } = await import("../../src/shared/util.js");
  const points = [1, 2, 3, 4];
  const bezier = [5, 6, 7, 8, 9, 10];
  const transform = [2, 0, 0, 2, 10, 20];
  const expectedPoints = [2 * 1 + 0 * 2 + 10, 2 * 2 + 0 * 1 + 20, 2 * 3 + 0 * 4 + 10, 2 * 4 + 0 * 3 + 20];
  const expectedBezier = [2 * 5 + 0 * 6 + 10, 2 * 6 + 0 * 5 + 20, 2 * 7 + 0 * 8 + 10, 2 * 8 + 0 * 7 + 20, 2 * 9 + 0 * 10 + 10, 2 * 10 + 0 * 9 + 20];
  Util.applyTransform(points, transform, 0);
  Util.applyTransformToBezier(bezier, transform, 0);
  expect(points).toEqual(expectedPoints);
  expect(bezier).toEqual(expectedBezier);
});
