it("should apply transformations correctly to command arguments", async () => {
  const { Util } = await import("../../src/shared/util.js");
  const { Commands } = await import("../../src/core/font_renderer.js");

  const commands = new Commands();
  commands.currentTransform = [1, 0, 0, 1, 10, 20];
  const args = [1, 2, 3, 4];
  const expectedArgs = [11, 22, 13, 24];

  commands.add("testCmd", args);

  const actualArgs = commands.cmds[0].slice(7).split(" ").map(Number);
  expect(actualArgs).toEqual(expectedArgs);
});
