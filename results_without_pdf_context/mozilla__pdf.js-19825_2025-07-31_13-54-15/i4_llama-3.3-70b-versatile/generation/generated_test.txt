it("should apply transform correctly to points in font rendering", async () => {
  const { Util } = await import("../../src/shared/util.js");
  const points = [10, 20, 30, 40];
  const transform = [2, 0, 0, 2, 10, 20];
  const expectedPoints = [50, 60, 70, 80];
  Util.applyTransform(points, transform, 0);
  Util.applyTransform(points, transform, 2);
  expect(points).toEqual(expectedPoints);
});
