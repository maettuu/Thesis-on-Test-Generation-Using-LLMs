it("should ignore invalid PDF operators like 'toString'", async () => {
  const { EvaluatorPreprocessor } = await import("../../src/core/evaluator.js");
  const { Lexer } = await import("../../src/core/parser.js");
  const { StringStream } = await import("../../src/core/stream.js");

  const stream = new StringStream("q 1 0 0 1 0 0 cm toString Q");
  const lexer = new Lexer(stream, EvaluatorPreprocessor.opMap);
  const knownCommands = EvaluatorPreprocessor.opMap;

  let token;
  let isValidOperator = true;
  while ((token = lexer.getObj()) !== EOF) {
    if (typeof token === "string" && !(token in knownCommands)) {
      isValidOperator = false;
      break;
    }
  }

  expect(isValidOperator).toBe(false);
});
