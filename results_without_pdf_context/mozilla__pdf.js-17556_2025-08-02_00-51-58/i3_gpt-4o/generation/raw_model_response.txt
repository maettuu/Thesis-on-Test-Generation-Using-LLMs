it("should ignore invalid PDF operators like 'toString'", async () => {
  const { Parser } = await import("../../src/core/parser.js");
  const { Lexer } = await import("../../src/core/parser.js");
  const stream = new StringStream("toString");
  const xref = new XRefMock([]);
  const parser = new Parser({
    lexer: new Lexer(stream, EvaluatorPreprocessor.opMap),
    xref,
  });

  const operation = { args: null };
  const result = parser.getObj();
  expect(result).toEqual(Name.get("toString"));

  const preprocessor = new EvaluatorPreprocessor(stream, xref);
  const hasOperation = preprocessor.read(operation);
  expect(hasOperation).toBe(false);
});
