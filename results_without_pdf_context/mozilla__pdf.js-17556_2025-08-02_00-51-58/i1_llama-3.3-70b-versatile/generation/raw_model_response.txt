it("should ignore invalid operator tokens that are also JavaScript object properties", async () => {
  const { PartialEvaluator, OPS, Dict, Name, Stream, StringStream } = await import("../../src/core/evaluator.js");
  const xref = new (await import("../../src/core/primitives.js")).XRefMock();
  const stream = new StringStream("toString");
  const evaluator = new PartialEvaluator(stream, xref);
  const operatorList = new (await import("../../src/core/operator_list.js")).OperatorList();
  let error;
  try {
    evaluator.read(operatorList);
  } catch (e) {
    error = e;
  }
  expect(error).toBeInstanceOf((await import("../../src/shared/util.js")).FormatError);
});
