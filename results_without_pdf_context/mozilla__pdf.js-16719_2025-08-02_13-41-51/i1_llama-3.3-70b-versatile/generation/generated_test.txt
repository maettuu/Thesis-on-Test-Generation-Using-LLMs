it("should delete the selected element when pressing the Delete key", async () => {
  const { AnnotationEditorUIManager, CommandManager } = await import("../../src/display/editor/tools.js");
  const { Annotation, MarkupAnnotation } = await import("../../src/core/annotation.js");
  const { getUuid } = await import("../../shared/util.js");
  const annotation = new MarkupAnnotation({
    subtype: "Highlight",
    rect: [10, 10, 20, 20],
    page: 1,
  });
  const editor = new AnnotationEditorUIManager(null, null, null, null);
  editor.#selectedEditors.add(annotation);
  const deleteSpy = jest.spyOn(editor, "delete");
  const event = {
    key: "Delete",
    preventDefault: jest.fn(),
  };
  editor.keydown(event);
  expect(deleteSpy).toHaveBeenCalledTimes(1);
});
