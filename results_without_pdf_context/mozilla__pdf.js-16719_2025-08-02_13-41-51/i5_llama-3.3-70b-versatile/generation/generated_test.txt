it("should delete the selected element when pressing the Delete key on a Mac", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { PDFDocument } = await import("../../src/core/document.js");
  const { PDFPage } = await import("../../src/core/document.js");
  const { Annotation } = await import("../../src/core/annotation.js");
  const { PDFViewer } = await import("../../web/pdf_viewer.js");

  const pdfDocument = await PDFDocument.load(await (await import("../../src/display/api.js")).getDocument({ url: "./test.pdf" }));
  const pdfPage = pdfDocument.getPage(1);
  const annotation = new Annotation();
  annotation.setPageNum(1);
  const annotationEditorUIManager = new AnnotationEditorUIManager(new PDFViewer(), pdfDocument, pdfPage);
  annotationEditorUIManager.selectAnnotation(annotation);

  const deleteKeyEvent = new KeyboardEvent('keydown', { key: 'Delete', metaKey: true });
  annotationEditorUIManager.handleKeyDown(deleteKeyEvent);

  const selectedAnnotation = annotationEditorUIManager.getSelectedAnnotation();
  expect(selectedAnnotation).toBeNull();
});
