it("should delete the selected element when pressing the Delete key", async () => {
  const { AnnotationEditorUIManager, KeyboardManager } = await import("../../src/display/editor/tools.js");
  const { AnnotationEditorPrefix, AnnotationEditorType } = await import("../../shared/util.js");
  const annotationEditorUIManager = new AnnotationEditorUIManager({}, {}, {}, {});
  const editor = { id: AnnotationEditorPrefix + "1", remove: jest.fn() };
  annotationEditorUIManager.addEditor(editor);
  annotationEditorUIManager.setSelected(editor);
  const keyboardManager = AnnotationEditorUIManager._keyboardManager;
  const event = new KeyboardEvent("keydown", { key: "Delete" });
  keyboardManager.exec(annotationEditorUIManager, event);
  expect(editor.remove).toHaveBeenCalledTimes(1);
});
