```javascript
it("should delete the selected element when pressing the Delete key on macOS", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { EventBus } = await import("../../web/event_utils.js");

  const eventBus = new EventBus();
  const container = document.createElement("div");
  const pdfDocument = { annotationStorage: new Map(), filterFactory: {} };
  const uiManager = new AnnotationEditorUIManager(container, eventBus, pdfDocument, null);

  const mockEditor = {
    id: "editor1",
    pageIndex: 0,
    remove: jest.fn(),
    isEmpty: () => false,
    select: jest.fn(),
    unselect: jest.fn(),
    commitOrRemove: jest.fn(),
    propertiesToUpdate: {},
  };

  uiManager.addEditor(mockEditor);
  uiManager.setSelected(mockEditor);

  const deleteEvent = new KeyboardEvent("keydown", { key: "Delete", metaKey: true });
  container.dispatchEvent(deleteEvent);

  expect(mockEditor.remove).toHaveBeenCalled();
});
```
