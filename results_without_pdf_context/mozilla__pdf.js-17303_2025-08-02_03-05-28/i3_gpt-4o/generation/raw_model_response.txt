```javascript
it("should throw JpegError when no frame data is found", async () => {
  const { JpegImage } = await import("../../src/core/jpg.js");
  const jpegImage = new JpegImage();
  const data = Buffer.from('ffd8', 'hex');
  let error;
  try {
    jpegImage.parse(data);
  } catch (e) {
    error = e;
  }
  expect(error).toBeInstanceOf(Error);
  expect(error.message).toBe("JPEG error: JpegImage.parse - no frame data found.");
});
```
