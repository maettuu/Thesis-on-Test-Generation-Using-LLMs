it("should not throw exception when parsing incomplete JPEG image", async () => {
  const { JpegImage } = await import("../../src/core/jpg.js");
  const jpegImage = new JpegImage();
  const imageData = Buffer.from('ffd8', 'hex');
  let error = null;
  try {
    jpegImage.parse(imageData);
  } catch (e) {
    error = e;
  }
  expect(error).toBeInstanceOf(Error);
  expect(error.message).toBe("JpegImage.parse - no frame data found.");
});
