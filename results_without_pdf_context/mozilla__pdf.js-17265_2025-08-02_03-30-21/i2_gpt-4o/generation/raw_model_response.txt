it("should load the module script without MIME type error", async () => {
  const { PDFWorker } = await import("../../src/display/api.js");
  const { createValidAbsoluteUrl } = await import("../../src/shared/util.js");

  const url = "https://cdnjs.cloudflare.com/ajax/libs/pdf.js/4.0.189/pdf.worker.min.mjs";
  const validUrl = createValidAbsoluteUrl(url, window.location.href);
  const blobUrl = URL.createObjectURL(new Blob([`await import("${validUrl}");`], { type: "text/javascript" }));

  let errorOccurred = false;
  try {
    await import(blobUrl);
  } catch (error) {
    errorOccurred = true;
  }

  expect(errorOccurred).toBe(false);
});
