```javascript
it("should load the module script without MIME type error", async () => {
  const { PDFWorker } = await import("../../src/display/api.js");
  const worker = new PDFWorker();
  const url = worker._createBlobUrl("https://mozilla.github.io/pdf.js/build/pdf.worker.mjs");
  const response = await fetch(url);
  const contentType = response.headers.get("content-type");
  expect(contentType).toBe("text/javascript");
});
```
