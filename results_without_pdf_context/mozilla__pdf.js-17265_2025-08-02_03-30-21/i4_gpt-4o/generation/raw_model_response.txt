```javascript
it("should load the module script without MIME type error", async () => {
  const { createValidAbsoluteUrl } = await import("../../src/shared/util.js");
  const url = "https://mozilla.github.io/pdf.js/build/pdf.worker.mjs";
  const blobUrl = URL.createObjectURL(new Blob([`await import("${url}");`], { type: "text/javascript" }));
  const expected = createValidAbsoluteUrl(url, window.location.href);
  const actual = createValidAbsoluteUrl(blobUrl, window.location.href);
  expect(actual).toBe(expected);
});
```
