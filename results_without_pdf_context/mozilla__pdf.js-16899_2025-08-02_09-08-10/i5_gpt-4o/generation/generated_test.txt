it("should display FreeText annotation with popup data", async () => {
  const { AnnotationLayer } = await import("../../src/display/annotation_layer.js");
  const { AnnotationType } = await import("../../src/shared/util.js");

  const annotationData = {
    annotationType: AnnotationType.FREETEXT,
    titleObj: { str: "Title" },
    contentsObj: { str: "Contents" },
    richText: { str: "Rich Text" },
    textContent: "Sample text",
    textPosition: { x: 10, y: 10 },
  };

  const parameters = {
    data: annotationData,
    page: { view: [0, 0, 100, 100] },
    viewport: { convertToViewportRectangle: (rect) => rect },
  };

  const annotationElement = new AnnotationLayer.FreeTextAnnotationElement(parameters);
  const container = annotationElement.render();

  const expected = true;
  const actual = container.querySelector(".popup") !== null;

  expect(actual).toBe(expected);
});
