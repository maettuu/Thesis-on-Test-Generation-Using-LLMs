it("should display FreeText annotation", async () => {
  const { AnnotationLayer } = await import("../../src/display/annotation_layer.js");
  const { PDFPageProxy } = await import("../../src/display/api.js");

  const mockPage = new PDFPageProxy();
  const mockViewport = { rotation: 0 };
  const mockDiv = document.createElement("div");
  const mockAnnotation = {
    annotationType: AnnotationType.FREETEXT,
    rect: [0, 0, 100, 100],
    titleObj: { str: "Title" },
    contentsObj: { str: "Contents" },
    richText: { str: "Rich Text" },
    id: "123",
  };

  const annotationLayer = new AnnotationLayer({
    div: mockDiv,
    page: mockPage,
    viewport: mockViewport,
  });

  await annotationLayer.render({
    annotations: [mockAnnotation],
    linkService: {},
    downloadManager: {},
    renderForms: false,
  });

  const annotationElement = mockDiv.querySelector("[data-annotation-id='123']");
  expect(annotationElement).not.toBeNull();
  expect(annotationElement.classList.contains("freeTextAnnotation")).toBe(true);
});
