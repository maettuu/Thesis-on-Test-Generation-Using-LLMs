it("should display FreeText annotation", async () => {
  const { AnnotationLayer } = await import("../../src/display/annotation_layer.js");
  const { PDFDocumentProxy } = await import("../../src/display/api.js");
  const { getDocument } = await import("../../src/display/api.js");

  const loadingTask = getDocument("path/to/pdf/with/freetext/annotation.pdf");
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);

  const annotationLayer = new AnnotationLayer();
  const annotations = await page.getAnnotations();
  const freeTextAnnotation = annotations.find(ann => ann.subtype === "FreeText");

  const container = document.createElement("div");
  annotationLayer.render({
    annotations: [freeTextAnnotation],
    div: container,
    page,
    viewport: page.getViewport({ scale: 1.0 }),
  });

  const actual = container.querySelector(".freeTextAnnotation") !== null;
  const expected = true;
  expect(actual).toBe(expected);
});
