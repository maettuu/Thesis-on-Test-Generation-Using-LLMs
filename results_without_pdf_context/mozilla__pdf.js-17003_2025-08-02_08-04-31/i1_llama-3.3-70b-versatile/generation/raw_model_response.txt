it("should not throw a warning when printing a pdf with an added annotation", async () => {
  const { AnnotationFactory, XRefMock, createIdFactory } = await import("../../src/core/annotation.js");
  const { DefaultCMapReaderFactory, DefaultStandardFontDataFactory } = await import("../../src/display/api.js");
  const { Dict, Name, Ref } = await import("../../src/core/primitives.js");
  const { warn } = await import("../../src/shared/util.js");

  const originalWarn = console.warn;
  console.warn = jest.fn();

  const xref = new XRefMock();
  const annotationGlobals = {
    pdfManager: {
      evaluatorOptions: {},
    },
    acroForm: new Dict(),
  };
  const idFactory = createIdFactory();
  const ref = new Ref(1, 0);
  const dict = new Dict();
  dict.set("Subtype", new Name("Widget"));
  dict.set("FT", new Name("Tx"));
  xref.fetchIfRef = () => dict;

  await AnnotationFactory.create(xref, ref, annotationGlobals, idFactory, true);

  expect(console.warn).not.toHaveBeenCalledWith("Unknown field name, falling back to empty field name.");
  console.warn = originalWarn;
});
