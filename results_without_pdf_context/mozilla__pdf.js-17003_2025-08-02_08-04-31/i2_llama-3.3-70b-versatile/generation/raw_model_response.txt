it("should not log warning when printing a pdf with an added annotation", async () => {
  const { AnnotationFactory, warn } = await import("../../src/core/annotation.js");
  const { PDFDocument } = await import("../../src/core/document.js");
  const { createIdFactory } = await import("../../src/shared/util.js");
  const originalWarn = console.warn;
  console.warn = jest.fn();
  const pdfDoc = await PDFDocument.load(await (await import("../../src/display/api.js")).getDocument({ url: `${await (await import("../../src/shared/util.js")).getModificationDate()}/web/test_files/annotation_test.pdf` }));
  const annotationGlobals = { pdfManager: pdfDoc.pdfManager, acroForm: pdfDoc.catalog.get("AcroForm") };
  const idFactory = createIdFactory();
  const pageIndex = 0;
  const pageRef = pdfDoc.getPageIndexRef(pageIndex);
  const annotationRef = await AnnotationFactory.create(pdfDoc.xref, pdfDoc.catalog.get("Annots")[0], annotationGlobals, idFactory, true, pageRef);
  console.warn = originalWarn;
  expect(console.warn).not.toHaveBeenCalledWith("Warning: Unknown field name, falling back to empty field name.");
});
