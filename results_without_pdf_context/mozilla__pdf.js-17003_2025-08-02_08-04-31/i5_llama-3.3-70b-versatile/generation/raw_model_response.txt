```javascript
it("should not throw warning when printing a pdf with an added annotation", async () => {
  const { PDFDocument, PDFPage } = await import("../../src/core/document.js");
  const { AnnotationFactory } = await import("../../src/core/annotation.js");
  const { createIdFactory } = await import("./test_utils.js");

  const pdfDoc = await PDFDocument.load(await fetch("path_to_pdf_file").then(response => response.arrayBuffer()));
  const page = pdfDoc.getPage(0);
  const annotationFactory = new AnnotationFactory({
    doc: pdfDoc,
    idFactory: createIdFactory(),
  });
  const annotation = annotationFactory.createAnnotation({
    subtype: "Widget",
    rect: [0, 0, 10, 10],
    pageIndex: 0,
  });
  const printResolution = 300;
  const printViewport = page.getViewport({ scale: printResolution / 72 });
  const renderTask = page.render({
    viewport: printViewport,
    intent: "print",
  });
  const consoleWarnSpy = jest.spyOn(console, "warn");
  await renderTask.promise;
  expect(consoleWarnSpy).not.toHaveBeenCalledWith("Unknown field name, falling back to empty field name.");
});
```
