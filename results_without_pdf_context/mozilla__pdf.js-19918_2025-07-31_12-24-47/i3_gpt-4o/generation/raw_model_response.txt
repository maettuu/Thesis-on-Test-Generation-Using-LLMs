it("should support decoding CMYK JPEG images using ImageDecoder", async () => {
  const { JpegStream } = await import("../../src/core/jpeg_stream.js");
  const { Stream } = await import("../../src/core/stream.js");

  const cmykJpegData = new Uint8Array([
    0xff, 0xd8, 0xff, 0xe0, 0x00, 0x10, 0x4a, 0x46, 0x49, 0x46, 0x00, 0x01,
    0x01, 0x01, 0x00, 0x60, 0x00, 0x60, 0x00, 0x00, 0xff, 0xee, 0x00, 0x0e,
    0x41, 0x64, 0x6f, 0x62, 0x65, 0x00, 0x64, 0x00, 0x00, 0x00, 0x00, 0x00,
    0xff, 0xc0, 0x00, 0x11, 0x08, 0x00, 0x60, 0x00, 0x60, 0x04, 0x01, 0x11,
    0x00, 0x02, 0x11, 0x01, 0x03, 0x11, 0x01, 0x04, 0x11, 0x01, 0xff, 0xda,
    0x00, 0x0c, 0x03, 0x01, 0x00, 0x02, 0x11, 0x03, 0x11, 0x00, 0x3f, 0x00,
    0xff, 0xd9
  ]);
  const stream = new Stream(cmykJpegData);
  const jpegStream = new JpegStream(stream, cmykJpegData.length, null);

  const canDecode = await JpegStream.canUseImageDecoderCMYK;
  const transferableImage = await jpegStream.getTransferableImage();

  expect(canDecode).toBe(true);
  expect(transferableImage).not.toBeNull();
});
