it("should support decoding cmyk jpeg image using ImageDecoder", async () => {
  const { PDFDocumentProxy, PDFPageProxy, PDFWorker } = await import("../../src/display/api.js");
  const { buildGetDocumentParams, TEST_PDFS_PATH } = await import("./test_utils.js");
  const { JpegImage } = await import("../../src/core/jpg.js");
  const pdfFile = `${TEST_PDFS_PATH}/cmyk-jpeg.pdf`;
  const pdfDocument = await getDocument(buildGetDocumentParams(pdfFile));
  const pdfPage = await pdfDocument.getPage(1);
  const image = await pdfPage.getImage(0);
  const jpegImage = new JpegImage();
  const data = await image.getData();
  jpegImage.parse(data);
  const canUseImageDecoder = await JpegImage.canUseImageDecoder(data);
  expect(canUseImageDecoder).not.toBeNull();
});
