it("should support decoding CMYK JPEG images using ImageDecoder", async () => {
  const { JpegStream } = await import("../../src/core/jpeg_stream.js");
  const { JpegImage } = await import("../../src/core/jpg.js");

  const cmykJpegData = new Uint8Array([/* CMYK JPEG data bytes */]);
  const expected = true;

  JpegStream.setOptions({ isImageDecoderSupported: true });
  const jpegInfo = JpegImage.canUseImageDecoder(cmykJpegData);
  const actual = jpegInfo.cmyk && await JpegStream.canUseImageDecoderCMYK;

  expect(actual).toBe(expected);
});
