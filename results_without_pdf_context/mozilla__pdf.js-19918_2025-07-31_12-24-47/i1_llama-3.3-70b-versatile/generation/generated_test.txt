it("should decode cmyk jpeg image using ImageDecoder", async () => {
  const { JpegStream, JpegImage } = await import("../../src/core/jpeg_stream.js");
  const { JpegImage as JpegImageClass } = await import("../../src/core/jpg.js");
  const data = new Uint8Array([0xff, 0xd8, 0xff, 0xe1, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00]);
  const useImageDecoder = JpegImageClass.canUseImageDecoder(data);
  const stream = { dict: {}, asyncGetBytes: async () => data };
  const jpegStream = new JpegStream(stream, null, { colorTransform: -1 });
  const image = await jpegStream.getTransferableImage();
  expect(image).not.toBeNull();
});
