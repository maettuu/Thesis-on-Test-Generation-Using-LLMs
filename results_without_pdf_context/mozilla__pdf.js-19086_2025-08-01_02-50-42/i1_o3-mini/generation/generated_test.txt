it("should populate options with fieldValue when no options are defined", async () => {
  const { ChoiceWidgetAnnotation } = await import("../../src/core/annotation.js");
  const { Dict } = await import("../../src/core/primitives.js");
  const dict = new Dict({});
  const params = {
    dict,
    xref: { fetchIfRef: x => x },
    data: { fieldValue: "Account123" }
  };
  const annot = new ChoiceWidgetAnnotation(params);
  if (annot.data.options.length !== 1) {
    throw new Error("Expected options to be populated from fieldValue.");
  }
  const option = annot.data.options[0];
  if (option.exportValue !== "Account123" || option.displayValue !== "Account123") {
    throw new Error("Option values do not match fieldValue.");
  }
});
