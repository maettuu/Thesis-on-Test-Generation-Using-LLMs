it("should render text layer with correct font family on Windows with Firefox", async () => {
  const { TextLayer, FeatureTest } = await import("../../src/display/text_layer.js");
  const { Util } = await import("../../src/shared/util.js");
  const { PageViewport } = await import("../../src/display/display_utils.js");

  const viewport = new PageViewport({ scale: 1, rotation: 0, width: 100, height: 100 });
  const textContentSource = { items: [{ str: "Hello World", fontName: "sans-serif" }] };
  const container = document.createElement("div");
  const textLayer = new TextLayer({ textContentSource, container, viewport });

  const isWindows = navigator.platform.includes("Win");
  const isFirefox = navigator.userAgent.includes("Firefox");
  const expectedFontFamily = isWindows && isFirefox ? "Calibri, sans-serif" : "sans-serif";

  const textDiv = container.querySelector("span");
  const actualFontFamily = textDiv.style.fontFamily;

  expect(actualFontFamily).toBe(expectedFontFamily);
});
