```javascript
it("should render text layer correctly on Windows with Firefox", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const { TextLayer } = await import("../../src/display/text_layer.js");
  const { FeatureTest } = await import("../../src/shared/util.js");

  const originalPlatform = FeatureTest.platform;
  FeatureTest.platform = {
    isWindows: true,
    isFirefox: true,
  };

  const textContentSource = {
    items: [
      {
        str: "Hello World",
        fontName: "sans-serif",
        fontSize: 12,
        transform: [1, 0, 0, 1, 0, 0],
      },
    ],
  };

  const container = document.createElement("div");
  const viewport = {
    scale: 1,
    rotation: 0,
    rawDims: {
      pageWidth: 100,
      pageHeight: 100,
      pageX: 0,
      pageY: 0,
    },
  };

  const textLayer = new TextLayer({
    textContentSource,
    container,
    viewport,
  });

  await textLayer.render();

  const textDiv = container.querySelector("span");
  const fontFamily = textDiv.style.fontFamily;

  expect(fontFamily).toBe("Calibri, sans-serif");

  FeatureTest.platform = originalPlatform;
});
```
