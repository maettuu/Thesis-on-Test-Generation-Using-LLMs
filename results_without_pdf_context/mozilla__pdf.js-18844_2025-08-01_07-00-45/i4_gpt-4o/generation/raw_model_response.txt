it("should correctly map font families on Windows with Firefox", async () => {
  const { TextLayer } = await import("../../src/display/text_layer.js");
  const { FeatureTest } = await import("../../src/shared/util.js");

  // Mock platform detection
  FeatureTest.platform = { isWindows: true, isFirefox: true };

  const expectedFontFamilyMap = new Map([
    ["sans-serif", "Calibri, sans-serif"],
    ["monospace", "Lucida Console, monospace"],
  ]);

  const actualFontFamilyMap = TextLayer.fontFamilyMap;

  for (const [key, value] of expectedFontFamilyMap) {
    if (actualFontFamilyMap.get(key) !== value) {
      throw new Error(`Font family mapping for ${key} is incorrect.`);
    }
  }
});
