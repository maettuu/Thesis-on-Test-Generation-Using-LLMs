it("should render text layer correctly on Windows with Firefox", async () => {
  const { TextLayer } = await import("../../src/display/text_layer.js");
  const { setLayerDimensions } = await import("../../src/display/display_utils.js");
  const { FeatureTest } = await import("../../src/shared/util.js");

  const container = document.createElement("div");
  const viewport = { scale: 1, rotation: 0, rawDims: { pageWidth: 100, pageHeight: 100, pageX: 0, pageY: 0 } };
  const textContentSource = { items: [{ str: "Hello World", fontName: "sans-serif" }] };

  const originalPlatform = FeatureTest.platform;
  FeatureTest.platform = { isWindows: true, isFirefox: true };

  const textLayer = new TextLayer({ textContentSource, container, viewport });
  await textLayer.render();

  const textDivs = textLayer.textDivs;
  const fontFamily = textDivs[0].style.fontFamily;

  expect(fontFamily).toBe("Calibri, sans-serif");

  FeatureTest.platform = originalPlatform;
});
