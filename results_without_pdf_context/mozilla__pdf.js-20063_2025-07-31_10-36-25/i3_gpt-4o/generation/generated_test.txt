it("should render PolygonAnnotation with fill color in PDFjs", async () => {
  const { PolygonAnnotation } = await import("../../src/core/annotation.js");
  const { Dict, Name } = await import("../../src/core/primitives.js");
  const xref = new XRefMock();
  const dict = new Dict(xref);
  dict.set("Subtype", Name.get("Polygon"));
  dict.set("C", [0, 0, 0]); // Stroke color
  dict.set("IC", [1, 0, 0]); // Fill color
  dict.set("Vertices", [0, 0, 100, 0, 100, 100, 0, 100]);
  const annotation = new PolygonAnnotation({
    dict,
    xref,
    annotationGlobals: { acroForm: Dict.empty },
  });
  const expectedFillColor = [1, 0, 0];
  const actualFillColor = annotation.data.fillColor;
  expect(actualFillColor).toEqual(expectedFillColor);
});
