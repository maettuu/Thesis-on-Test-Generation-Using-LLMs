it("should render PolygonAnnotation fill color in PDF.js web viewer", async () => {
  const { AnnotationFactory, MarkupAnnotation } = await import("../../src/core/annotation.js");
  const { AnnotationType } = await import("../../src/shared/util.js");
  const { Dict, Name } = await import("../../src/core/primitives.js");
  const annotation = new MarkupAnnotation({
    subtype: AnnotationType.POLYGON,
    rectangle: [0, 0, 100, 100],
    vertices: [10, 10, 90, 10, 90, 90, 10, 90],
    color: [1, 0, 0], // red
    interiorColor: [0, 1, 0], // green
    dict: new Dict({
      Subtype: new Name("Polygon"),
      Rect: [0, 0, 100, 100],
      Vertices: [10, 10, 90, 10, 90, 90, 10, 90],
      C: [1, 0, 0], // red
      IC: [0, 1, 0], // green
    }),
  });
  const expectedOperator = "f";
  const actualOperator = annotation.pointsCallback([], [0, 0, 100, 100])[2];
  expect(actualOperator).toBe(expectedOperator);
});
