it("should render PolygonAnnotation with fill color in PDF.js web viewer", async () => {
  const { AnnotationFactory } = await import("../../src/core/annotation.js");
  const { Dict, Name, Ref } = await import("../../src/core/primitives.js");
  const { RenderingIntentFlag } = await import("../../src/shared/util.js");

  const dict = new Dict();
  dict.set("Type", Name.get("Annot"));
  dict.set("Subtype", Name.get("Polygon"));
  dict.set("Vertices", [0, 0, 100, 0, 100, 100, 0, 100]);
  dict.set("IC", [0.5, 0.5, 0.5]); // Fill color
  dict.set("C", [0, 0, 0]); // Stroke color

  const xref = new XRefMock([{ ref: Ref.get(1, 0), data: dict }]);
  const annotation = await AnnotationFactory.create(xref, dict, Ref.get(1, 0));

  const appearance = await annotation._getAppearance(
    null,
    null,
    RenderingIntentFlag.DISPLAY,
    null
  );

  expect(appearance).toContain("B"); // Check if both fill and stroke are applied
});
