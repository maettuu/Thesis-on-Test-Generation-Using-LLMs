it("should render PolygonAnnotation with fill color in PDF.js web viewer", async () => {
  const { AnnotationFactory } = await import("../../src/core/annotation.js");
  const { Dict, Name, Ref } = await import("../../src/core/primitives.js");
  const { RenderingIntentFlag } = await import("../../src/shared/util.js");

  const xref = {
    fetchIfRef: (ref) => ref,
  };

  const dict = new Dict();
  dict.set("Subtype", Name.get("Polygon"));
  dict.set("IC", [0.5, 0.5, 0.5]); // Fill color
  dict.set("C", [0, 0, 0]); // Stroke color
  dict.set("Vertices", [0, 0, 1, 1, 2, 2, 3, 3]);

  const annotation = await AnnotationFactory.create(
    xref,
    new Ref(1, 0),
    { pdfManager: { evaluatorOptions: {} } },
    { createObjId: () => 1 }
  );

  const intent = RenderingIntentFlag.DISPLAY;
  const operatorList = await annotation.getOperatorList(
    { getOperatorList: () => {} },
    {},
    intent
  );

  const expectedOperator = "B"; // Expecting both fill and stroke
  const actualOperator = operatorList.opList.args[operatorList.opList.args.length - 1];

  expect(actualOperator).toBe(expectedOperator);
});
