it("should render PolygonAnnotation with fill color in PDF.js web viewer", async () => {
  const { PDFDocument } = await import("../../src/display/api.js");
  const { AnnotationLayerBuilder } = await import("../../web/annotation_layer_builder.js");
  const pdfData = await (await import("../../src/display/node_utils.js")).fetchData("https://github.com/user-attachments/files/21099399/document.pdf");
  const pdf = await PDFDocument.load(pdfData);
  const page = await pdf.getPage(1);
  const viewport = page.getViewport({ scale: 1 });
  const annotationLayer = new AnnotationLayerBuilder().build({
    page,
    viewport,
    renderInteractiveForms: false,
  });
  const annotations = await page.getAnnotations();
  const polygonAnnotation = annotations.find((annotation) => annotation.subtype === "Polygon");
  const fillColor = polygonAnnotation.getFillColor();
  const expectedFillColor = [1, 0, 0]; // Assuming the fill color is red
  expect(fillColor).toEqual(expectedFillColor);
});
