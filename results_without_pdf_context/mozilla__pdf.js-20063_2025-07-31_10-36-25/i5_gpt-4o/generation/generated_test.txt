it("should render PolygonAnnotation with both outline and fill color", async () => {
  const { PDFDocumentProxy, getDocument } = await import("../../src/display/api.js");
  const { AnnotationLayer } = await import("../../src/display/annotation_layer.js");
  const { RenderingIntentFlag } = await import("../../src/shared/util.js");

  const loadingTask = getDocument("path/to/document.pdf");
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const annotations = await page.getAnnotations({ intent: RenderingIntentFlag.DISPLAY });

  const polygonAnnotation = annotations.find(ann => ann.subtype === "Polygon");
  const annotationLayer = new AnnotationLayer();
  const renderResult = annotationLayer.render({
    annotations: [polygonAnnotation],
    viewport: page.getViewport({ scale: 1.0 }),
    div: document.createElement("div"),
    intent: RenderingIntentFlag.DISPLAY,
  });

  const expectedFillColor = [1, 0, 0]; // Example fill color in RGB
  const actualFillColor = renderResult.annotations[0].fillColor;

  expect(actualFillColor).toEqual(expectedFillColor);
});
