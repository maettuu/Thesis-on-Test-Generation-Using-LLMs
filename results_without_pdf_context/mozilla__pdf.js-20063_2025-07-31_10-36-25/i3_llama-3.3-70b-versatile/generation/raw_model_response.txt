it("should render PolygonAnnotation with fill color", async () => {
  const pdfData = await fetch('https://github.com/user-attachments/files/21099399/document.pdf').then(response => response.arrayBuffer());
  const xref = new XRefMock(pdfData);
  const annotationFactory = new AnnotationFactory();
  const annotationGlobals = await annotationFactory.createGlobals(xref);
  const pageIndex = 0;
  const pageRef = await xref.fetchIfRefAsync(await xref.getPageIndex(pageIndex));
  const annotations = await xref.fetchIfRefAsync(pageRef).getArray("Annots");
  const annotation = await annotationFactory.create(xref, annotations[0], annotationGlobals, createIdFactory(), true, null, pageRef);
  const operatorList = await annotation.getOperatorList(new PartialEvaluator(), new WorkerTask(), RenderingIntentFlag.DISPLAY);
  const expectedOperator = "f";
  const actualOperator = operatorList.opList.ops[operatorList.opList.ops.length - 2].args[0];
  assert.equal(actualOperator, expectedOperator, `Expected operator to be "${expectedOperator}", but got "${actualOperator}"`);
});
