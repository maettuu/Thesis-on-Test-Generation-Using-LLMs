it("should decrypt PDFs with RC encryption and key length less than bits", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const pdfUrl = "https://github.com/user-attachments/files/18788789/R.4.V.4.40-bit.RC4.pdf";
  const pdf = await PDFDocument.load(pdfUrl);
  const firstPage = await pdf.getPage(1);
  const textContent = await firstPage.getTextContent();
  const expectedText = "This is a test PDF with RC4 encryption and a 40-bit key.";
  expect(textContent.items[0].str).toContain(expectedText);
});
