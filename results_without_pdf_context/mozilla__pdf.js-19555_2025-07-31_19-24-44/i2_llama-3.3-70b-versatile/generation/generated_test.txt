it("should decrypt PDFs with RC encryption where R and key length bits", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { DefaultFileReaderFactory } = await import("./test_utils.js");
  const fileReaderFactory = new DefaultFileReaderFactory();
  const pdfFile = await fileReaderFactory.readFile("R=4, V=4, 40-bit RC4.pdf");
  const pdfDoc = await PDFDocument.load({ data: pdfFile, cMapUrl: "https://cdn.jsdelivr.net/npm/pdfjs-dist@2.16.105/build/generic/web/cmaps/" });
  const firstPage = await pdfDoc.getPage(1);
  const textContent = await firstPage.getTextContent();
  const expectedText = "Lorem ipsum dolor sit amet, consectetur adipiscing elit.";
  expect(textContent.items[0].str).toContain(expectedText);
});
