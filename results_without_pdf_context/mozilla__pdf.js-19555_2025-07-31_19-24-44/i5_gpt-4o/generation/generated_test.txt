it("should correctly handle RC encryption with key length bits", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");

  const url = "https://github.com/user-attachments/files/18788789/R.4.V.4.40-bit.RC4.pdf";
  const loadingTask = getDocument(buildGetDocumentParams(url));
  const pdfDocument = await loadingTask.promise;

  const page = await pdfDocument.getPage(1);
  const textContent = await page.getTextContent();
  const actualText = textContent.items.map(item => item.str).join(" ");

  const expectedText = "Expected text from the PDF";
  expect(actualText).toContain(expectedText);
});
