it("should support rare RC4 encryption where R=4, key length < 128 bits", async () => {
  const { CipherTransformFactory } = await import("../../src/core/crypto.js");
  const { Dict, Name } = await import("../../src/core/primitives.js");
  const { stringToBytes } = await import("../../src/shared/util.js");

  const dict = new Dict();
  dict.set("Filter", Name.get("Standard"));
  dict.set("V", 4);
  dict.set("R", 4);
  dict.set("Length", 40);
  dict.set("O", stringToBytes("ownerPassword"));
  dict.set("U", stringToBytes("userPassword"));
  dict.set("P", -4);
  dict.set("EncryptMetadata", true);

  const fileId = "fileId";
  const password = "";

  const factory = new CipherTransformFactory(dict, fileId, password);
  const expectedKeyLength = 16;
  const actualKeyLength = factory.encryptionKey.length;

  expect(actualKeyLength).toBe(expectedKeyLength);
});
