it("should release ImageBitmap data after cleanup", async () => {
  const { PDFPageProxy } = await import("../../src/display/api.js");
  const { PDFObjects } = await import("../../src/display/api.js");

  const pageProxy = new PDFPageProxy(0, { rotate: 0, ref: null, userUnit: 1, view: [0, 0, 612, 792] }, null);
  const objs = new PDFObjects();
  const imageData = { bitmap: { close: jest.fn() } };
  const id = "testImage";

  pageProxy.objs = objs;
  pageProxy.objs.resolve(id, imageData);

  pageProxy.cleanup();

  expect(imageData.bitmap.close).toHaveBeenCalled();
});
