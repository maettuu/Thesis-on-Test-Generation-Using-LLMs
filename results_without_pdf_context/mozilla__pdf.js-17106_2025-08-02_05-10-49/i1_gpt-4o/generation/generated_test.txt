it("should release ImageBitmap data after cleanup", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const loadingTask = getDocument(buildGetDocumentParams("tracemonkey.pdf"));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const viewport = page.getViewport({ scale: 1.0 });
  const canvasFactory = new DefaultCanvasFactory();
  const canvasAndContext = canvasFactory.create(viewport.width, viewport.height);
  const renderTask = page.render({
    canvasContext: canvasAndContext.context,
    viewport,
  });
  await renderTask.promise;

  const imageData = page.objs.get("img_p0_1");
  const bitmapClosedBefore = imageData.bitmap ? imageData.bitmap.closed : true;

  page.cleanup();

  const bitmapClosedAfter = imageData.bitmap ? imageData.bitmap.closed : true;

  expect(bitmapClosedBefore).toBe(false);
  expect(bitmapClosedAfter).toBe(true);
});
