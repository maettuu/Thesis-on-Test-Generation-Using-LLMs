it("should correctly open a PDF with mismatched xref table entries in Firefox", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");
  const pdfUrl = `${TEST_PDFS_PATH}mismatched_xref_table.pdf`;
  const loadingTask = getDocument(buildGetDocumentParams(pdfUrl));
  const pdfDocument = await loadingTask.promise;
  const numPages = pdfDocument.numPages;
  expect(numPages).toBeGreaterThan(0);
  await pdfDocument.getPage(1);
  await loadingTask.destroy();
});
