it("should resolve HTML Spec page printed to PDF cannot be opened in Firefox", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { PDFFetchStream } = await import("../../src/display/fetch_stream.js");
  const { PDFNetworkStream } = await import("../../src/display/network.js");
  const { createHeaders } = await import("../../src/display/network_utils.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");
  const { TEST_PDFS_PATH } = await import("./test_utils.js");

  const pdfUrl = `${TEST_PDFS_PATH}/html-spec-page.pdf`;
  const params = buildGetDocumentParams(pdfUrl, null, createHeaders());
  const stream = new PDFFetchStream(params);
  const networkStream = new PDFNetworkStream(stream);
  const doc = await PDFDocument.load({ cMapUrl: "", cMapPacked: true, networkStream });

  const expected = true;
  const actual = doc !== null;

  expect(actual).toBe(expected);
});
