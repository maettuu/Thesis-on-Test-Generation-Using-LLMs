it("should resolve the issue with HTML Spec page printed to PDF not opening in Firefox", async () => {
  const { PDFDocumentProxy, PDFDocumentLoadingTask, PDFWorker } = await import("../../src/display/api.js");
  const { buildGetDocumentParams, TestPdfsServer } = await import("./test_utils.js");
  const server = new TestPdfsServer();
  const pdfUrl = `${server.url}/html-spec.pdf`;
  const params = buildGetDocumentParams(pdfUrl);
  const worker = new PDFWorker({ workerSrc: "../../build/generic/build/pdf.worker.mjs" });
  const loadingTask = PDFDocumentLoadingTask.fromURL(pdfUrl, params);
  const document = await loadingTask.promise;
  const firstPage = await document.getPage(1);
  const viewport = firstPage.getViewport({ scale: 1 });
  const renderTask = firstPage.render({
    viewport,
    intent: "display",
  });
  await renderTask.promise;
  const pdfDocumentProxy = new PDFDocumentProxy(document);
  await pdfDocumentProxy.ensureDoc();
  const xref = pdfDocumentProxy.xref;
  const firstXRefStmPos = xref.#firstXRefStmPos;
  expect(firstXRefStmPos).not.toBeNull();
});
