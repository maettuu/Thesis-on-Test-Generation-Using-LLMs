it("should fix the XRef table generation issue when printing HTML spec pages to PDF", async () => {
  const { PDFDocumentLoadingTask } = await import("../../src/display/api.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");
  const params = buildGetDocumentParams(`${TEST_PDFS_PATH}html_spec.pdf`);
  const loadingTask = PDFDocumentLoadingTask.fromParams(params);
  const pdfDoc = await loadingTask.promise;
  const catalog = await pdfDoc.catalog;
  const expectedRootRef = pdfDoc.xref.root;
  const actualRootRef = catalog.get("Root");
  expect(actualRootRef).toEqual(expectedRootRef);
});
