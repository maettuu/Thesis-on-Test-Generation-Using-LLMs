it("should respect the PDFjs verbosity levels for OpenJPEG warnings", async () => {
  const { setVerbosityLevel, VerbosityLevel, JpxImage } = await import("../../src/core/jpx.js");
  const { Util } = await import("../../src/shared/util.js");
  setVerbosityLevel(VerbosityLevel.DISABLE);
  const originalWarn = console.warn;
  let warned = false;
  console.warn = () => {
    warned = true;
  };
  const imageData = JpxImage.decode(new Uint8Array([0x00, 0x00, 0x00, 0x0C, 0x6A, 0x50, 0x20, 0x20, 0x0D, 0x0A, 0x87, 0x0A]));
  console.warn = originalWarn;
  if (imageData instanceof Error) {
    warned = true;
  }
  expect(warned).toBe(false);
});
