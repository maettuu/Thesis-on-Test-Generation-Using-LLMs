it("should respect verbosity levels for OpenJPEG warnings", async () => {
  const { JpxImage } = await import("../../src/core/jpx.js");
  const { setVerbosityLevel, VerbosityLevel } = await import("../../src/shared/util.js");

  setVerbosityLevel(VerbosityLevel.ERRORS);

  const originalConsoleWarn = console.warn;
  let consoleWarnCalled = false;
  console.warn = () => {
    consoleWarnCalled = true;
  };

  try {
    JpxImage.decode(new Uint8Array([0, 1, 2, 3]), false);
  } catch (e) {
    // Catch the error to prevent test failure due to exception
  }

  console.warn = originalConsoleWarn;

  expect(consoleWarnCalled).toBe(false);
});
