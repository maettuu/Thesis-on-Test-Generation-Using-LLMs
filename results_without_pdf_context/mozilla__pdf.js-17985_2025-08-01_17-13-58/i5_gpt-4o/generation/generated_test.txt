it("should respect verbosity levels for OpenJPEG warnings", async () => {
  const { JpxImage } = await import("../../src/core/jpx.js");
  const { setVerbosityLevel, VerbosityLevel } = await import("../../src/shared/util.js");

  setVerbosityLevel(VerbosityLevel.ERRORS);

  let warningMessage = null;
  const originalWarn = console.warn;
  console.warn = (msg) => {
    warningMessage = msg;
  };

  try {
    const data = new Uint8Array([0, 1, 2, 3]); // Invalid data to trigger a warning
    JpxImage.decode(data);
  } catch (e) {
    // Ignore errors, we are testing warnings
  }

  console.warn = originalWarn;

  expect(warningMessage).toBeNull();
});
