it("should render PDF with transparency filters correctly when <base href> is defined", async () => {
  const { DOMFilterFactory } = await import("../../src/display/display_utils.js");
  const { PDFDocument } = await import("../../src/display/api.js");
  const { PDFNetworkStream } = await import("../../src/display/network.js");
  const pdfUrl = "https://github.com/mozilla/pdf.js/files/11229069/test.pdf";
  const pdfDocument = await PDFDocument.load({ url: pdfUrl, verbosity: 0 });
  const filterFactory = new DOMFilterFactory();
  const filter = filterFactory.addFilter([0, 1]);
  const baseHref = document.createElement("base");
  baseHref.href = ".";
  document.head.appendChild(baseHref);
  const svg = document.createElementNS("http://www.w3.org/2000/svg", "svg");
  const filterElement = document.createElementNS("http://www.w3.org/2000/svg", "filter");
  filterElement.id = filter.substring(4);
  svg.appendChild(filterElement);
  document.body.appendChild(svg);
  const actualFilter = filterElement.getAttribute("id");
  const expectedFilter = filter.substring(4);
  expect(actualFilter).toBe(expectedFilter);
});
