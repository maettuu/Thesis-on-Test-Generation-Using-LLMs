it("should render password fields correctly", async () => {
  const pdfManager = new XRefMock();
  const annotationGlobals = await AnnotationFactory.createGlobals(pdfManager);
  const idFactory = createIdFactory();
  const dict = new Dict();
  dict.set("FT", Name.get("Tx"));
  dict.set("Ff", AnnotationFieldFlag.PASSWORD);
  dict.set("T", stringToBytes("Test"));
  dict.set("Rect", [0, 0, 100, 100]);
  const annotation = await AnnotationFactory.create(
    pdfManager,
    new Ref(1),
    annotationGlobals,
    idFactory,
    false
  );
  const widgetAnnotation = annotation instanceof WidgetAnnotation;
  expect(widgetAnnotation).toBe(true);
  const data = annotation.data;
  expect(data.password).toBe(true);
  const textWidgetAnnotation = new TextWidgetAnnotation({
    dict,
    xref: pdfManager.xref,
    annotationGlobals,
    ref: new Ref(1),
    evaluatorOptions: {},
  });
  const fieldObject = textWidgetAnnotation.getFieldObject();
  expect(fieldObject.password).toBe(true);
});
