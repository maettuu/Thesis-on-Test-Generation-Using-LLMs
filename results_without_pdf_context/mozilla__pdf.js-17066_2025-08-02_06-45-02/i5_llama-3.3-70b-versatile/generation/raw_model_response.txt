it("should render read-only form fields when AnnotationMode is ENABLED", async () => {
  const { PDFDocument, PDFPage } = await import("../../src/core/document.js");
  const { AnnotationMode } = await import("../../src/shared/util.js");
  const { PDFFetchStream } = await import("../../src/display/fetch_stream.js");
  const pdf = await PDFDocument.load(await new PDFFetchStream("simple-form-readonly.pdf").getBuffer());
  const page = await pdf.getPage(1);
  const renderTask = await page.render({
    annotationMode: AnnotationMode.ENABLED,
  });
  const textWidgetAnnotations = renderTask.annotationLayer.getAnnotations().filter(annotation => annotation instanceof TextWidgetAnnotationElement);
  const isRenderable = textWidgetAnnotations.length > 0 && textWidgetAnnotations[0].isRenderable;
  expect(isRenderable).toBe(true);
});
