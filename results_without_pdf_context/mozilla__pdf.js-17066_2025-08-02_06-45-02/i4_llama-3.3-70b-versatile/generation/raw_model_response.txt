it("should render read-only form fields when AnnotationMode is ENABLED", async () => {
  const { PDFPageProxy } = await import("./api.js");
  const { PageViewport } = await import("./display_utils.js");
  const { AnnotationLayer } = await import("./annotation_layer.js");
  const { AnnotationStorage } = await import("./annotation_storage.js");
  const { NullL10n } = await import("../../web/l10n_utils.js");

  const viewport = new PageViewport({ width: 100, height: 100, rotation: 0 });
  const annotationLayer = new AnnotationLayer({
    div: document.createElement("div"),
    accessibilityManager: null,
    annotationCanvasMap: null,
    l10n: new NullL10n(),
    page: new PDFPageProxy({}),
    viewport,
  });

  const annotations = [
    {
      annotationType: "Widget",
      fieldType: "Tx",
      rect: [0, 0, 100, 100],
      id: "test",
      fieldValue: "Test",
      readOnly: true,
    },
  ];

  const params = {
    viewport,
    div: annotationLayer.div,
    annotations,
    page: new PDFPageProxy({}),
    linkService: {},
    downloadManager: {},
    annotationStorage: new AnnotationStorage(),
    renderForms: false,
    enableScripting: false,
    hasJSActions: false,
  };

  await annotationLayer.render(params);

  const renderedElement = annotationLayer.div.querySelector(`[data-annotation-id="test"]`);
  expect(renderedElement).not.toBeNull();
});
