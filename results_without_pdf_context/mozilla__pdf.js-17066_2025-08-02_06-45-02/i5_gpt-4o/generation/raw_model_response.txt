it("should render read-only form field when AnnotationMode is 1 (ENABLED)", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const { AnnotationMode } = await import("../../src/shared/util.js");
  const { PDFNodeStream } = await import("../../src/display/node_stream.js");
  const fs = await import("fs");
  const path = await import("path");

  const pdfPath = path.resolve(__dirname, "simple-form-readonly.pdf");
  const data = new Uint8Array(fs.readFileSync(pdfPath));
  const loadingTask = getDocument({
    data,
    annotationMode: AnnotationMode.ENABLED,
  });

  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const annotations = await page.getAnnotations();

  const readOnlyField = annotations.find(
    (annotation) => annotation.fieldName === "readonlyField"
  );

  expect(readOnlyField).toBeDefined();
  expect(readOnlyField.fieldValue).toBe("Foo");
});
