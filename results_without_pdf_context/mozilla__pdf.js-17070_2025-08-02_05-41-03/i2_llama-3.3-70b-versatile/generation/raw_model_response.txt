it("should render comment and sticky note annotations in the viewer", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { AnnotationLayerBuilder } = await import("../../web/annotation_layer_builder.js");
  const { PDFPageView } = await import("../../web/pdf_page_view.js");

  const pdfData = await fetch("https://github.com/mozilla/pdf.js/files/12802422/Testdoc.pdf").then(response => response.arrayBuffer());
  const pdfDocument = await PDFDocument.load({ data: pdfData });
  const pdfPage = await pdfDocument.getPage(1);
  const pageView = new PDFPageView({ pdfPage });
  const annotationLayerBuilder = new AnnotationLayerBuilder();
  const annotationLayer = annotationLayerBuilder.createAnnotationLayer(pageView, pdfPage.getAnnotations());

  const expectedAnnotations = 2;
  const actualAnnotations = annotationLayer.annotations.length;

  expect(actualAnnotations).toBe(expectedAnnotations);
});
