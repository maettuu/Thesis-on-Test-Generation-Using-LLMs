it("should render comment and sticky note annotations", async () => {
  const { AnnotationFactory } = await import("../../src/core/annotation.js");
  const { Dict, Ref } = await import("../../src/core/primitives.js");
  const xref = new XRefMock([
    {
      ref: Ref.get(1, 0),
      data: Dict.empty,
    },
  ]);
  const annotationParams = {
    xref,
    ref: Ref.get(1, 0),
    dict: new Dict(xref),
    subtype: "Text",
    id: "1R",
    annotationGlobals: {
      structTreeRoot: null,
    },
    collectFields: false,
    needAppearances: false,
    pageIndex: 0,
    evaluatorOptions: {},
    pageRef: null,
  };
  const annotation = AnnotationFactory._create(annotationParams);
  const expected = {
    annotationType: AnnotationType.TEXT,
    noHTML: false,
  };
  expect(annotation.data.annotationType).toEqual(expected.annotationType);
  expect(annotation.data.noHTML).toEqual(expected.noHTML);
});
