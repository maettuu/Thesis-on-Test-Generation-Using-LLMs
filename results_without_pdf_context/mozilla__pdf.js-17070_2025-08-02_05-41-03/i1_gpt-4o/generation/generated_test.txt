it("should render comment and sticky note annotations", async () => {
  const { AnnotationFactory } = await import("../../src/core/annotation.js");
  const { Dict, Name } = await import("../../src/core/primitives.js");
  const { AnnotationType } = await import("../../src/shared/util.js");

  const dict = new Dict();
  dict.set("Type", Name.get("Annot"));
  dict.set("Subtype", Name.get("Text"));
  dict.set("Name", Name.get("Note"));
  dict.set("Contents", "This is a sticky note");

  const annotation = AnnotationFactory.create({
    dict,
    xref: null,
    id: "1",
    pageIndex: 0,
  });

  const expected = {
    annotationType: AnnotationType.TEXT,
    name: "Note",
    contentsObj: "This is a sticky note",
    noHTML: false,
  };

  expect(annotation.data.annotationType).toEqual(expected.annotationType);
  expect(annotation.data.name).toEqual(expected.name);
  expect(annotation.data.contentsObj).toEqual(expected.contentsObj);
  expect(annotation.data.noHTML).toEqual(expected.noHTML);
});
