it("should render comment and sticky note annotations", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const { AnnotationLayer } = await import("../../src/display/annotation_layer.js");
  const { PDFJS } = await import("../../src/shared/util.js");

  const loadingTask = getDocument({ url: "https://github.com/mozilla/pdf.js/files/12802422/Testdoc.pdf" });
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);

  const annotations = await page.getAnnotations();
  const annotationLayer = new AnnotationLayer();
  const annotationLayerDiv = document.createElement("div");

  annotationLayer.render({
    annotations,
    div: annotationLayerDiv,
    page,
    viewport: page.getViewport({ scale: 1.0 }),
    linkService: null,
    downloadManager: null,
  });

  const stickyNote = annotationLayerDiv.querySelector(".annotationLayer .popupAnnotation");
  const comment = annotationLayerDiv.querySelector(".annotationLayer .textAnnotation");

  expect(stickyNote).not.toBeNull();
  expect(comment).not.toBeNull();
});
