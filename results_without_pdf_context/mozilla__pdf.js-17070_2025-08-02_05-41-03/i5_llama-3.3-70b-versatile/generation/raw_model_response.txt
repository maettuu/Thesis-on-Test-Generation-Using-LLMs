```javascript
it("should render comment and sticky note annotations in the viewer", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { AnnotationLayer } = await import("../../src/display/annotation_layer.js");
  const { PDFPageView } = await import("../../src/web/pdf_page_view.js");
  const { PDFViewer } = await import("../../src/web/pdf_viewer.js");
  const { Util } = await import("../../src/shared/util.js");

  const pdfUrl = "./Testdoc.pdf";
  const pdfDocument = await PDFDocument.load(pdfUrl);
  const pdfPage = await pdfDocument.getPage(1);
  const pdfPageView = new PDFPageView({ pdfPage });
  const annotationLayer = new AnnotationLayer({
    pdfPage,
    viewport: pdfPageView.viewport,
  });
  const pdfViewer = new PDFViewer();
  pdfViewer.setDocument(pdfDocument);
  const annotations = await pdfPage.getAnnotations();
  const commentAnnotation = annotations.find(
    (annotation) => annotation.subtype === "Comment"
  );
  const stickyNoteAnnotation = annotations.find(
    (annotation) => annotation.subtype === "StickyNote"
  );
  const expectedAnnotations = [commentAnnotation, stickyNoteAnnotation];
  const actualAnnotations = annotationLayer.getAnnotations();
  Util.assert(
    expectedAnnotations.every((annotation) =>
      actualAnnotations.includes(annotation)
    ),
    "Comment and sticky note annotations are not rendered in the viewer"
  );
});
```
