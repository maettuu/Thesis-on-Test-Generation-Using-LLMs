it("should render comment and sticky note annotations in the viewer", async () => {
  const { AnnotationFactory } = await import("../../src/core/annotation.js");
  const { PDFDocument } = await import("../../src/core/document.js");
  const { PDFFetchStream } = await import("../../src/display/fetch_stream.js");
  const { createIdFactory } = await import("./test_utils.js");
  const pdfUrl = "https://github.com/mozilla/pdf.js/files/12802422/Testdoc.pdf";
  const pdf = await PDFDocument.load({ url: pdfUrl, cMapUrl: CMAP_URL, cMapPacked: true, standardFontDataUrl: STANDARD_FONT_DATA_URL });
  const annotationFactory = new AnnotationFactory();
  const annotations = await pdf.getAnnotations();
  const renderedAnnotations = annotations.map(annotation => annotationFactory.createAnnotation(annotation));
  const commentAnnotation = renderedAnnotations.find(annotation => annotation.data.annotationType === AnnotationType.TEXT);
  const stickyNoteAnnotation = renderedAnnotations.find(annotation => annotation.data.annotationType === AnnotationType.POPUP);
  expect(commentAnnotation).not.toBeNull();
  expect(stickyNoteAnnotation).not.toBeNull();
});
