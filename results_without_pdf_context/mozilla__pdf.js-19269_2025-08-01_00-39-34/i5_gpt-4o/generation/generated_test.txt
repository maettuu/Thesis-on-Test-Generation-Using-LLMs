it("should display top and bottom images in the PDF file", async () => {
  const { PDFDocumentProxy, getDocument } = await import("../../src/display/api.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");

  const loadingTask = getDocument(buildGetDocumentParams("test_pdf_with_images.pdf"));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);

  const operatorList = await page.getOperatorList();
  const imageOps = operatorList.fnArray.filter(fn => fn === OPS.paintImageXObject);

  const expectedImageOpsCount = 2; // Expecting top and bottom images
  const actualImageOpsCount = imageOps.length;

  expect(actualImageOpsCount).toBe(expectedImageOpsCount);
});
