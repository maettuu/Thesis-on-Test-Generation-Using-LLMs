it("should change the mouse cursor state according to the edge we resize it from for added images and drawings in PDF documents", async () => {
  const { AnnotationEditor, AnnotationEditorUIManager } = await import("../../src/display/editor/editor.js");
  const { AnnotationEditorType } = await import("../../src/shared/util.js");
  const { InkEditor } = await import("../../src/display/editor/ink.js");
  const { StampEditor } = await import("../../src/display/editor/stamp.js");

  const annotationEditorUIManager = new AnnotationEditorUIManager();
  const inkEditor = new InkEditor();
  const stampEditor = new StampEditor();

  annotationEditorUIManager.mode = AnnotationEditorType.INK;
  inkEditor.makeResizable();
  stampEditor.makeResizable();

  const expectedCursor = "nwse-resize";
  const actualCursorInk = inkEditor.div.style.cursor;
  const actualCursorStamp = stampEditor.div.style.cursor;

  expect(actualCursorInk).toBe(expectedCursor);
  expect(actualCursorStamp).toBe(expectedCursor);
});
