```javascript
it("should change mouse cursor state according to the edge we resize from", async () => {
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const { InkEditor } = await import("../../src/display/editor/ink.js");
  const { StampEditor } = await import("../../src/display/editor/stamp.js");

  const mockUIManager = { stopUndoAccumulation: jest.fn() };
  const inkEditor = new InkEditor({ uiManager: mockUIManager });
  const stampEditor = new StampEditor({ uiManager: mockUIManager });

  inkEditor.div = document.createElement("div");
  stampEditor.div = document.createElement("div");

  inkEditor.select();
  stampEditor.select();

  const inkResizers = inkEditor.div.querySelectorAll(".resizer");
  const stampResizers = stampEditor.div.querySelectorAll(".resizer");

  expect(inkResizers.length).toBeGreaterThan(0);
  expect(stampResizers.length).toBeGreaterThan(0);

  inkResizers.forEach(resizer => {
    resizer.dispatchEvent(new Event("pointerdown"));
    expect(mockUIManager.stopUndoAccumulation).toHaveBeenCalled();
  });

  stampResizers.forEach(resizer => {
    resizer.dispatchEvent(new Event("pointerdown"));
    expect(mockUIManager.stopUndoAccumulation).toHaveBeenCalled();
  });
});
```
