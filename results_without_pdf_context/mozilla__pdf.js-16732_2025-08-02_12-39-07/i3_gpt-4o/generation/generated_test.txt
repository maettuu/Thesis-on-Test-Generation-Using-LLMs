it("should change mouse cursor state according to the edge we resize from", async () => {
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");

  const uiManager = new AnnotationEditorUIManager(null, null, null, null);
  const editor = new AnnotationEditor({
    uiManager,
    parent: { pageIndex: 0, viewport: { rotation: 0, rawDims: { pageWidth: 100, pageHeight: 100, pageX: 0, pageY: 0 } } },
    id: "test-editor",
    x: 0,
    y: 0,
  });

  editor.render();
  editor.select();

  const resizers = editor.div.querySelectorAll(".resizer");
  const cursorStates = Array.from(resizers).map(resizer => window.getComputedStyle(resizer).cursor);

  const expectedCursorStates = ["nwse-resize", "nesw-resize", "nwse-resize", "nesw-resize", "ns-resize", "ew-resize", "ns-resize", "ew-resize"];
  expect(cursorStates).toEqual(expectedCursorStates);
});
