it("should not display the pin when fillAlpha is", async () => {
  const { FileAttachmentAnnotation } = await import("../../src/core/annotation.js");
  const { Dict, Name } = await import("../../src/core/primitives.js");
  const { AnnotationType } = await import("../../src/shared/util.js");

  const dict = new Dict();
  dict.set("Name", Name.get("PushPin"));
  dict.set("ca", 0);

  const annotation = new FileAttachmentAnnotation({
    dict,
    xref: null,
  });

  const expected = {
    annotationType: AnnotationType.FILEATTACHMENT,
    name: "PushPin",
    fillAlpha: 0,
  };

  expect(annotation.data).toMatchObject(expected);
});
