it("should not display the pin icon when fillAlpha is 0", async () => {
  const { FileAttachmentAnnotation } = await import("../../src/core/annotation.js");
  const { FileAttachmentAnnotationElement } = await import("../../src/display/annotation_layer.js");

  const params = {
    dict: {
      get: (key) => {
        if (key === "Name") return { name: "PushPin" };
        if (key === "ca") return 0; // fillAlpha set to 0
        return null;
      },
    },
    xref: null,
  };

  const annotation = new FileAttachmentAnnotation(params);
  const element = new FileAttachmentAnnotationElement({
    data: annotation.data,
    imageResourcesPath: "",
    linkService: { eventBus: { dispatch: () => {} } },
  });

  const container = element.render();
  const trigger = container.querySelector(".popupTriggerArea");

  const expected = null;
  const actual = trigger.querySelector("img");

  expect(actual).toBe(expected);
});
