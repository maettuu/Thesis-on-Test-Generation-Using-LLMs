it("should render FileAttachmentAnnotation with transparent icon when fillAlpha is 0", async () => {
  const { PDFDocument, PDFPage } = await import("../../src/core/document.js");
  const { AnnotationLayer } = await import("../../src/display/annotation_layer.js");
  const { PDFDataRangeTransport } = await import("../../src/display/api.js");

  const pdfData = await fetch('https://github.com/mozilla/pdf.js/files/12287878/TS354.Week.2.slides.pdf').then(response => response.arrayBuffer());
  const pdf = await PDFDocument.load(pdfData, { cMapUrl: 'https://cdn.jsdelivr.net/npm/pdfjs-dist@2.16.105/build/generic/web/cmaps/', cMapPacked: true });
  const page = await pdf.getPage(2);
  const annotationLayer = new AnnotationLayer({
    page,
    renderInteractiveForms: false,
    linkService: null,
    downloadManager: null,
    annotationStorage: null,
  });

  const annotations = await page.getAnnotations();
  const fileAttachmentAnnotation = annotations.find(annotation => annotation.subtype === 'FileAttachment');
  const fillAlpha = 0;
  fileAttachmentAnnotation.set('ca', fillAlpha);

  const expectedStyle = `filter: opacity(0%)`;
  const actualStyle = annotationLayer.render().querySelector('.popupTriggerArea').style;

  expect(actualStyle).toBe(expectedStyle);
});
