it("should render the PDF file without throwing an error", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const pdfUrl = "https://github.com/mozilla/pdf.js/files/12185373/issue16759.pdf";
  const pdfDocument = await PDFDocument.load({ url: pdfUrl });
  const expectedNumberOfPages = 1;
  const actualNumberOfPages = pdfDocument.numPages;
  expect(actualNumberOfPages).toBe(expectedNumberOfPages);
});
