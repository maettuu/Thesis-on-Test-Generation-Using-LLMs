it("should render the PDF file issue16759.pdf", async () => {
  const { PDFDocumentProxy } = await import("../../src/display/api.js");
  const { PDFDataRangeTransport } = await import("../../src/display/api.js");
  const pdfUrl = `${TEST_PDFS_PATH}issue16759.pdf`;
  const pdfDocument = await PDFDocumentProxy.load({

    cMapUrl: CMAP_URL,
    cMapPacked: true,
  });

  const viewport = firstPage.getViewport({ scale: 1 });
  const renderTask = firstPage.render({
    viewport,
    canvasContext: null,
    renderInteractive: false,
  });
  await renderTask.promise;
});
