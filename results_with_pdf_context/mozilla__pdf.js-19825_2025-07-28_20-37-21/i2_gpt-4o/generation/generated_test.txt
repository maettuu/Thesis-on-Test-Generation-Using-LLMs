it("should apply transformations correctly without creating subarrays", async () => {
  const { Util } = await import("../../src/shared/util.js");
  const { Commands } = await import("../../src/core/font_renderer.js");

  const commands = new Commands();
  commands.currentTransform = [2, 0, 0, 2, 10, 10];
  const args = [1, 1, 2, 2];
  const expected = [12, 12, 14, 14];

  commands.add("testCmd", args);

  const actual = commands.cmds[0].split("testCmd")[1].split(" ").map(Number);
  expect(actual).toEqual(expected);
});
