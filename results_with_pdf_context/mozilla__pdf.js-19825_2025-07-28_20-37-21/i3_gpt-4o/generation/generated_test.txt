it("should apply transform correctly with position offset", async () => {
  const { Util } = await import("../../src/shared/util.js");
  const point = [1, 2, 3, 4];
  const transform = [1, 0, 0, 1, 5, 6];
  const pos = 2;
  const expected = [1, 2, 8, 10];
  Util.applyTransform(point, transform, pos);
  expect(point).toEqual(expected);
});
