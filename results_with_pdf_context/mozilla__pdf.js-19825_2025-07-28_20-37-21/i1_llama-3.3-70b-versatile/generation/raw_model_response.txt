```javascript
it("should apply transform to points correctly", async () => {
  const { Util } = await import("../../src/shared/util.js");
  const transform = [2, 0, 0, 2, 1, 1];
  const points = [1, 2, 3, 4, 5, 6];
  const expectedPoints = [3, 5, 7, 9, 11, 13];
  Util.applyTransform(points, transform);
  expect(points).toEqual(expectedPoints);
});
```
