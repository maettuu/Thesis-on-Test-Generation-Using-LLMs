```javascript
it("should render the issue16759.pdf document without errors", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue16759.pdf'));
  await loadingTask.promise;
  const doc = loadingTask.doc;
  expect(doc).not.toBeNull();
  const firstPagePromise = doc.getPage(1);
  await firstPagePromise;
  const firstPage = await firstPagePromise;
  expect(firstPage).not.toBeNull();
});
```
