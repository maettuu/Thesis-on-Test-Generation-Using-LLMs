```javascript
it("should render PDF with transparency filters correctly when <base href> is defined", async () => {
  const { DOMFilterFactory } = await import("../../src/display/display_utils.js");
  const { document } = await import("../../src/shared/util.js");
  const filterFactory = new DOMFilterFactory();
  const url = "test.pdf";
  const baseUrl = "https://example.com";
  const baseElement = document.createElement("base");
  baseElement.href = baseUrl;
  document.head.appendChild(baseElement);
  const filterUrlBefore = filterFactory.addFilter([255, 255, 255]);
  document.head.removeChild(baseElement);
  const filterFactoryAfter = new DOMFilterFactory();
  const filterUrlAfter = filterFactoryAfter.addFilter([255, 255, 255]);
  expect(filterUrlBefore).not.toBe(filterUrlAfter);
  expect(filterUrlAfter).toContain(baseUrl);
});
```
