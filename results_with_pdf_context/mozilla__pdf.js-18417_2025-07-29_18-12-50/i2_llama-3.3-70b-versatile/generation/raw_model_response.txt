it("should render PDF with transparency filters correctly when <base href> is defined", async () => {
  const { PDFDocument, PDFPage } = await import("../../src/display/api.js");
  const { DOMFilterFactory } = await import("../../src/display/display_utils.js");
  const { Util } = await import("../../src/shared/util.js");

  const pdfUrl = "/test/pdfs/test.pdf";
  const pdf = await PDFDocument.load(await fetch(pdfUrl).then((res) => res.arrayBuffer()));
  const page = await pdf.getPage(1);
  const viewport = page.getViewport({ scale: 1 });
  const filterFactory = new DOMFilterFactory();

  const baseHref = document.createElement("base");
  baseHref.href = ".";
  document.head.appendChild(baseHref);

  const filterUrl = filterFactory.addFilter([Util.makeHexColor(0, 0, 0)]);
  const expectedUrl = `url(#{id})`.replace(/{id}/, `g_${filterFactory.#docId}_transfer_map_0`);

  const actualUrl = filterUrl;

  expect(actualUrl).toBe(`url(.#${`g_${filterFactory.#docId}_transfer_map_0`})`);
});
