it("should render PDF with transparency filters correctly when <base href> is defined", async () => {
  const { DOMFilterFactory } = await import("../../src/display/display_utils.js");
  const { document } = await import("../../src/shared/util.js");

  const filterFactory = new DOMFilterFactory();
  const baseUrl = "https://example.com/base";
  const id = "test-filter";
  const url = `url(#${id})`;

  const originalUrl = filterFactory.addFilter([new Uint8Array(256).fill(255)]);
  const modifiedUrl = filterFactory.addFilter([new Uint8Array(256).fill(255)]);

  document.baseURI = baseUrl;
  filterFactory = new DOMFilterFactory();
  const newUrl = filterFactory.addFilter([new Uint8Array(256).fill(255)]);

  expect(originalUrl).toBe(modifiedUrl);
  expect(newUrl).toBe(`url(${baseUrl}#${id})`);
});
