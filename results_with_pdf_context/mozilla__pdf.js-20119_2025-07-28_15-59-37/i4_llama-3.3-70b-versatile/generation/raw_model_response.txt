it("should add a comment button to the highlight toolbar", async () => {
  const { AnnotationEditorUIManager, HighlightToolbar } = await import("./tools.js");
  const { noContextMenu, stopEvent } = await import("../display_utils.js");
  const uiManager = new AnnotationEditorUIManager(null, null, null, null, null, null, null, null, null, null, null, null, null);
  uiManager.hasCommentManager = () => true;
  const toolbar = new HighlightToolbar(uiManager);
  const buttons = toolbar.render().querySelector(".buttons");
  let commentButton = null;
  for (const button of buttons.children) {
    if (button.classList.contains("commentButton")) {
      commentButton = button;
    }
  }
  expect(commentButton).not.toBeNull();
});
