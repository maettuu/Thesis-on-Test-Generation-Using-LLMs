```javascript
it("should add highlight and comment buttons to the context menu", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { FloatingToolbar } = await import("../../src/display/editor/toolbar.js");
  const uiManager = new AnnotationEditorUIManager(
    document.createElement("div"),
    document.createElement("div"),
    document.createElement("div"),
    null,
    { open: () => {} },
    null,
    { _on: () => {} },
    { annotationStorage: new Map(), filterFactory: {} },
    null,
    null,
    true,
    false,
    false,
    null,
    null,
    true
  );
  const toolbar = new FloatingToolbar(uiManager);
  const toolbarDiv = toolbar.show(document.createElement("div"), [{ x: 0, y: 0, width: 1, height: 1 }], true);
  const buttons = toolbarDiv.querySelectorAll("button");
  const buttonClasses = Array.from(buttons).map(button => button.className);
  expect(buttonClasses).toContain("commentButton");
  expect(buttonClasses).toContain("highlightButton");
});
```
