```javascript
it("should not throw an exception when parsing a JPEG image with no frame data", async () => {
  const { JpegImage } = await import("../../src/core/jpg.js");
  const jpegImage = new JpegImage();
  const imageData = Buffer.from('ffd8', 'hex');
  await expect(jpegImage.parse(imageData)).rejects.toThrowError("JpegImage.parse - no frame data found.");
});
```
