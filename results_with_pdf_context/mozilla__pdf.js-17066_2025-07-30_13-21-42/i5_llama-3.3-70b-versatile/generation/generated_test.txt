it("should render readonly form fields when AnnotationMode is ENABLED", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const pdfFile = 'issue17064_readonly.pdf';
  const loadingTask = getDocument(buildGetDocumentParams(pdfFile));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const annotationMode = 1; // ENABLED
  const renderForms = annotationMode === 2; // ENABLED_FORMS
  const { AnnotationLayer } = await import('../../src/display/annotation_layer.js');
  const annotationLayer = new AnnotationLayer({
    page: pdfPage,
    renderForms,
    annotationMode,
  });
  const textWidgetAnnotations = annotationLayer.getAnnotations().filter(annotation => annotation.subtype === 'Widget');
  const textWidgetAnnotation = textWidgetAnnotations.find(annotation => annotation.fieldValue === 'Foo');
  const isRenderable = textWidgetAnnotation.isRenderable;
  expect(isRenderable).toBe(true);
});
