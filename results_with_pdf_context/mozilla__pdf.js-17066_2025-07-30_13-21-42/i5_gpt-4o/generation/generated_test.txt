it("should render readonly form field when AnnotationMode is ENABLED", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue17064_readonly.pdf'));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const annotations = await page.getAnnotations({ intent: 'display' });

  const readOnlyField = annotations.find(annotation => annotation.fieldName === 'ReadOnlyField');
  const isRenderable = readOnlyField && readOnlyField.hasOwnCanvas;

  expect(isRenderable).toBe(true);
});
