it("should render readonly TextWidget-annotations when forms are disabled", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const { AnnotationMode } = await import('../../src/shared/util.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue17064_readonly.pdf'), { annotationMode: AnnotationMode.ENABLED });
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const annotationLayer = await pdfPage.createAnnotationLayer();
  const textWidgetAnnotations = annotationLayer.annotations.filter(annotation => annotation instanceof annotationLayer.constructor.TextWidgetAnnotationElement);
  const readonlyTextWidgetAnnotations = textWidgetAnnotations.filter(annotation => annotation.data.readOnly);
  const expected = true;
  const actual = readonlyTextWidgetAnnotations.some(annotation => annotation.render().childElementCount > 0);
  expect(actual).toBe(expected);
});
