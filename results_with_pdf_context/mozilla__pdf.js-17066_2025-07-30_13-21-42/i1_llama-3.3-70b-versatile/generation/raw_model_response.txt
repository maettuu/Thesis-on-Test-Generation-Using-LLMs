it("should render read-only form fields when AnnotationMode is ENABLED", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue17064_readonly.pdf', { annotationMode: 1 }));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const pageViewport = pdfPage.getViewport({ scale: 1 });
  const renderTask = pdfPage.render({
    viewport: pageViewport,
    annotationMode: 1,
  });
  const renderResult = await renderTask.promise;
  const textWidgetAnnotations = renderResult.annotations.filter(annotation => annotation.subtype === 'Widget');
  const readOnlyTextWidgetAnnotations = textWidgetAnnotations.filter(annotation => annotation.flags & 0x0001);
  const isRenderable = readOnlyTextWidgetAnnotations.length > 0;
  expect(isRenderable).toBe(true);
});
