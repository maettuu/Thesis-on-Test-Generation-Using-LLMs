it("should render password fields as password fields, not text fields", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue19389.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const annotationStorage = new (await import('../../src/display/annotation_storage.js')).AnnotationStorage();
  const annotations = await pdfPage.getAnnotations({ annotationStorage });
  const widgetAnnotation = annotations.find(annotation => annotation instanceof (await import('../../src/core/annotation.js')).WidgetAnnotation);
  const textWidgetAnnotationElement = new (await import('../../src/display/annotation_layer.js')).TextWidgetAnnotationElement({
    annotation: widgetAnnotation,
    annotationStorage,
    renderForms: true,
    linkService: {},
    enableScripting: false,
  });
  const element = textWidgetAnnotationElement.render();
  const inputElement = element.querySelector('input');
  expect(inputElement.type).toBe('password');
});
