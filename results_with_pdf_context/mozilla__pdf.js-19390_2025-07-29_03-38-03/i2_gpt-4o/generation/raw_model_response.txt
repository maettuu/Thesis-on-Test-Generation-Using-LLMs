it("should render password fields with obscured characters", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue19389.pdf'));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const annotations = await page.getAnnotations();
  const passwordField = annotations.find(ann => ann.fieldName === "PasswordField");
  const { AnnotationFieldFlag } = await import('../../src/shared/util.js');
  const isPassword = (passwordField.fieldFlags & AnnotationFieldFlag.PASSWORD) !== 0;
  expect(isPassword).toBe(true);
});
