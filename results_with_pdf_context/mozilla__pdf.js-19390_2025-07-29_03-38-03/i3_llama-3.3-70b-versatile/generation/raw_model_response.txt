it("should render password fields as password fields", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue19389.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const annotationStorage = new AnnotationStorage();
  const renderForms = true;
  const annotationLayer = new AnnotationLayer({
    div: document.createElement('div'),
    page: pdfPage,
    viewport: pdfPage.getViewPort({ scale: 1 }),
    linkService: null,
    downloadManager: null,
    annotationStorage,
    renderForms,
  });
  await annotationLayer.render({
    annotations: await pdfPage.getAnnotations({ intent: 'display' }),
  });
  const passwordField = annotationLayer.getEditableAnnotation('passwordFieldId');
  const inputElement = passwordField.container.querySelector('input');
  expect(inputElement.type).toBe('password');
});
