  const annotationLayer = await import('../../src/display/annotation_layer.js');

  const annotationStorage = await import('../../src/display/annotation_storage.js');
  const annotations = await pdfPage.getAnnotations();
  const annotationLayerElement = new annotationLayer.TextWidgetAnnotationElement(annotations[0]);
  const inputElement = annotationLayerElement.element;
  const expectedType = 'password';
  const actualType = inputElement.type;
