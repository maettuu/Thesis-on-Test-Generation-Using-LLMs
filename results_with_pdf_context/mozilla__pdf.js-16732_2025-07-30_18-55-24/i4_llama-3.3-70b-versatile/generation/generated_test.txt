it("should change the mouse cursor state according to the edge we resize from for added images and drawings in PDF documents", async () => {
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const editor = new AnnotationEditor({
    parent: {
      pageIndex: 0,
      pageDimensions: [100, 100],
      getNextId: () => "id",
      add: () => {},
      remove: () => {},
    },
    uiManager: {
      viewParameters: {
        realScale: 1,
        rotation: 0,
      },
    },
    id: "id",
    name: "editor",
  });
  editor.width = 50;
  editor.height = 50;
  editor.x = 25;
  editor.y = 25;
  editor.render();
  editor.makeResizable();
  const resizer = editor.#resizersDiv.children[0];
  const event = new PointerEvent("pointerdown", {
    bubbles: true,
    cancelable: true,
    clientX: 30,
    clientY: 30,
  });
  resizer.dispatchEvent(event);
  const resizingClassName = `resizing${"topLeft".charAt(0).toUpperCase()}${"topLeft".slice(1)}`;
  expect(editor.parent.div.classList.contains(resizingClassName)).toBe(true);
});
