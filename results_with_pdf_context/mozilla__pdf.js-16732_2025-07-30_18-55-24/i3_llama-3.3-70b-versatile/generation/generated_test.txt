it("should change the mouse cursor state according to the edge we resize from for added images and drawings in PDF documents", async () => {
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const editor = new AnnotationEditor({
    parent: { pageIndex: 0 },
    id: "test-editor",
    uiManager: {},
  });
  editor.render();
  const resizerPointerdown = jest.fn();
  editor.#resizerPointerdown = resizerPointerdown;
  const event = { preventDefault: jest.fn(), clientX: 10, clientY: 10 };
  editor.div.dispatchEvent(new PointerEvent("pointerdown", event));
  expect(resizerPointerdown).toHaveBeenCalledTimes(1);
  expect(event.preventDefault).toHaveBeenCalledTimes(1);
});
