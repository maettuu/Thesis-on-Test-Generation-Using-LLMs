```javascript
it("should change mouse cursor state according to the edge when resizing images and drawings in PDF", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const { InkEditor } = await import("../../src/display/editor/ink.js");
  const { StampEditor } = await import("../../src/display/editor/stamp.js");

  const uiManager = new AnnotationEditorUIManager();
  const inkEditor = new InkEditor({ uiManager });
  const stampEditor = new StampEditor({ uiManager });

  inkEditor.select();
  stampEditor.select();

  const inkResizers = inkEditor.div.querySelectorAll(".resizer");
  const stampResizers = stampEditor.div.querySelectorAll(".resizer");

  const expectedInkResizerCount = inkEditor.isResizable ? 4 : 0;
  const expectedStampResizerCount = stampEditor.isResizable ? 8 : 0;

  const actualInkResizerCount = inkResizers.length;
  const actualStampResizerCount = stampResizers.length;

  expect(actualInkResizerCount).toBe(expectedInkResizerCount);
  expect(actualStampResizerCount).toBe(expectedStampResizerCount);
});
```
