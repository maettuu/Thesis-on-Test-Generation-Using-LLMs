it("should change mouse cursor state according to the edge when resizing images and drawings", async () => {
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const { InkEditor } = await import("../../src/display/editor/ink.js");
  const { StampEditor } = await import("../../src/display/editor/stamp.js");

  const mockUIManager = { viewParameters: { rotation: 0 } };
  const mockParent = { viewport: { rotation: 0, rawDims: { pageWidth: 100, pageHeight: 100, pageX: 0, pageY: 0 } }, pageIndex: 0 };

  const inkEditor = new InkEditor({ parent: mockParent, uiManager: mockUIManager, x: 0, y: 0 });
  const stampEditor = new StampEditor({ parent: mockParent, uiManager: mockUIManager, x: 0, y: 0, bitmapUrl: "" });

  inkEditor.select();
  stampEditor.select();

  const inkResizers = inkEditor.div.querySelectorAll(".resizer");
  const stampResizers = stampEditor.div.querySelectorAll(".resizer");

  const expectedInkCursorStates = ["nwse-resize", "nesw-resize", "nwse-resize", "nesw-resize"];
  const expectedStampCursorStates = ["nwse-resize", "nesw-resize", "nwse-resize", "nesw-resize"];

  const actualInkCursorStates = Array.from(inkResizers).map(resizer => getComputedStyle(resizer).cursor);
  const actualStampCursorStates = Array.from(stampResizers).map(resizer => getComputedStyle(resizer).cursor);

  expect(actualInkCursorStates).toEqual(expectedInkCursorStates);
  expect(actualStampCursorStates).toEqual(expectedStampCursorStates);
});
