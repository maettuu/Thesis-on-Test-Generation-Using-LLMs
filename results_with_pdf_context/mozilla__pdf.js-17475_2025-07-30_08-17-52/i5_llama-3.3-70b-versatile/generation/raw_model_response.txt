it("should display highlight on cropped pdf when saving/printing", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1872721.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const { AnnotationLayerBuilder } = await import('../../web/annotation_layer_builder.js');
  const annotationLayerBuilder = new AnnotationLayerBuilder();
  const annotationLayer = await annotationLayerBuilder.createAnnotationLayerForPage(pdfPage);
  const highlights = annotationLayer.getAnnotations().filter(annotation => annotation.type === 'highlight');
  const expectedHighlightCount = 1;
  const actualHighlightCount = highlights.length;
  expect(actualHighlightCount).toBe(expectedHighlightCount);
});
