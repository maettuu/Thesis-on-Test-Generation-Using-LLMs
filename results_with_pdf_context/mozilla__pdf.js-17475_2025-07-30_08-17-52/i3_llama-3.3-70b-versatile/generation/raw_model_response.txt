it("should display highlight on cropped pdf when saving/printing", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1872721.pdf'));
  const pdfDoc = await loadingTask.promise;
  const pdfPage = await pdfDoc.getPage(1);
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const { HighlightEditor } = await import("../../src/display/editor/highlight.js");
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { DrawLayer } = await import("../../src/display/draw_layer.js");
  const uiManager = new AnnotationEditorUIManager();
  const drawLayer = new DrawLayer();
  const highlightEditor = new HighlightEditor({
    pageIndex: 0,
    page: pdfPage,
    parent: drawLayer,
    uiManager: uiManager,
    boxes: [{ x: 0.1, y: 0.1, width: 0.2, height: 0.2 }],
  });
  const serializedData = highlightEditor.serialize();
  const quadPoints = serializedData.quadPoints;
  const outlines = serializedData.outlines;
  const expectedQuadPoints = [10, 710, 10, 690, 30, 690, 30, 710];
  const expectedOutlines = [[10, 690], [30, 690], [30, 710], [10, 710]];
  expect(quadPoints).toEqual(expectedQuadPoints);
  expect(outlines).toEqual(expectedOutlines);
});
