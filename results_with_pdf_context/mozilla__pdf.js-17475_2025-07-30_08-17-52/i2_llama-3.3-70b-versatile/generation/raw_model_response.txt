it("should highlight text correctly when saving/printing after applying the patch", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const { PDFPageProxy } = await import('../../src/display/api.js');
  const { AnnotationEditor } = await import('./editor.js');
  const { HighlightEditor } = await import('./highlight.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1872721.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const annotationEditor = new AnnotationEditor({ page: pdfPage });
  const highlightEditor = new HighlightEditor({ page: pdfPage, color: 'yellow' });
  const quadPoints = highlightEditor.serialize().quadPoints;
  const expectedQuadPoints = [10, 700, 50, 700, 50, 750, 10, 750];
  expect(quadPoints).toEqual(expectedQuadPoints);
});
