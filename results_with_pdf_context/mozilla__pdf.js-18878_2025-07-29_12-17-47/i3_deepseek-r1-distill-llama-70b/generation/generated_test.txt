it("should swallow extra data errors in asyncGetBytes", async () => {
  const { FlateStream } = await import("../../src/core/flate_stream.js");
  const { StringStream } = await import("../../src/core/stream.js");
  const bytes = new Uint8Array([0x78, 0x9C, 0x00, 0x00, 0x00]);
  const dataStr = String.fromCharCode(...bytes);
  const stream = new StringStream(dataStr);
  stream.dict = {};
  const flateStream = new FlateStream(stream);
  const result = await flateStream.asyncGetBytes();
  chai.expect(result).to.be.null;
});
