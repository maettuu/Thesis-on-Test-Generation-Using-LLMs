it("should not block the UI when changing the font size in a text editor", async () => {
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const uiManager = new AnnotationEditorUIManager();
  const editor = new AnnotationEditor({ uiManager, parent: null, id: "test" });
  const moveInDOMSpy = jest.spyOn(editor, "moveInDOM");
  editor.moveInDOM();
  expect(moveInDOMSpy).toHaveBeenCalledTimes(1);
  await new Promise(resolve => setTimeout(resolve, 0));
  expect(moveInDOMSpy).toHaveBeenCalledTimes(1);
});
