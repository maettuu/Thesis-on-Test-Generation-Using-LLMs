it("should delay DOM movement when changing font size with slider", async () => {
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const mockParent = {
    moveEditorInDOM: jasmine.createSpy("moveEditorInDOM"),
    pageIndex: 0,
    viewport: {
      rotation: 0,
      rawDims: { pageWidth: 100, pageHeight: 100, pageX: 0, pageY: 0 },
    },
  };
  const editor = new AnnotationEditor({
    parent: mockParent,
    id: "test-editor",
    name: "Test Editor",
    uiManager: {},
    x: 0,
    y: 0,
    isCentered: false,
  });

  editor.moveInDOM();
  expect(mockParent.moveEditorInDOM).not.toHaveBeenCalled();

  await new Promise((resolve) => setTimeout(resolve, 10));
  expect(mockParent.moveEditorInDOM).toHaveBeenCalledWith(editor);
});
