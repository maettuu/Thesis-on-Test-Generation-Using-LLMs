it("should change font size smoothly with the slider in a text editor", async () => {
  const { PDFDocument, PDFPageProxy } = await import("../../src/display/api.js");
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const { PDFPageView } = await import("../../src/web/pdf_page_view.js");

  const pdfDoc = await PDFDocument.load(await (await import("../../src/display/fetch_stream.js")).PDFFetchStream.fetch("path_to_pdf_file"));
  const pdfPage = pdfDoc.getPage(1);
  const pageView = new PDFPageView({ page: pdfPage });
  const annotationEditor = new AnnotationEditor({ pageView });

  const fontSizeBefore = annotationEditor.fontSize;
  annotationEditor.moveInDOM();
  const fontSizeAfter = annotationEditor.fontSize;

  annotationEditor.fontSize = 20;
  annotationEditor.moveInDOM();
  const newFontSize = annotationEditor.fontSize;

  expect(newFontSize).toBe(20);
});
