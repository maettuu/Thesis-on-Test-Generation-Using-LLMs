```javascript
it("should delay DOM movement when changing font size with slider", async () => {
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const editor = new AnnotationEditor({
    parent: { moveEditorInDOM: jest.fn(), pageIndex: 0 },
    id: "test-editor",
    uiManager: { viewParameters: { rotation: 0 } },
    x: 0,
    y: 0,
  });

  editor.moveInDOM();
  expect(editor.parent.moveEditorInDOM).not.toHaveBeenCalled();

  await new Promise(resolve => setTimeout(resolve, 0));
  expect(editor.parent.moveEditorInDOM).toHaveBeenCalled();
});
```
