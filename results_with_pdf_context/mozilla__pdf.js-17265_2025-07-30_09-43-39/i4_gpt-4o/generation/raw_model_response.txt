```javascript
it("should load the module script without MIME type error", async () => {
  const { createCDNWrapper } = await import("../../src/display/api.js");
  const url = "https://mozilla.github.io/pdf.js/build/pdf.worker.mjs";
  const objectUrl = createCDNWrapper(url);
  const expectedType = "text/javascript";
  const actualType = await fetch(objectUrl).then(response => response.headers.get("Content-Type"));
  expect(actualType).toBe(expectedType);
});
```
