it("should render text with correct spacing for embedded WinAnsiEncoding font", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const doc = await getDocument(TEST_PDFS_PATH + "/PDFcustomFontTest.pdf").promise;
  const page = await doc.getPage(1);
  const textContent = await page.getTextContent();
  const items = textContent.items;
  for (let i = 1; i < items.length; i++) {
    if (items[i].transform[4] <= items[i - 1].transform[4]) {
      throw new Error("Overlapping or non-increasing x-coordinate detected");
    }
  }
});
