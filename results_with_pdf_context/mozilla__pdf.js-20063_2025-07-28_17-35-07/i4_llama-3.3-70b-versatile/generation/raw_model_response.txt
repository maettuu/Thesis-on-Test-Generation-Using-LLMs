it("should render PolygonAnnotation with fill color", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue20062.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const operatorList = await pdfPage.getOperatorList({
    intent: 'display',
  });
  const renderTasks = await pdfPage.getRenderTasks({
    intent: 'display',
    annotationFlags: 0,
  });
  const annotationFlags = renderTasks.annotations[0].flags;
  const annotationColor = renderTasks.annotations[0].color;
  const annotationFillColor = renderTasks.annotations[0].fillColor;
  assert.ok(annotationFlags & 4, "Annotation is not rendered with fill color");
  assert.deepEqual(annotationColor, [0, 0, 0, 1], "Annotation color is incorrect");
  assert.deepEqual(annotationFillColor, [1, 0, 0, 1], "Annotation fill color is incorrect");
});
