it("should render polygon annotation with fill color", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue20062.pdf'));
  const pdfDoc = await loadingTask.promise;
  const page = await pdfDoc.getPage(1);
  const annotations = await page.getAnnotations();
  const polygonAnnotation = annotations[0];
  const operator = await page.render({
    annotation: polygonAnnotation,
    viewport: page.getViewport({ scale: 1 }),
  });
  const expectedOperator = "f";
  expect(operator.pointsCallback[operator.pointsCallback.length - 1]).toBe(expectedOperator);
});
