it("should render PolygonAnnotation with fill color in PDF.js web viewer", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");
  const loadingTask = getDocument(buildGetDocumentParams("issue20062.pdf"));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const annotations = await page.getAnnotations();
  const polygonAnnotation = annotations.find(
    annotation => annotation.subtype === "Polygon"
  );
  expect(polygonAnnotation).toBeDefined();
  expect(polygonAnnotation.fillColor).not.toBeNull();
});
