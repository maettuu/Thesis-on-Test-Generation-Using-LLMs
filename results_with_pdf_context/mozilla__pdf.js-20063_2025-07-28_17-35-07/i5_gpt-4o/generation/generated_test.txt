it("should render PolygonAnnotation with both fill and outline in PDFjs viewer", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue20062.pdf'));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const annotations = await page.getAnnotations();

  const polygonAnnotation = annotations.find(ann => ann.subtype === 'Polygon');
  const expectedFillColor = [1, 0, 0]; // Assuming the fill color is red (RGB)
  const expectedStrokeColor = [0, 0, 0]; // Assuming the stroke color is black (RGB)

  const actualFillColor = polygonAnnotation.fillColor;
  const actualStrokeColor = polygonAnnotation.color;

  expect(actualFillColor).toEqual(expectedFillColor);
  expect(actualStrokeColor).toEqual(expectedStrokeColor);
});
