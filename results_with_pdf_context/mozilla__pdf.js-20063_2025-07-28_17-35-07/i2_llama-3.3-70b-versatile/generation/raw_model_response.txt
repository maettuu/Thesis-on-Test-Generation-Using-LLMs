it("should render the fill color of a PolygonAnnotation", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue20062.pdf'));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const operatorList = await page.getOperatorList();
  const annotationList = await page.getAnnotations();
  const polygonAnnotation = annotationList.find(annotation => annotation instanceof PolygonAnnotation);
  const renderTask = page.render({
    annotationMode: 'visible',
    viewport: page.getViewport({ scale: 1 }),
    renderInteractive: false,
  });
  await renderTask.promise;
  const canvasContext = renderTask.canvas.getContext('2d');
  const fillColor = getRgbColor(polygonAnnotation.dict.getArray("IC"), null);
  const expectedFillColor = getPdfColorArray(fillColor);
  const actualFillColor = canvasContext.getImageData(10, 10, 1, 1).data;
  expect(actualFillColor).toEqual(expectedFillColor);
});
