it("should support highlighting images of text in addition to text", async () => {
  const { DrawLayer } = await import("../../src/display/draw_layer.js");
  const { AnnotationEditorLayer } = await import("../../src/display/editor/annotation_editor_layer.js");
  const { HighlightEditor } = await import("../../src/display/editor/highlight.js");
  const { DOMSVGFactory } = await import("../../src/display/display_utils.js");

  const svgFactory = new DOMSVGFactory();
  const drawLayer = new DrawLayer({ pageIndex: 0 });
  const annotationEditorLayer = new AnnotationEditorLayer({ pageIndex: 0 });
  const highlightEditor = new HighlightEditor({
    parent: annotationEditorLayer,
    color: "#FF0000",
    opacity: 0.5,
    highlightId: 1,
    highlight: {
      getOutlines: () => ({
        box: { x: 0.1, y: 0.1, width: 0.8, height: 0.8 },
        toSVGPath: () => "M0,0 L1,0 L1,1 L0,1 Z",
      }),
    },
    clipPathId: "clipPathId",
  });

  const expectedSVGPath = "M0,0 L1,0 L1,1 L0,1 Z";
  const actualSVGPath = highlightEditor.#highlightOutlines.toSVGPath();

  expect(actualSVGPath).toBe(expectedSVGPath);
});
