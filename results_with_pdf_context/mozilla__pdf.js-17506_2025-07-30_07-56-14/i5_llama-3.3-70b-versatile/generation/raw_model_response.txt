it("should highlight images of text in addition to text", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { PDFPage } = await import("../../src/core/document.js");
  const { AnnotationEditorLayer } = await import("../../src/display/editor/annotation_editor_layer.js");
  const { DrawLayer } = await import("../../src/display/draw_layer.js");
  const { HighlightEditor } = await import("../../src/display/editor/highlight.js");
  const { Outliner } = await import("../../src/display/editor/outliner.js");

  const pdfDoc = await PDFDocument.load(await (await import("../../src/display/api.js")).getDocument({ url: "path_to_pdf_file" }));
  const pdfPage = pdfDoc.getPage(1);
  const viewport = pdfPage.getViewport({ scale: 1 });
  const annotationEditorLayer = new AnnotationEditorLayer({ pdfPage, viewport, renderInteractiveForms: false });
  const drawLayer = new DrawLayer({ pageIndex: 0 });
  const highlightEditor = new HighlightEditor({ parent: annotationEditorLayer, pageIndex: 0 });

  const expectedOutline = new Outliner();
  expectedOutline.addPoint(10, 10);
  expectedOutline.addPoint(20, 20);

  const actualOutline = highlightEditor.getOutline();
  const actualPath = actualOutline.toSVGPath();

  const expectedPath = "M10 10L20 20";
  expect(actualPath).toBe(expectedPath);
});
