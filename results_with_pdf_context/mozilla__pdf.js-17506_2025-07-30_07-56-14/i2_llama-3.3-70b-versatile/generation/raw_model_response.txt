it("should highlight images of text in addition to text", async () => {
  const { DrawLayer } = await import("../../src/display/draw_layer.js");
  const { AnnotationEditorLayer } = await import("../../src/display/editor/annotation_editor_layer.js");
  const { HighlightEditor } = await import("../../src/display/editor/highlight.js");
  const { FreeOutliner } = await import("../../src/display/editor/outliner.js");

  const drawLayer = new DrawLayer({ pageIndex: 1 });
  const annotationEditorLayer = new AnnotationEditorLayer({
    uiManager: {},
    pageIndex: 1,
    div: document.createElement("div"),
    accessibilityManager: {},
    annotationLayer: {},
    drawLayer,
    textLayer: {},
    viewport: {},
    l10n: {},
  });

  const freeOutliner = new FreeOutliner({ x: 10, y: 10 }, [0, 0, 100, 100], 1, 5);
  freeOutliner.add({ x: 20, y: 20 });
  freeOutliner.add({ x: 30, y: 30 });

  const highlightId = drawLayer.highlight(freeOutliner, "red", 0.5, true);
  const highlightEditor = new HighlightEditor({
    parent: annotationEditorLayer,
    id: highlightId.id,
    x: 10,
    y: 10,
    uiManager: {},
    isCentered: false,
    highlightId: highlightId.id,
    highlight: freeOutliner,
    clipPathId: highlightId.clipPathId,
  });

  const expectedSvgPath = freeOutliner.toSVGPath();
  const actualSvgPath = drawLayer.highlight(freeOutliner, "red", 0.5, true);

  expect(actualSvgPath).toEqual(expectedSvgPath);
});
