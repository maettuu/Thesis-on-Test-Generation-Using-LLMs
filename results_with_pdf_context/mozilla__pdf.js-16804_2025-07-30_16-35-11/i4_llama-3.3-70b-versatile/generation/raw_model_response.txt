it("should remove embedded images from annotations", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue16800.pdf'));
  const pdfDoc = await loadingTask.promise;
  const pdfPage = await pdfDoc.getPage(2);
  const annotation = await pdfPage.getAnnotations();
  const annotationData = annotation[0];
  const { Annotation } = await import('../../src/core/annotation.js');
  const annotationInstance = new Annotation({
    dict: annotationData,
    xref: pdfDoc.xref,
    pdfManager: pdfDoc.pdfManager,
    idFactory: pdfDoc.idFactory,
  });
  const expectedAppearance = null;
  const actualAppearance = annotationInstance.appearance;
  expect(actualAppearance).toBe(expectedAppearance);
});
