it("should render file attachment annotation without pin when fillAlpha is", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue16800.pdf'));
  const pdfDoc = await loadingTask.promise;
  const page = await pdfDoc.getPage(2);
  const annotationLayer = await import('../../src/display/annotation_layer.js');
  const annotationLayerInstance = new annotationLayer.AnnotationLayer({
    page,
    viewport: page.getViewport({ scale: 1 }),
    renderInteractiveForms: false,
  });
  const annotations = await annotationLayerInstance.render();
  const fileAttachmentAnnotation = annotations.find(annotation => annotation instanceof annotationLayer.FileAttachmentAnnotationElement);
  const trigger = fileAttachmentAnnotation.#trigger;
  const expectedStyle = 'filter: opacity(0%);';
  expect(trigger.style).toBe(expectedStyle);
});
