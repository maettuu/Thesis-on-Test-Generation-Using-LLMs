it("should render FileAttachmentAnnotation with correct opacity", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue16800.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(2);
  const annotationLayer = await import('../../src/display/annotation_layer.js');
  const annotationLayerInstance = new annotationLayer.AnnotationLayer({
    page: pdfPage,
    viewport: pdfPage.getViewport({ scale: 1 }),
    renderInteractiveForms: false,
  });
  const annotations = await pdfPage.getAnnotations();
  const fileAttachmentAnnotation = annotations.find(annotation => annotation instanceof annotationLayer.FileAttachmentAnnotation);
  const trigger = annotationLayerInstance.renderAnnotation(fileAttachmentAnnotation);
  const expectedOpacity = 0.5; // assuming fill-alpha is 0.5 in the pdf
  const actualOpacity = parseFloat(trigger.style.filter.match(/opacity\((\d+)%/)[1]) / 100);
  expect(actualOpacity).toBeCloseTo(expectedOpacity);
});
