it("should remove embedded images", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue16800.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(2);
  const annotations = await pdfPage.getAnnotations();
  const annotation = annotations[0];
  const annotationElement = new AnnotationElement({
    data: annotation,
    layer: document.createElement('div'),
    linkService: null,
    downloadManager: null,
    annotationStorage: null,
    renderForms: true,
    svgFactory: null,
  });
  const renderedAnnotation = annotationElement.render();
  const image = renderedAnnotation.querySelector('img');
  expect(image).toBeNull();
});
