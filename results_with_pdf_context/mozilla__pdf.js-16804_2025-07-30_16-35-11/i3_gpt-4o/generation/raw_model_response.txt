it("should not render the pin when fillAlpha is 0", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");
  const loadingTask = getDocument(buildGetDocumentParams("issue16800.pdf"));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(2);
  const annotations = await page.getAnnotations();
  const fileAttachmentAnnotation = annotations.find(
    ann => ann.subtype === "FileAttachment"
  );
  expect(fileAttachmentAnnotation).toBeDefined();
  expect(fileAttachmentAnnotation.fillAlpha).toBe(0);
});
