it("should render comments and sticky notes in the viewer", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue17069.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const annotations = await pdfPage.getAnnotations();
  const annotationTypes = annotations.map(annotation => annotation.data.annotationType);
  const expectedTypes = [6, 7]; // 6: Text, 7: Popup
  expect(annotationTypes).toEqual(expect.arrayContaining(expectedTypes));
});
