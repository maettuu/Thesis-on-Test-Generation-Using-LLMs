it("should render comment and sticky note annotations", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue17069.pdf'));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const annotations = await page.getAnnotations();
  const hasComment = annotations.some(annotation => annotation.subtype === 'Text');
  const hasStickyNote = annotations.some(annotation => annotation.subtype === 'Popup');
  expect(hasComment).toBe(true);
  expect(hasStickyNote).toBe(true);
});
