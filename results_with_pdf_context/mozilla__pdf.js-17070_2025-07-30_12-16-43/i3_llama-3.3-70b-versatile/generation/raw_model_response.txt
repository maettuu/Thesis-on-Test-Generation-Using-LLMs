it("should render comment and sticky note annotations in the viewer", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue17069.pdf'));
  const doc = await loadingTask.promise;
  const page = await doc.getPage(1);
  const annotations = await page.getAnnotations();
  const annotation0 = annotations[0];
  const annotation1 = annotations[1];
  expect(annotation0.data.annotationType).toBe(AnnotationType.STICKY_NOTE);
  expect(annotation1.data.annotationType).toBe(AnnotationType.STICKY_NOTE);
  const renderTask = page.render({
    annotationMode: 'visible',
  });
  await renderTask.promise;
  expect(renderTask.annotationCanvasContext).not.toBeNull();
});
