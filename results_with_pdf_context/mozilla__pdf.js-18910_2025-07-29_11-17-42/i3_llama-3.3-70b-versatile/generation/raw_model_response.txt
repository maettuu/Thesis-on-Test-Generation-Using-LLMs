it("should speed up decoding images in PDFs using ImageDecoder", async () => {
  const { PDFImage, JpegStream } = await import("../../src/core/image.js");
  const { JpegImage } = await import("../../src/core/jpg.js");
  const xref = new XRefMock();
  const image = new JpegStream(new StringStream(), null, {});
  const pdfImage = new PDFImage({
    xref,
    res: {},
    image,
    isInline: false,
  });
  const originalWidth = 100;
  const originalHeight = 100;
  const drawWidth = originalWidth;
  const drawHeight = originalHeight;
  const imageData = await pdfImage.createImageData(false, true);
  const imageDecoderSupported = await JpegStream.canUseImageDecoder();
  if (imageDecoderSupported) {
    expect(imageData.data).toBeNull();
    expect(imageData.bitmap).not.toBeNull();
  } else {
    expect(imageData.data).not.toBeNull();
    expect(imageData.bitmap).toBeNull();
  }
});
