it("should use ImageDecoder for JPEG images", async () => {
  const { JpegStream } = await import("../../src/core/jpeg_stream.js");
  const { BaseStream } = await import("../../src/core/base_stream.js");

  const mockStream = new BaseStream();
  const jpegStream = new JpegStream(mockStream, null, null);

  const canUseImageDecoder = await JpegStream.canUseImageDecoder;
  const transferableImage = await jpegStream.getTransferableImage();

  const expected = canUseImageDecoder ? "object" : null;
  const actual = transferableImage ? typeof transferableImage : null;

  expect(actual).toBe(expected);
});
