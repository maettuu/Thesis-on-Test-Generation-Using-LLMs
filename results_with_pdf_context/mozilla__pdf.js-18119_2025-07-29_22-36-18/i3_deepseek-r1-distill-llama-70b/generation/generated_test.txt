it("should correctly handle char codes greater than 2311 in adjustMapping", async () => {
  const { adjustMapping } = await import("../../src/core/fonts.js");
  // create a mapping with a key greater than 2147483647
  const inputMapping = { "3000000000": 42 };
  const hasGlyph = glyphId => true;
  const newGlyphZeroId = 0;
  const toUnicode = new Map();
  const result = adjustMapping(inputMapping, hasGlyph, newGlyphZeroId, toUnicode);
  if (result.toFontChar["3000000000"] === undefined) {
    throw new Error("Failed to map char code '3000000000' correctly.");
  }
});
