it("should not throw an error when OpenJPEG decode fails", async () => {
  const { JpxImage } = await import("../../src/core/jpx.js");
  const { JpxError } = await import("../../src/core/jpx.js");
  const invalidData = new Uint8Array([0x00, 0x00, 0x00, 0x00]);
  const ignoreColorSpace = false;
  let error;
  try {
    JpxImage.decode(invalidData, ignoreColorSpace);
  } catch (e) {
    error = e;
  }
  expect(error).toBeInstanceOf(JpxError);
  expect(error.message).not.toBe("JPX decode failed");
});
