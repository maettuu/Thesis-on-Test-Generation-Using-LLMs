it("should throw JpxError with Unknown format message when OpenJPEG logs Unknown format", async () => {
  const { JpxImage } = await import("../../src/core/jpx.js");
  const invalidData = new Uint8Array([0, 1, 2, 3]); // Example invalid data
  let errorMessage = "";

  try {
    JpxImage.decode(invalidData, false);
  } catch (error) {
    errorMessage = error.message;
  }

  const expectedMessage = "OpenJPEG: Unknown format";
  expect(errorMessage).toBe(expectedMessage);
});
