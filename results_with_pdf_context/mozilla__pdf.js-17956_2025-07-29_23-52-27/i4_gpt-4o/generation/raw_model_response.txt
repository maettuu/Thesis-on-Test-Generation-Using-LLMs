it("should throw JpxError with message from OpenJPEG without logging to console", async () => {
  const { JpxImage } = await import("../../src/core/jpx.js");
  const invalidData = new Uint8Array([0, 1, 2, 3]); // Invalid JPX data
  let errorMessage = "";
  try {
    JpxImage.decode(invalidData, false);
  } catch (error) {
    errorMessage = error.message;
  }
  const expectedMessage = "JPX error: Unknown format";
  expect(errorMessage).toBe(expectedMessage);
});
