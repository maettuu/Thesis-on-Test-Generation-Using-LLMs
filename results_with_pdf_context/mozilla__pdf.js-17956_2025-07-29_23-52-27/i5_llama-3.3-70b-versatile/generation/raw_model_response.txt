```javascript
it("should not log 'OpenJPEG: Unknown format' when decoding JPX image", async () => {
  const { JpxImage } = await import("../../src/core/jpx.js");
  const consoleSpy = jest.spyOn(console, 'log');
  const imageData = new Uint8Array([0x00, 0x00, 0x00, 0x0C, 0x6A, 0x50, 0x20, 0x20]);
  try {
    JpxImage.decode(imageData, false);
  } catch (error) {
    if (error instanceof JpxImage.JpxError) {
      expect(consoleSpy).not.toHaveBeenCalledWith('OpenJPEG: Unknown format');
    } else {
      throw error;
    }
  }
});
```
