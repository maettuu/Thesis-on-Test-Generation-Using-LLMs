it("should load a SVG image with a blob url", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const imageManager = new ImageManager();
  const blob = new Blob(['<svg></svg>'], {type: 'image/svg+xml'});
  const key = 'image_1';
  const rawData = blob;
  const dataBeforePatch = await imageManager.#get(key, rawData);
  expect(dataBeforePatch).toBeNull();

  const { ImageManager: ImageManagerPatched } = await import("./tools_patched.js");
  const imageManagerPatched = new ImageManagerPatched();
  const dataAfterPatch = await imageManagerPatched.#get(key, rawData);
  expect(dataAfterPatch).not.toBeNull();
  expect(dataAfterPatch.bitmap).not.toBeNull();
});
