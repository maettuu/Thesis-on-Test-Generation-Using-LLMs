it("should add a SVG image with a blob url", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const { Util } = await import("../../src/shared/util.js");
  const imageManager = new ImageManager();
  const svgImage = `<svg xmlns="http://www.w3.org/2000/svg" width="100" height="100"><circle cx="50" cy="50" r="40" stroke="green" stroke-width="4" fill="yellow" /></svg>`;
  const blob = new Blob([svgImage], { type: 'image/svg+xml' });
  const url = URL.createObjectURL(blob);
  const image = await fetch(url).then(response => response.blob());
  const data = {};
  await imageManager.addImage(image, data);
  const expected = true;
  const actual = data.isSvg;
  expect(actual).toBe(expected);
});
