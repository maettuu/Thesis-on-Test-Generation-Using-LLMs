it("should add a SVG image with a blob url", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const imageManager = new ImageManager();
  const svgBlob = new Blob(['<svg xmlns="http://www.w3.org/2000/svg" width="100" height="100"><rect x="10" y="10" width="50" height="50" fill="#ff0000"/></svg>'], { type: 'image/svg+xml' });
  const fileReader = new FileReader();
  const svgUrlPromise = new Promise(resolve => {
    fileReader.onload = () => {
      resolve(fileReader.result);
    };
  });
  fileReader.readAsDataURL(svgBlob);
  const svgUrl = await svgUrlPromise;
  const image = await imageManager.#get('svg-image', svgBlob);
  expect(image).not.toBeNull();
  expect(image.bitmap).not.toBeNull();
  expect(image.isSvg).toBe(true);
});
