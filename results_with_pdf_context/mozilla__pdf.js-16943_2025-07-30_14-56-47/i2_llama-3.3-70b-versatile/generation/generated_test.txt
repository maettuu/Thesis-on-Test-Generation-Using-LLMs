it("should change the style of the line and the resizers around a selected editor", async () => {
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { getComputedStyle } = await import("../../src/shared/util.js");

  const uiManager = new AnnotationEditorUIManager();
  const editor = new (class extends AnnotationEditor {
    constructor() {
      super({ parent: {}, id: "test", uiManager });
    }
  })();
  editor.div = document.createElement("div");
  document.body.appendChild(editor.div);

  const style = getComputedStyle(document.documentElement);
  const outlineWidth = parseFloat(style.getPropertyValue("--outline-width")) || 0;

  editor.initialize();
  editor.drag(0, 0);

  const resizersDiv = editor.div.querySelector(".resizers");
  const borderWidth = parseFloat(getComputedStyle(resizersDiv).borderWidth);

  expect(borderWidth).toBe(outlineWidth);
});
