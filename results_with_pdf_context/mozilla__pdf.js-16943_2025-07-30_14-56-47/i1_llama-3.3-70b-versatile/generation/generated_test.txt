it("should change the style of the line and the resizers around a selected editor", async () => {
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const { ColorManager } = await import("../../src/display/editor/tools.js");
  const editor = new AnnotationEditor({ parent: { viewport: { rotation: 0 } }, id: 1, uiManager: { viewParameters: { rotation: 0 } } });
  const initialStyle = getComputedStyle(editor.div);
  const initialBorderWidth = parseFloat(initialStyle.getPropertyValue("border-width"));
  await AnnotationEditor.initialize({ get: () => "" });
  const updatedStyle = getComputedStyle(editor.div);
  const updatedBorderWidth = parseFloat(updatedStyle.getPropertyValue("border-width"));
  expect(updatedBorderWidth).not.toEqual(initialBorderWidth);
});
