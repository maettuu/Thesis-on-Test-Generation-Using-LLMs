it("should ensure free highlighting in PDF does not lag behind the mouse cursor", async () => {
  const { FreeOutliner } = await import("../../src/display/editor/outliner.js");
  
  const initialPoint = { x: 10, y: 10 };
  const box = [0, 0, 100, 100];
  const scaleFactor = 1;
  const thickness = 2;
  const isLTR = true;
  const innerMargin = 0;

  const freeOutliner = new FreeOutliner(initialPoint, box, scaleFactor, thickness, isLTR, innerMargin);
  
  const pointsToAdd = [
    { x: 20, y: 20 },
    { x: 30, y: 30 },
    { x: 40, y: 40 },
    { x: 50, y: 50 }
  ];

  pointsToAdd.forEach(point => freeOutliner.add(point));

  const expectedSVGPath = "M0.1 0.1 L0.2 0.2 L0.3 0.3 L0.4 0.4 L0.5 0.5 Z";
  const actualSVGPath = freeOutliner.toSVGPath();

  expect(actualSVGPath).toBe(expectedSVGPath);
});
