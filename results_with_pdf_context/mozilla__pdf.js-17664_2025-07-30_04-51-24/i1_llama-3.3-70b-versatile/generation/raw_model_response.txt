it("should not lag behind the mouse cursor when using free highlighting in PDF", async () => {
  const { FreeOutliner } = await import("../../src/display/editor/outliner.js");
  const { Util } = await import("../../src/shared/util.js");
  const outliner = new FreeOutliner({ x: 10, y: 10 }, [0, 0, 100, 100], 1, 1, true);
  outliner.add({ x: 20, y: 20 });
  outliner.add({ x: 30, y: 30 });
  const path = outliner.toSVGPath();
  const expectedPath = `M10 10 L20 20 L30 30 L40 40 Z`;
  expect(path).toBe(expectedPath);
});
