it("should end a free highlight where the mouse pointer is", async () => {
  const { FreeOutliner } = await import("../../src/display/editor/outliner.js");
  const box = { x: 0, y: 0, width: 100, height: 100 };
  const scaleFactor = 1;
  const thickness = 1;
  const isLTR = true;
  const innerMargin = 0;
  const outliner = new FreeOutliner({ x: 10, y: 10 }, box, scaleFactor, thickness, isLTR, innerMargin);
  outliner.add({ x: 20, y: 20 });
  outliner.add({ x: 30, y: 30 });
  const expected = `M${(10 - 0) / 100} ${(10 - 0) / 100} L${(20 - 0) / 100} ${(20 - 0) / 100} L${(30 - 0) / 100} ${(30 - 0) / 100} L${(30 - 0) / 100} ${(30 - 0) / 100} L${(30 - 0) / 100} ${(30 - 0) / 100} Z`;
  const actual = outliner.toSVGPath();
  expect(actual).toBe(expected);
});
