it("should not lag behind the mouse cursor when highlighting in PDF", async () => {
  const { FreeOutliner } = await import("../../src/display/editor/outliner.js");
  const box = { x: 0, y: 0, width: 100, height: 100 };
  const scaleFactor = 1;
  const thickness = 1;
  const isLTR = true;
  const innerMargin = 0;
  const outliner = new FreeOutliner({ x: 10, y: 10 }, box, scaleFactor, thickness, isLTR, innerMargin);
  outliner.add({ x: 20, y: 20 });
  outliner.add({ x: 30, y: 30 });
  const outlines = outliner.getOutlines();
  const expectedOutline = new Float64Array(36);
  expectedOutline.set([NaN, NaN, NaN, NaN, 10, 10], 0);
  expectedOutline.set([NaN, NaN, NaN, NaN, 20, 20], 12);
  expectedOutline.set([NaN, NaN, NaN, NaN, 30, 30], 24);
  expectedOutline.set([NaN, NaN, NaN, NaN, 20, 20], 30);
  expectedOutline.set([NaN, NaN, NaN, NaN, 30, 30], 32);
  const actualOutline = outlines.outline;
  for (let i = 0; i < actualOutline.length; i++) {
    if (Math.abs(actualOutline[i] - expectedOutline[i]) > 0.01) {
      throw new Error("Outlines do not match");
    }
  }
});
