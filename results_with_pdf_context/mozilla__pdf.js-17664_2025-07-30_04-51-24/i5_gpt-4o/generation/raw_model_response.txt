```javascript
it("should update highlighting position without lag", async () => {
  const { FreeOutliner } = await import("../../src/display/editor/outliner.js");
  
  const outliner = new FreeOutliner();
  outliner.#box = [0, 0, 100, 100];
  
  outliner.add({ x: 10, y: 10 });
  outliner.add({ x: 20, y: 20 });
  
  const expected = [
    (10 - 0) / 100, (10 - 0) / 100,
    (20 - 0) / 100, (20 - 0) / 100
  ];
  
  const actual = outliner.#getLastCoords();
  
  expect(actual).toEqual(expected);
});
```
