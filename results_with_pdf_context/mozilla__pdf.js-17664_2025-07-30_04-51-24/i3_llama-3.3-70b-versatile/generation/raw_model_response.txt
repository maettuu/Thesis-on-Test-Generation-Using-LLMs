it("should fix the free highlighting in PDF not lagging behind the mouse cursor", async () => {
  const { FreeOutliner } = await import("../../src/display/editor/outliner.js");
  const outliner = new FreeOutliner({ x: 10, y: 10 }, { x: 0, y: 0, width: 100, height: 100 }, 1, 1, true);
  outliner.add({ x: 20, y: 20 });
  outliner.add({ x: 30, y: 30 });
  const outlines = outliner.getOutlines();
  const expectedOutline = new Float64Array(36);
  expectedOutline.set([NaN, NaN, NaN, NaN, 10, 10], 0);
  expectedOutline.set([NaN, NaN, NaN, NaN, 20, 20], 8);
  expectedOutline.set([NaN, NaN, NaN, NaN, 30, 30], 16);
  expectedOutline.set([NaN, NaN, NaN, NaN, 20, 20], 24);
  expectedOutline.set([NaN, NaN, NaN, NaN, 30, 30], 28);
  expectedOutline.set([NaN, NaN, NaN, NaN, 20, 20], 32);
  const actualOutline = outlines.outline;
  for (let i = 0; i < expectedOutline.length; i++) {
    if (expectedOutline[i] !== actualOutline[i]) {
      throw new Error("Outlines do not match");
    }
  }
});
