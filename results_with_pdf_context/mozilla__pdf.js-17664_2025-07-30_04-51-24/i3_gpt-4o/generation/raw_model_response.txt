it("should update the outline coordinates correctly after adding points", async () => {
  const { FreeOutliner } = await import("../../src/display/editor/outliner.js");
  const initialPoint = { x: 10, y: 10 };
  const box = [0, 0, 100, 100];
  const scaleFactor = 1;
  const thickness = 2;
  const isLTR = true;
  const outliner = new FreeOutliner(initialPoint, box, scaleFactor, thickness, isLTR);

  outliner.add({ x: 20, y: 20 });
  outliner.add({ x: 30, y: 30 });

  const expectedOutline = outliner.getOutlines().serialize([0, 0, 100, 100], 0);
  const actualOutline = outliner.getOutlines().serialize([0, 0, 100, 100], 0);

  expect(actualOutline).toEqual(expectedOutline);
});
