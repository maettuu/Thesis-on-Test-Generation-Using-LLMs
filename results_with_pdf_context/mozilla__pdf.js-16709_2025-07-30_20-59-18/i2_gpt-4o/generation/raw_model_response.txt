it("should redo an added image/text/drawing to the PDF file using Keyboard shortcuts", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const eventBus = { _on: jest.fn() };
  const pdfDocument = { annotationStorage: {}, filterFactory: {} };
  const manager = new AnnotationEditorUIManager(null, eventBus, pdfDocument, null);

  const redoSpy = jest.spyOn(manager, 'redo');
  const keyboardEvent = new KeyboardEvent("keydown", { key: "Z", metaKey: true, shiftKey: true });
  
  manager.#boundKeydown(keyboardEvent);
  
  expect(redoSpy).toHaveBeenCalled();
});
