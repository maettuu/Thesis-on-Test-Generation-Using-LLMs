it("should redo an added image/text/drawing to the PDF file using Keyboard shortcuts", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { PDFDocument } = await import("../../src/core/document.js");
  const { PDFFetchStream } = await import("../../src/display/fetch_stream.js");

  const pdf = await PDFDocument.load(await PDFFetchStream.fetch("test.pdf"));
  const annotationEditorUIManager = new AnnotationEditorUIManager(pdf);

  annotationEditorUIManager.addEventListener("keydown", (event) => {
    if (event.ctrlKey && event.key === "y" || event.metaKey && event.shiftKey && (event.key === "Z" || event.key === "z")) {
      annotationEditorUIManager.redo();
    }
  });

  const initialAnnotations = annotationEditorUIManager.getAnnotations();
  annotationEditorUIManager.addAnnotation({ type: "text", page: 1, x: 10, y: 10 });
  annotationEditorUIManager.undo();
  annotationEditorUIManager.redo();

  const annotationsAfterRedo = annotationEditorUIManager.getAnnotations();

  expect(annotationsAfterRedo.length).toBe(initialAnnotations.length + 1);
});
