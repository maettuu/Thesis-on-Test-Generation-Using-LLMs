it("should redo an added imagetextdrawing to the PDF file using keyboard shortcuts", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const manager = new AnnotationEditorUIManager();

  // Simulate adding an annotation
  manager.addAnnotation({ type: "text", content: "Sample Text" });

  // Perform undo operation
  manager.undo();

  // Check that the annotation is removed
  let annotations = manager.getAnnotations();
  expect(annotations.length).toBe(0);

  // Perform redo operation using keyboard shortcut
  const event = new KeyboardEvent("keydown", { key: "Z", ctrlKey: true, shiftKey: true });
  document.dispatchEvent(event);

  // Check that the annotation is added back
  annotations = manager.getAnnotations();
  expect(annotations.length).toBe(1);
  expect(annotations[0].content).toBe("Sample Text");
});
