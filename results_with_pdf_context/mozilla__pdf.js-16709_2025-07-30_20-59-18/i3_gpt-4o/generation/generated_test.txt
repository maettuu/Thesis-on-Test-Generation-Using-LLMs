it("should redo an added imagetextdrawing to the PDF file using Keyboard shortcuts", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { EventBus } = await import("../../web/event_utils.js");

  const eventBus = new EventBus();
  const container = document.createElement("div");
  const pdfDocument = { annotationStorage: new Map(), filterFactory: {} };
  const manager = new AnnotationEditorUIManager(container, eventBus, pdfDocument, null);

  manager.updateMode(AnnotationEditorType.FREETEXT);
  manager.addCommands({
    cmd: () => manager.addEditor({ id: "editor1", pageIndex: 0 }),
    undo: () => manager.removeEditor({ id: "editor1" }),
    mustExec: true,
  });

  manager.undo();
  expect(manager.getEditor("editor1")).toBeUndefined();

  manager.redo();
  expect(manager.getEditor("editor1")).toBeDefined();
});
