  const { AnnotationFactory } = await import("../../src/core/annotation.js");
  const { warn } = await import("../../src/shared/util.js");
  
  let warningLogged = false;
  const originalWarn = warn;
  warn = (msg) => {
    if (msg.includes("Unknown field name, falling back to empty field name.")) {
      warningLogged = true;
    }
  };

  const xref = {}; // Mocked xref
  const ref = {}; // Mocked ref
  const annotationGlobals = { acroForm: { get: () => false }, pdfManager: { ensure: () => Promise.resolve() } };
  const idFactory = { createObjId: () => "1" };
  const pageRef = {}; // Mocked pageRef

  await AnnotationFactory.create(xref, ref, annotationGlobals, idFactory, true, pageRef);

  warn = originalWarn;
  expect(warningLogged).toBe(false);
});
