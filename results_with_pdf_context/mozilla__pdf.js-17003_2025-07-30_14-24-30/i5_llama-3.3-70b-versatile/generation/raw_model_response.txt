```javascript
it("should not throw a warning when printing a pdf with an added annotation", async () => {
  const { PDFDocument } = await import("../../src/core/document.js");
  const { AnnotationFactory } = await import("../../src/core/annotation.js");
  const { createIdFactory } = await import("./test_utils.js");
  const pdfDoc = await PDFDocument.load(await (await import("../../src/display/api.js")).getDocument({ url: "./test.pdf" }));
  const idFactory = createIdFactory();
  const annotationFactory = new AnnotationFactory(pdfDoc, idFactory);
  const annotation = annotationFactory.createAnnotation({ subtype: "Highlight" });
  const page = pdfDoc.getPage(1);
  page.addAnnotation(annotation);
  const consoleWarnSpy = jest.spyOn(console, "warn");
  await pdfDoc.print();
  expect(consoleWarnSpy).not.toHaveBeenCalledWith("Warning: Unknown field name, falling back to empty field name.");
});
```
