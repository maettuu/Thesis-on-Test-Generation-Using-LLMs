it("should not warn when printing a pdf with an added annotation", async () => {
  const { AnnotationFactory, Annotation, Dict, Ref, Name } = await import("../../src/core/annotation.js");
  const { createIdFactory, XRefMock } = await import("./test_utils.js");
  const idFactory = createIdFactory();
  const xref = new XRefMock();
  const annotationGlobals = { pdfManager: { evaluatorOptions: {} } };
  const ref = new Ref(1, 0);
  const dict = new Dict({ Subtype: new Name("Widget") });
  xref.fetchIfRef = () => dict;
  const consoleWarnSpy = jest.spyOn(console, 'warn');
  await AnnotationFactory.create(xref, ref, annotationGlobals, idFactory, true);
  expect(consoleWarnSpy).not.toHaveBeenCalledWith("Unknown field name, falling back to empty field name.");
});
