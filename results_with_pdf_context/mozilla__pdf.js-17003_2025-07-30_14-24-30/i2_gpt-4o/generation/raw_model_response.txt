```javascript
it("should not log warning when printing a PDF with an added annotation", async () => {
  const { AnnotationFactory } = await import("../../src/core/annotation.js");
  const { Ref, Dict } = await import("../../src/core/primitives.js");
  const { warn } = await import("../../src/shared/util.js");

  const xref = {};
  const ref = new Ref(1, 0);
  const annotationGlobals = {
    acroForm: new Dict(),
    pdfManager: {
      ensure: (context, methodName, args) => context[methodName](...args),
      evaluatorOptions: {},
    },
  };
  const idFactory = { createObjId: () => "1" };
  const pageRef = new Ref(2, 0);

  let warningLogged = false;
  const originalWarn = warn;
  warn = (msg) => {
    if (msg.includes("Unknown field name")) {
      warningLogged = true;
    }
  };

  await AnnotationFactory.create(xref, ref, annotationGlobals, idFactory, true, pageRef);

  warn = originalWarn;
  expect(warningLogged).toBe(false);
});
```
