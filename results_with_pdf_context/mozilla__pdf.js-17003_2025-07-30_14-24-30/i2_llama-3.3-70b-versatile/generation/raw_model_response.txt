it("should not throw a warning when printing a pdf with an added annotation", async () => {
  const { AnnotationFactory, warn } = await import("../../src/core/annotation.js");
  const { PDFDocument } = await import("../../src/core/document.js");
  const originalWarn = console.warn;
  let warningMessage;
  console.warn = (message) => {
    warningMessage = message;
  };

  const pdfDoc = await PDFDocument.load(await (await import("../../src/shared/util.js")).stringToBytes("test.pdf"));
  const annotationGlobals = {
    pdfManager: pdfDoc.manager,
    acroForm: pdfDoc.catalog.get("AcroForm"),
  };
  const idFactory = await (await import("../../src/shared/util.js")).createIdFactory();
  const ref = await (await import("../../src/core/primitives.js")).Ref.fromNum(1, 0);
  await AnnotationFactory.create(pdfDoc.xref, ref, annotationGlobals, idFactory, true);

  console.warn = originalWarn;
  expect(warningMessage).not.toContain("Unknown field name, falling back to empty field name.");
});
