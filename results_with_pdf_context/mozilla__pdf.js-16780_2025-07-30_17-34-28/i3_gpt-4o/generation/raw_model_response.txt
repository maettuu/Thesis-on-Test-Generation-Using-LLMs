it("should resize SVG images correctly", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const svgData = `data:image/svg+xml;charset=UTF-8,<svg viewBox="0 0 100 100" xmlns="http://www.w3.org/2000/svg"><rect width="100" height="100" style="fill:red;"/></svg>`;
  const imageManager = new ImageManager();
  const imageData = await imageManager.getFromUrl(svgData);

  const canvas = new OffscreenCanvas(100, 200);
  const ctx = canvas.getContext("2d");
  ctx.drawImage(imageData.bitmap, 0, 0, 100, 100, 0, 0, 100, 200);
  const imageDataArray = ctx.getImageData(0, 0, 100, 200).data;

  const isResized = imageDataArray[0] === 255 && imageDataArray[1] === 0 && imageDataArray[2] === 0;
  expect(isResized).toBe(true);
});
