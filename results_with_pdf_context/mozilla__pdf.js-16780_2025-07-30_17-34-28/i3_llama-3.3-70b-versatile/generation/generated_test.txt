it("should allow resizing of SVG images horizontally and vertically in pdfjs", async () => {
  const { AnnotationEditorUIManager, ImageManager } = await import("../../src/display/editor/tools.js");
  const { AnnotationEditorType } = await import("../../shared/util.js");
  const { createIdFactory } = await import("./test_utils.js");

  const idFactory = createIdFactory();
  const imageManager = new ImageManager();
  const annotationEditorUIManager = new AnnotationEditorUIManager({}, {}, {}, {});

  const svgUrl = "data:image/svg+xml;charset=UTF-8,<svg viewBox='0 0 100 100' width='100' height='100' xmlns='http://www.w3.org/2000/svg'><rect width='100' height='100' style='fill:red;'/></svg>";

  const image = await imageManager.getFromUrl(svgUrl);
  const editor = {
    id: idFactory(),
    pageIndex: 0,
    annotationElementId: idFactory(),
    type: AnnotationEditorType.IMAGE,
    isEmpty: () => false,
    commitOrRemove: () => {},
    remove: () => {},
    translateInPage: () => {},
    select: () => {},
    unselect: () => {},
    updateParams: () => {},
    propertiesToUpdate: {},
  };

  annotationEditorUIManager.addEditor(editor);
  annotationEditorUIManager.updateMode(AnnotationEditorType.IMAGE);

  const resizedImage = await imageManager.getFromId(image.id);
  const resizedSvgUrl = resizedImage.svgUrl;

  expect(resizedSvgUrl).toContain("svgView(preserveAspectRatio(none))");
});
