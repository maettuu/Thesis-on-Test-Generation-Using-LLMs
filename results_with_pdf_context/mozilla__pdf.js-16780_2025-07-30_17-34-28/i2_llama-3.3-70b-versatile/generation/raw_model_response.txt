it("should resize SVG images horizontally/vertically without keeping the aspect ratio", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const { Util } = await import("../../shared/util.js");
  const imageManager = new ImageManager();
  const svgImage = new Blob([`
    <svg xmlns="http://www.w3.org/2000/svg" width="100" height="100">
      <rect width="100" height="100" style="fill:red;"/>
    </svg>
  `], { type: "image/svg+xml" });
  const image = await imageManager.#get("svgImage", svgImage);
  const canvas = new OffscreenCanvas(50, 200);
  const ctx = canvas.getContext("2d");
  ctx.drawImage(image.bitmap, 0, 0, 50, 200);
  const pixels = new Uint32Array(ctx.getImageData(0, 0, 1, 1).data.buffer);
  const expectedColor = 0xFF0000FF;
  expect(pixels[0]).toBe(expectedColor);
});
