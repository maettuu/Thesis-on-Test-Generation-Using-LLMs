it("should resize SVG images horizontallyvertically in pdfjs", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const { Util } = await import("../../shared/util.js");
  const imageManager = new ImageManager();
  const svgImage = `data:image/svg+xml;charset=UTF-8,<svg viewBox="0 0 100 100" width="100" height="100" xmlns="http://www.w3.org/2000/svg"><rect width="100" height="100" style="fill:red;"/></svg>`;
  const image = await imageManager.#get("svgImage", svgImage);
  const imageElement = image.bitmap;
  const canvas = new OffscreenCanvas(50, 50);
  const ctx = canvas.getContext("2d");
  ctx.drawImage(imageElement, 0, 0, 50, 50);
  const imageData = ctx.getImageData(0, 0, 50, 50).data;
  const expectedPixel = [255, 0, 0, 255];
  const actualPixel = [
    imageData[0],
    imageData[1],
    imageData[2],
    imageData[3],
  ];
  Util.assert(
    actualPixel.every((val, idx) => val === expectedPixel[idx]),
    "SVG image is not resized correctly"
  );
});
