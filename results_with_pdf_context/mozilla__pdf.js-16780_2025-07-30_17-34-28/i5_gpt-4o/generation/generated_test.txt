it("should resize SVG images without preserving aspect ratio", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const { DOMCanvasFactory } = await import("../../src/display/display_utils.js");

  const svgData = `data:image/svg+xml;charset=UTF-8,<svg viewBox="0 0 100 100" xmlns="http://www.w3.org/2000/svg"><rect width="100" height="100" style="fill:red;"/></svg>`;
  const canvasFactory = new DOMCanvasFactory();
  const canvasAndCtx = canvasFactory.create(50, 100);
  const ctx = canvasAndCtx.context;
  const image = new Image();
  image.src = svgData;

  await new Promise((resolve) => (image.onload = resolve));
  ctx.drawImage(image, 0, 0, 50, 100);

  const imageData = ctx.getImageData(0, 0, 50, 100).data;
  const isResizedCorrectly = Array.from(imageData).some((value, index) => index % 4 === 0 && value === 255);

  expect(isResizedCorrectly).toBe(true);
});
