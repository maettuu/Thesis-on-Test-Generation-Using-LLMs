it("should allow resizing of SVG images horizontally and vertically in pdfjs", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const imageManager = new ImageManager();
  const svg = `data:image/svg+xml;charset=UTF-8,<svg viewBox="0 0 1 1" width="1" height="1" xmlns="http://www.w3.org/2000/svg"><rect width="1" height="1" style="fill:red;"/></svg>`;
  const image = await imageManager.getFromUrl(svg);
  const svgUrl = image.svgUrl;
  const mustRemoveAspectRatioPromise = await ImageManager._checkIfSVGFitsInCanvas();
  const newSvgUrl = mustRemoveAspectRatioPromise ? `${svgUrl}#svgView(preserveAspectRatio(none))` : svgUrl;
  expect(newSvgUrl).not.toBe(svgUrl);
});
