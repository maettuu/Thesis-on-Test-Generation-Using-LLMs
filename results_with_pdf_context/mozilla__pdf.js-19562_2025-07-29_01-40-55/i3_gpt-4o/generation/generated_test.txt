it("should render Russian characters correctly", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");
  const loadingTask = getDocument(buildGetDocumentParams("issue19550.pdf"));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const textContent = await pdfPage.getTextContent();
  const textItems = textContent.items.map(item => item.str).join("");
  const expectedText = "Привет мир"; // Example expected Russian text
  expect(textItems).toContain(expectedText);
});
