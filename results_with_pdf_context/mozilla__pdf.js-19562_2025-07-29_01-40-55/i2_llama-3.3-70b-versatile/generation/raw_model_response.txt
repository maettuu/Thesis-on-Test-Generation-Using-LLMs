it("should render Cyrillic characters correctly", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue19550.pdf'));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const textContent = await page.getTextContent();
  const expectedText = "Привет"; // example Cyrillic text
  const actualText = textContent.items[0].str;
  expect(actualText).toBe(expectedText);
});
