it("should render text correctly when font-file entry doesn't contain a Stream", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue18941.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const textContent = await pdfPage.getTextContent();
  const expectedText = "FAMI INVERSION SAS";
  const actualText = textContent.items[0].str;
  expect(actualText).toContain(expectedText);
});
