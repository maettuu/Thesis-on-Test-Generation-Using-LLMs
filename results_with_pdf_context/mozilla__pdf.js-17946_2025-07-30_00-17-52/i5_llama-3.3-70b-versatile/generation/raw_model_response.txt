it("should render large images in PDF", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue17289.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const pageViewport = pdfPage.getViewport({ scale: 1 });
  const renderTask = pdfPage.render({
    viewport: pageViewport,
    intent: 'display',
  });
  const renderedPage = await renderTask.promise;
  const imageRendered = renderedPage.items.some((item) => item.type === 'image');
  expect(imageRendered).toBe(true);
});
