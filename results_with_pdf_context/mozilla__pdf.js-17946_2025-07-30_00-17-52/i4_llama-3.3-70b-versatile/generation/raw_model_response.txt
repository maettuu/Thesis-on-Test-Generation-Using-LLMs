it("should render large images in PDF that other renderers do", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue17289.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const image = await pdfPage.getOperatorList().find((operator) => operator.name === 'Do');
  const imageData = await image.getImage();
  const expectedWidth = 1654;
  const expectedHeight = 2339;
  assert.equal(imageData.width, expectedWidth, `Expected width to be ${expectedWidth}`);
  assert.equal(imageData.height, expectedHeight, `Expected height to be ${expectedHeight}`);
});
