it("should not throw an error when parsing a CFF file", async () => {
  const { CFFParser } = await import("../../src/core/cff_parser.js");
  const parser = new CFFParser();
  const cffFile = new Uint8Array([/* sample CFF file data */]);
  const stream = new (await import("../../src/core/stream.js")).Stream(cffFile);
  await expect(parser.parse(stream)).resolves.not.toThrow();
});
