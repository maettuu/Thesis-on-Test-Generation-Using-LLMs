```javascript
it("should not have obsolete files in the project", async () => {
  const fs = await import("fs");
  const path = await import("path");

  const obsoleteFiles = [
    path.resolve(__dirname, "../../src/shared/cffStandardStrings.js"),
    path.resolve(__dirname, "../../src/shared/fonts_utils.js"),
  ];

  const filesExist = obsoleteFiles.map(filePath => fs.existsSync(filePath));
  const expected = [false, false];
  
  expect(filesExist).toEqual(expected);
});
```
