it("should not have CFF related globals after patch", async () => {
  const { CFFParser } = await import("../../src/core/cff_parser.js");
  const globalsBefore = Object.keys(globalThis);
  new CFFParser(); // Trigger any potential global pollution
  const globalsAfter = Object.keys(globalThis);
  const newGlobals = globalsAfter.filter(g => !globalsBefore.includes(g));
  expect(newGlobals).not.toContain("CFFDictDataMap");
  expect(newGlobals).not.toContain("CFFDictPrivateDataMap");
  expect(newGlobals).not.toContain("CFFEncodingMap");
});
