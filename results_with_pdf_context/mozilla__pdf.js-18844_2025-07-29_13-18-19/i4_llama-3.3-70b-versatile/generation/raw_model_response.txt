```javascript
it("should render text layer with correct font family on Windows and Firefox", async () => {
  const { TextLayer } = await import("../../src/display/text_layer.js");
  const { FeatureTest } = await import("../../src/shared/util.js");
  const { setLayerDimensions } = await import("../../src/display/display_utils.js");

  const textContentSource = {
    items: [
      {
        str: "Hello World",
        fontName: "sans-serif",
      },
    ],
    styles: {
      "sans-serif": {
        fontFamily: "sans-serif",
      },
    },
  };

  const container = document.createElement("div");
  const viewport = {
    scale: 1,
    rotation: 0,
    rawDims: {
      pageWidth: 100,
      pageHeight: 100,
      pageX: 0,
      pageY: 0,
    },
  };

  const textLayer = new TextLayer({
    textContentSource,
    container,
    viewport,
  });

  await textLayer.render();

  const expectedFontFamily = FeatureTest.platform.isWindows && FeatureTest.platform.isFirefox ? "Calibri, sans-serif" : "sans-serif";
  const actualFontFamily = textLayer.textDivs[0].style.fontFamily;

  expect(actualFontFamily).toBe(expectedFontFamily);
});
```
