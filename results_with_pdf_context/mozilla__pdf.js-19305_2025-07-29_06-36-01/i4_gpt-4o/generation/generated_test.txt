it("should correctly extract bbox and background for MeshShadingPattern", async () => {
  const { MeshShadingPattern } = await import("../../src/display/pattern_helper.js");
  const IR = [null, null, "coords", "colors", "figures", "bounds", "expected_bbox", "expected_background"];
  const pattern = new MeshShadingPattern(IR);
  expect(pattern._bbox).toEqual("expected_bbox");
  expect(pattern._background).toEqual("expected_background");
});
