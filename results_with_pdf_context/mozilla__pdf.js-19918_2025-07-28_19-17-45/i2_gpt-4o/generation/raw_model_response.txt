it("should decode CMYK JPEG images using ImageDecoder", async () => {
  const { JpegStream } = await import("../../src/core/jpeg_stream.js");
  const { JpegImage } = await import("../../src/core/jpg.js");

  const cmykJpegData = new Uint8Array([
    0xff, 0xd8, // SOI
    0xff, 0xc0, // SOF0
    0x00, 0x11, // Length
    0x08, 0x00, 0x10, 0x00, 0x10, // Image size
    0x04, // Number of components (CMYK)
    0x01, 0x11, 0x00, 0x02, 0x11, 0x01, 0x03, 0x11, 0x01, 0x04, 0x11, 0x01,
    0xff, 0xd9, // EOI
  ]);

  const stream = {
    dict: null,
    asyncGetBytes: async () => cmykJpegData,
  };

  const jpegStream = new JpegStream(stream, cmykJpegData.length, {});
  const expected = true;
  const actual = await JpegStream.canUseImageDecoderCMYK;
  expect(actual).toBe(expected);
});
