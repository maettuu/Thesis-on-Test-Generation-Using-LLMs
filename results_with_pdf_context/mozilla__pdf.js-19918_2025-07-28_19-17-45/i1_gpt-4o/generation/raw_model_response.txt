it("should support decoding CMYK JPEG images using ImageDecoder", async () => {
  const { JpegStream } = await import("../../src/core/jpeg_stream.js");
  const { JpegImage } = await import("../../src/core/jpg.js");

  const mockStream = {
    dict: {},
    asyncGetBytes: async () => new Uint8Array([0xff, 0xd8, 0xff, 0xc0, 0x00, 0x11, 0x08, 0x00, 0x01, 0x00, 0x01, 0x04, 0x01, 0x11, 0x00, 0xff, 0xd9]),
  };

  const jpegStream = new JpegStream(mockStream, null, {});
  const jpegOptions = { colorTransform: -1 };
  jpegStream.jpegOptions = jpegOptions;

  const transferableImage = await jpegStream.getTransferableImage();
  expect(transferableImage).not.toBeNull();
});
