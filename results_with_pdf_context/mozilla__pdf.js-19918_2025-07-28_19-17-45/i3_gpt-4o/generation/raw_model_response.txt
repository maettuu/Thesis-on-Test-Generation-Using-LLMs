it("should support decoding CMYK JPEG images using ImageDecoder", async () => {
  const { JpegStream } = await import("../../src/core/jpeg_stream.js");
  const { JpegImage } = await import("../../src/core/jpg.js");

  const cmykJpegData = new Uint8Array([
    0xff, 0xd8, 0xff, 0xc0, 0x00, 0x11, 0x08, 0x00, 0x10, 0x00, 0x10, 0x04, 0x01, 0x11, 0x00, 0x02, 0x11, 0x01, 0x03, 0x11, 0x01, 0xff, 0xda, 0x00, 0x0c, 0x03, 0x01, 0x00, 0x02, 0x11, 0x03, 0x11, 0x00, 0x3f, 0x00, 0xff, 0xd9
  ]);

  JpegStream.setOptions({ isImageDecoderSupported: true });

  const jpegImage = new JpegImage();
  const useImageDecoder = JpegImage.canUseImageDecoder(cmykJpegData, -1);

  const expected = { cmyk: true };
  expect(useImageDecoder).toEqual(expected);
});
