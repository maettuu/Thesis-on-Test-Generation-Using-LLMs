it("should respect the PDF.js verbosity levels for OpenJPEG warnings", async () => {
  const { setVerbosityLevel, VerbosityLevel, JpxImage } = await import("../../src/core/util.js");
  const { OpenJPEG } = await import("../../external/openjpeg/openjpeg.js");

  setVerbosityLevel(VerbosityLevel.DISABLE);
  const originalWarn = console.warn;
  const warnCalls = [];
  console.warn = (...args) => {
    warnCalls.push(...args);
  };

  const jpxImage = new JpxImage();
  jpxImage.decode(new Uint8Array([0x00, 0x00, 0x00, 0x00]));

  console.warn = originalWarn;
  expect(warnCalls.length).toBe(0);

  setVerbosityLevel(VerbosityLevel.WARN);
  warnCalls.length = 0;
  console.warn = (...args) => {
    warnCalls.push(...args);
  };

  const jpxImage2 = new JpxImage();
  jpxImage2.decode(new Uint8Array([0x00, 0x00, 0x00, 0x00]));

  console.warn = originalWarn;
  expect(warnCalls.length).toBeGreaterThan(0);
});
