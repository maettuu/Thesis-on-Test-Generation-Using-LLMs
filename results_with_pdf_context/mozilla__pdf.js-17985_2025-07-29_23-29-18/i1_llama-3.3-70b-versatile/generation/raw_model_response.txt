it("should respect the PDF.js verbosity levels for OpenJPEG warnings", async () => {
  const { setVerbosityLevel, getVerbosityLevel, VerbosityLevel, warn } = await import("../../src/shared/util.js");
  const { JpxImage } = await import("../../src/core/jpx.js");
  setVerbosityLevel(VerbosityLevel.DISABLE);
  const originalWarn = console.warn;
  let warned = false;
  console.warn = () => {
    warned = true;
  };
  try {
    JpxImage.decode("invalid_data");
  } catch (error) {
    // ignore error
  }
  expect(warned).toBe(false);
  setVerbosityLevel(VerbosityLevel.WARN);
  warned = false;
  try {
    JpxImage.decode("invalid_data");
  } catch (error) {
    // ignore error
  }
  expect(warned).toBe(false);
  setVerbosityLevel(VerbosityLevel.ERROR);
  warned = false;
  try {
    JpxImage.decode("invalid_data");
  } catch (error) {
    // ignore error
  }
  expect(warned).toBe(false);
  console.warn = originalWarn;
});
