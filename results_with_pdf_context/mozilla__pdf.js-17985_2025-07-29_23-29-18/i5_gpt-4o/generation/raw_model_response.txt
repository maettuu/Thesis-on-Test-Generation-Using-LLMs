it("should respect PDF.js verbosity levels for OpenJPEG warnings", async () => {
  const { JpxImage } = await import("../../src/core/jpx.js");
  const { setVerbosityLevel, VerbosityLevel } = await import("../../src/shared/util.js");

  setVerbosityLevel(VerbosityLevel.ERRORS);

  const data = new Uint8Array([0, 1, 2, 3]); // Example data
  const originalConsoleWarn = console.warn;
  let warningLogged = false;
  console.warn = () => {
    warningLogged = true;
  };

  try {
    JpxImage.decode(data);
  } catch (e) {
    // Ignore errors for this test
  }

  console.warn = originalConsoleWarn;

  expect(warningLogged).toBe(false);
});
