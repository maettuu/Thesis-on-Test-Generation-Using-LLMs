it("should respect the PDF.js verbosity levels for OpenJPEG warnings", async () => {
  const { setVerbosityLevel, getVerbosityLevel, VerbosityLevel } = await import("../../src/shared/util.js");
  const { JpxImage } = await import("../../src/core/jpx.js");
  const originalVerbosityLevel = getVerbosityLevel();
  setVerbosityLevel(VerbosityLevel.WARN);
  const warnSpy = jest.spyOn(console, 'warn');
  const imageData = JpxImage.decode(new Uint8Array([0xff, 0x4f, 0xff, 0x51]), true);
  expect(warnSpy).not.toHaveBeenCalled();
  setVerbosityLevel(VerbosityLevel.ERROR);
  warnSpy.mockReset();
  const imageData2 = JpxImage.decode(new Uint8Array([0xff, 0x4f, 0xff, 0x51]), true);
  expect(warnSpy).toHaveBeenCalledTimes(1);
  setVerbosityLevel(originalVerbosityLevel);
});
