17 import OpenJPEG from "../../external/openjpeg/openjpeg.js";

19 class JpxError extends BaseException {
20   constructor(msg) {
21     super(msg, "JpxError");
22   }
23 }
24 
25 class JpxImage {
26   static #module = null;
27 
28   static decode(data, ignoreColorSpace = false) {
29     this.#module ||= OpenJPEG();
30     const imageData = this.#module.decode(data, ignoreColorSpace);
31     if (typeof imageData === "string") {
32       throw new JpxError(imageData);
33     }
34     return imageData;
35   }
36 
37   static cleanup() {
38     this.#module = null;
39   }
40 
41   static parseImageProperties(stream) {
42     // No need to use OpenJPEG here since we're only getting very basic
43     // information which are located in the first bytes of the file.
44     let newByte = stream.getByte();
45     while (newByte >= 0) {
46       const oldByte = newByte;
47       newByte = stream.getByte();
48       const code = (oldByte << 8) | newByte;
49       // Image and tile size (SIZ)
50       if (code === 0xff51) {
51         stream.skip(4);
52         const Xsiz = stream.getInt32() >>> 0; // Byte 4
53         const Ysiz = stream.getInt32() >>> 0; // Byte 8
54         const XOsiz = stream.getInt32() >>> 0; // Byte 12
55         const YOsiz = stream.getInt32() >>> 0; // Byte 16
56         stream.skip(16);
57         const Csiz = stream.getUint16(); // Byte 36
58         return {
59           width: Xsiz - XOsiz,
60           height: Ysiz - YOsiz,
61           // Results are always returned as `Uint8ClampedArray`s.
62           bitsPerComponent: 8,
63           componentsCount: Csiz,
64         };
65       }
66     }
67     throw new JpxError("No size marker found in JPX stream");
68   }
69 }
70
