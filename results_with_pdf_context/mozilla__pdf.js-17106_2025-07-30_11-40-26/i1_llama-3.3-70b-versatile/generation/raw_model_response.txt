it("should release ImageBitmap data after cleanup", async () => {
  const { PDFWorker, PDFDocumentLoadingTask, PDFPageProxy } = await import("../../src/display/api.js");
  const { buildGetDocumentParams, TEST_PDFS_PATH } = await import("./test_utils.js");
  const params = buildGetDocumentParams({
    cMapUrl: CMAP_URL,
    cMapPacked: true,
    disableFontFace: true,
    disableWorker: false,
    useWorkerFetch: true,
  });
  const pdfWorker = new PDFWorker({
    workerSrc: WORKER_SRC,
  });
  const loadingTask = pdfWorker.createLoadingTask(params);
  const pdfDocument = await loadingTask.promise;
  const pageProxy = await pdfDocument.getPage(1);
  const page = new PDFPageProxy(pageProxy);
  const imageData = { bitmap: { close: jest.fn() } };
  page.objs.resolve(1, imageData);
  page._intentStates.clear();
  expect(imageData.bitmap.close).not.toHaveBeenCalled();
  page._cleanup();
  expect(imageData.bitmap.close).toHaveBeenCalledTimes(1);
});
