it("should release ImageBitmap data after cleanup", async () => {
  const { PDFPageProxy } = await import("../../src/display/api.js");
  const pageProxy = new PDFPageProxy(0, { view: [0, 0, 100, 100] }, {}, false);
  const imageData = { bitmap: { close: jest.fn() } };
  pageProxy.objs.resolve("testImage", imageData);
  pageProxy.cleanup();
  expect(imageData.bitmap.close).toHaveBeenCalled();
});
