it("should release ImageBitmap data after cleanup", async () => {
  const { PDFPageProxy, PDFDocumentProxy, PDFWorker, getDocument } = await import("../../src/display/api.js");
  const { createIdFactory } = await import("./test_utils.js");
  const idFactory = createIdFactory();
  const doc = await getDocument({ url: 'https://example.com/example.pdf' });
  const page = await doc.getPage(1);
  const imageData = { bitmap: { close: jest.fn() } };
  page.objs.resolve(idFactory.next(), imageData);
  await page.cleanup();
  expect(imageData.bitmap.close).toHaveBeenCalledTimes(1);
});
