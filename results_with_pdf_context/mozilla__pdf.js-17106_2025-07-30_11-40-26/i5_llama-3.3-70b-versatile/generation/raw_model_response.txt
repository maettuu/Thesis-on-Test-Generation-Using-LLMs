it("should release ImageBitmap data after cleanup", async () => {
  const { PDFWorker, PDFDocumentProxy, PDFPageProxy } = await import("../../src/display/api.js");
  const { PDFDocument } = await import("../../src/core/document.js");
  const { createIdFactory } = await import("./test_utils.js");

  const idFactory = createIdFactory();
  const pdfWorker = new PDFWorker();
  const pdfDocument = new PDFDocument({
    url: "test.pdf",
    worker: pdfWorker,
    idFactory,
  });
  const pageProxy = new PDFPageProxy({
    pdfDocument,
    pageIndex: 0,
    idFactory,
  });

  const imageData = { bitmap: { close: jest.fn() } };
  pageProxy._intentStates.set("test", "test");
  pageProxy.objs.set("test", imageData);
  pageProxy._cleanup();

  expect(imageData.bitmap.close).not.toHaveBeenCalled();

  pageProxy._intentStates.clear();
  pageProxy._cleanup();

  expect(imageData.bitmap.close).toHaveBeenCalledTimes(1);
});
