it("should correctly reflect radio button state in print view", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1860602.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pageNumber = 1;
  const page = await pdfDocument.getPage(pageNumber);
  const annotations = await page.getAnnotations();
  const radioButtons = annotations.filter(annotation => annotation.subtype === 'Widget' && annotation.fieldFlags === 4096);
  const expectedState = 'On';
  const actualState = radioButtons[0].buttonValue;
  expect(actualState).toBe(expectedState);
});
