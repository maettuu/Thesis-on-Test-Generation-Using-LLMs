```javascript
it("should reflect the correct state of radio buttons in print view on the PDF form", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");
  const loadingTask = getDocument(buildGetDocumentParams("bug1860602.pdf"));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const annotations = await page.getAnnotations();
  
  const radioButton = annotations.find(
    annotation => annotation.subtype === "Widget" && annotation.fieldType === "Btn" && annotation.radioButton
  );

  const expectedFieldValue = radioButton.buttonValue;
  const actualFieldValue = radioButton.fieldValue;

  expect(actualFieldValue).toEqual(expectedFieldValue);
});
```
