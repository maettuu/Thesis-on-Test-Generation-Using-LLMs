it("should correctly reflect radio button state in print view", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1860602.pdf'));
  const pdfDoc = await loadingTask.promise;
  const pageNumber = 1;
  const page = await pdfDoc.getPage(pageNumber);
  const annotations = await page.getAnnotations();
  const radioButtons = annotations.filter(annotation => annotation.subtype === 'Widget' && annotation.fieldName.includes('Radio'));
  const expectedStates = ['On', 'Off']; // Define the expected states of the radio buttons
  const actualStates = await Promise.all(radioButtons.map(async radioButton => {
    const widgetAnnotation = await radioButton.getWidgetAnnotation();
    return widgetAnnotation.data.buttonValue;
  }));
  expectedStates.forEach((state, index) => {
    if (actualStates[index] !== state) {
      throw new Error(`Radio button ${index} state does not match expected state`);
    }
  });
});
