it("should correctly reflect the state of radio buttons in print view", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1860602.pdf'));
  const pdfDocument = await loadingTask.promise;
  const annotationFactory = new AnnotationFactory();
  const page = await pdfDocument.getPage(1);
  const annotations = await page.getAnnotations();
  const annotation = annotations[0];
  const widgetAnnotation = await annotationFactory.create(
    pdfDocument.xref,
    annotation,
    {
      pdfManager: pdfDocument,
      acroForm: pdfDocument.catalog.get("acroForm"),
      xfaDatasets: null,
      structTreeRoot: null,
      baseUrl: null,
      attachments: null,
    },
    createIdFactory(),
    true
  );
  expect(widgetAnnotation.data.fieldValue).not.toBeNull();
});
