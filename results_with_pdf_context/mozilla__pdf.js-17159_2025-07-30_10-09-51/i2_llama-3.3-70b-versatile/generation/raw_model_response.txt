it("should correctly reflect radio button state in print view", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1860602.pdf'));
  const pdfDocument = await loadingTask.promise;
  const annotation = pdfDocument.getAnnotations(0).find(annotation => annotation instanceof ButtonWidgetAnnotation && annotation.data.radioButton);
  const fieldValue = annotation.data.fieldValue;
  const buttonValue = annotation.data.buttonValue;
  expect(fieldValue).toBe(buttonValue);
});
