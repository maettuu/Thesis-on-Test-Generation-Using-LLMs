it("should render images with inlined images without the closing 'EI'", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue17794.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const operatorList = await pdfPage.getOperatorList();
  const imageOps = operatorList.getOperatorList().filter(op => op.fn === 'paintImageMaskXObject' || op.fn === 'paintInlineImageXObject');
  const expectedImageCount = 2;
  assert.equal(imageOps.length, expectedImageCount, `Expected ${expectedImageCount} images, but got ${imageOps.length}`);
});
