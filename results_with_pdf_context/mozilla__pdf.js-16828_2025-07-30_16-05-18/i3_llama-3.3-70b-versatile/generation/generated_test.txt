it("should support pasting an image from the clipboard", async () => {
  const { AnnotationEditorUIManager, AnnotationEditorLayer, StampEditor } = await import("../../src/display/editor/tools.js");
  const { AnnotationEditor } = await import("../../src/display/editor/editor.js");
  const uiManager = new AnnotationEditorUIManager(null, null, null, null, null);
  const layer = new AnnotationEditorLayer({ uiManager, div: document.createElement("div"), pageIndex: 0, viewport: { width: 100, height: 100 } });
  uiManager.addLayer(layer);
  const event = new ClipboardEvent("paste", { clipboardData: new DataTransfer() });
  event.clipboardData.items.add(new File(["image"], "image.png", { type: "image/png" }));
  uiManager.paste(event);
  expect(uiManager.getMode()).toBe(AnnotationEditor.Type.STAMP);
});
