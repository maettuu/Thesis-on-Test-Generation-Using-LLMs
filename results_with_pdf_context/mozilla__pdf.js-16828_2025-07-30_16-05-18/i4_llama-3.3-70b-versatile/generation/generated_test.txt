it("should support pasting an image from the clipboard", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { StampEditor } = await import("../../src/display/editor/stamp.js");
  const uiManager = new AnnotationEditorUIManager(null, null, null, null, null);
  const editor = new StampEditor({ parent: null, id: "id", uiManager });
  const item = { type: "image/png", getAsFile: () => new File(["image"], "image.png") };
  const clipboardData = { items: [item] };
  const event = { clipboardData, preventDefault: () => {} };
  uiManager.paste(event);
  expect(StampEditor.isHandlingMimeForPasting(item.type)).toBe(true);
});
