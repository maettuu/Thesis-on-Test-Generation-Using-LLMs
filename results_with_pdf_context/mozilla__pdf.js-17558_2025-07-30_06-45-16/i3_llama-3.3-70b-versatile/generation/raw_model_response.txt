it("should render emojis in PDFs", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1669097.pdf'));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const operatorList = await page.getOperatorList({
    intent: RenderingIntentFlag.DISPLAY,
  });
  const textItems = operatorList.opList.items.filter(
    (item) => item.fn === OPS.showText
  );
  const renderedText = textItems
    .map((item) => item.args.join(" "))
    .join("");
  const expectedText = "Hello ðŸ‘‹ World!";
  expect(renderedText).toContain(expectedText);
});
