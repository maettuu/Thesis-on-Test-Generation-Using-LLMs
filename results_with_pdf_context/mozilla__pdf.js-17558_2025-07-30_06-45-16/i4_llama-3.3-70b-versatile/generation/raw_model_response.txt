it("should render emojis in PDFs", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1669097.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const operatorList = await pdfPage.getOperatorList({
    intent: 'display',
  });
  const textItems = operatorList.opList.items.filter(item => item[0] === 'Tj');
  const renderedText = textItems.map(item => item[1]).join('');
  const expectedText = 'ğŸ‘';
  expect(renderedText).toContain(expectedText);
});
