it("should decrypt PDFs with RC encryption where R and key length bits", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue19484_1.pdf'));
  const pdf = await loadingTask.promise;
  const page = await pdf.getPage(1);
  const textContent = await page.getTextContent();
  const items = textContent.items;
  const expectedText = "Lorem ipsum dolor sit amet, consectetur adipiscing elit.";
  const actualText = items.map(item => item.str).join('');
  expect(actualText).toContain(expectedText);
});
