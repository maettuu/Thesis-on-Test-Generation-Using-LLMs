it("should support rare RC4 encryption where R=4, key length < 128 bits", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue19484_1.pdf'));
  await loadingTask.promise;
  const pdfDocument = loadingTask.doc;
  const firstPage = await pdfDocument.getPage(1);
  const pageTextContent = await firstPage.getTextContent();
  expect(pageTextContent.items.length).toBeGreaterThan(0);
});
