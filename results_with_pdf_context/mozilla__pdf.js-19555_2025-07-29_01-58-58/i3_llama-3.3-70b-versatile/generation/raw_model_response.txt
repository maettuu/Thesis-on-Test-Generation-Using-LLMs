it("should decrypt PDFs with RC4 encryption where R=4, key length < 128 bits", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue19484_1.pdf'));
  await loadingTask.promise;
  const doc = loadingTask.doc;
  const pages = await doc.getPages();
  const page = pages[0];
  const textContent = await page.getTextContent();
  expect(textContent.items.length).toBeGreaterThan(0);
});
