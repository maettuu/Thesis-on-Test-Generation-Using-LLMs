```javascript
it("should throw an error when numberToString is called with null", async () => {
  const { numberToString } = await import("../../src/core/core_utils.js");
  const { assert } = await import("../shared/util.js");
  let error;
  try {
    numberToString(null);
  } catch (e) {
    error = e;
  }
  assert(error instanceof Error, `Expected numberToString(null) to throw an error, but got ${error}`);
});
```
