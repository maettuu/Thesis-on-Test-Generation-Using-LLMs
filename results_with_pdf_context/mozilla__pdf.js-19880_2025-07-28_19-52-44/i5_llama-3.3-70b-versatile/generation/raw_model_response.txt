```javascript
it("should throw an error when null is passed to numberToString function", async () => {
  const { numberToString } = await import("../../src/core/core_utils.js");
  const value = null;
  let error;
  try {
    numberToString(value);
  } catch (e) {
    error = e;
  }
  expect(error).not.toBeUndefined();
});
```
