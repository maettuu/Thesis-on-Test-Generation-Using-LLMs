it("should throw an error when input is null", async () => {
  const { assert } = await import("../../src/shared/util.js");
  let errorThrown = false;
  try {
    numberToString(null);
  } catch (error) {
    errorThrown = true;
    assert(error.message.includes("should be a number"));
  }
  assert(errorThrown, "Expected an error to be thrown for null input");
});
