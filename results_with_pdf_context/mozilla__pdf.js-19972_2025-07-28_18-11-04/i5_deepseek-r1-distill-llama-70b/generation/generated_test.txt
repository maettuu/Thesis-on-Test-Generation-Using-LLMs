it("should not throw and leave buffer unchanged when minMax is undefined", async () => {
  const { OperatorList, DrawOPS } = await import("../../src/core/operator_list.js");
  const opList = new OperatorList();
  const transform = [1, 0, 0, 1, 0, 0];
  const buffer = [DrawOPS.moveTo, 10, 20, DrawOPS.lineTo, 30, 40, DrawOPS.curveTo, 1, 2, 3, 4, 5, 6];
  const args = [null, [buffer], undefined];
  expect(() => opList.addState([transform], 0, args)).not.toThrow();
  expect(buffer).toEqual([DrawOPS.moveTo, 10, 20, DrawOPS.lineTo, 30, 40, DrawOPS.curveTo, 1, 2, 3, 4, 5, 6]);
});
