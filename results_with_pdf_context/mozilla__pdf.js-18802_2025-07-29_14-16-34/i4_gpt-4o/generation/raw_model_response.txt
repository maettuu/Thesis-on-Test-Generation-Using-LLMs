```javascript
it("should render PDF text correctly with GillSansMT font", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue18801.pdf'));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const textContent = await page.getTextContent();
  const textItems = textContent.items.map(item => item.str);
  const expectedText = ["Expected text from PDF"];
  expect(textItems).toEqual(expect.arrayContaining(expectedText));
});
```
