it("should resize SVG images horizontallyvertically in pdfjs", async () => {
  const { PDFDocument, PDFPageProxy } = await import("../../src/display/api.js");
  const { AnnotationEditorUIManager, CommandManager } = await import("../../src/display/editor/tools.js");
  const { PDFDateString, PageViewport } = await import("../../src/display/display_utils.js");
  const { Util } = await import("../../src/shared/util.js");

  const pdfDoc = await PDFDocument.load(await (await import("../../src/display/fetch_stream.js")).PDFFetchStream.fetch("path_to_pdf_file"));
  const pdfPage = pdfDoc.getPage(1);
  const viewport = pdfPage.getViewport({ scale: 1 });
  const annotationEditorUIManager = new AnnotationEditorUIManager();
  const commandManager = new CommandManager();

  const svgImage = `<svg width="100" height="100"><rect x="10" y="10" width="50" height="50" fill="#ff0000"/></svg>`;

  const expectedWidth = 200;
  const expectedHeight = 200;

  const annotation = await annotationEditorUIManager.addSVGAnnotation(svgImage, viewport);
  await commandManager.resizeAnnotation(annotation, expectedWidth, expectedHeight);

  const actualWidth = annotation.width;
  const actualHeight = annotation.height;

  Util.assert(actualWidth === expectedWidth, `Expected width ${expectedWidth} but got ${actualWidth}`);
  Util.assert(actualHeight === expectedHeight, `Expected height ${expectedHeight} but got ${actualHeight}`);
});
