it("should allow SVG images to be resized horizontallyvertically", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const svgData = `data:image/svg+xml;charset=UTF-8,<svg xmlns="http://www.w3.org/2000/svg" width="100" height="100"><rect width="100" height="100" style="fill:blue;"/></svg>`;
  const imageManager = new ImageManager();
  const data = await imageManager.#get("testSvg", svgData);
  const canvas = new OffscreenCanvas(200, 200);
  const ctx = canvas.getContext("2d");
  ctx.drawImage(data.bitmap, 0, 0, 200, 200);
  const imageData = ctx.getImageData(0, 0, 1, 1).data;
  const isResized = imageData[2] === 255; // Check if the pixel is blue
  expect(isResized).toBe(true);
});
