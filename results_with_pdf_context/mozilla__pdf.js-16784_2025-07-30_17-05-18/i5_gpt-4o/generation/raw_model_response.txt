it("should allow SVG images to be resized horizontally/vertically", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const { shadow } = await import("../../src/shared/util.js");

  const imageManager = new ImageManager();
  const svgImage = { type: "image/svg+xml" };
  const rawData = new Uint8Array();

  const mustRemoveAspectRatioPromise = ImageManager._isSVGFittingCanvas;
  const expected = shadow(imageManager, "_isSVGFittingCanvas", mustRemoveAspectRatioPromise);

  const actual = await imageManager.#get(svgImage, rawData);
  expect(actual).toEqual(expected);
});
