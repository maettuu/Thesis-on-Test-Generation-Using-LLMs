it("should allow resizing of SVG images horizontally/vertically in pdf.js", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const { Util } = await import("../../shared/util.js");
  const imageManager = new ImageManager();
  const svgImage = new Blob([`<svg viewBox="0 0 1 1" width="1" height="1" xmlns="http://www.w3.org/2000/svg"><rect width="1" height="1" style="fill:red;"/></svg>`], { type: "image/svg+xml" });
  const image = await imageManager.#get("svgImage", svgImage);
  const expected = true;
  const actual = image.isSvg;
  Util.assert(actual === expected, `Expected image.isSvg to be ${expected}, but got ${actual}`);
});
