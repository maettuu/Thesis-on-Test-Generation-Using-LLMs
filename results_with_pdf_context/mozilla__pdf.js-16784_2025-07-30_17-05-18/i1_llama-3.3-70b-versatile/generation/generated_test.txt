it("should allow resizing of SVG images in pdfjs", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const { Util } = await import("../../shared/util.js");
  const imageManager = new ImageManager();
  const svgImage = new Blob([`
    <svg viewBox="0 0 100 100" width="100" height="100" xmlns="http://www.w3.org/2000/svg">
      <rect width="100" height="100" style="fill:red;"/>
    </svg>
  `], { type: "image/svg+xml" });
  const imageData = await imageManager.#get("svgImage", svgImage);
  const imageElement = imageData.bitmap;
  imageElement.width = 200;
  imageElement.height = 200;
  const expectedWidth = 200;
  const expectedHeight = 200;
  assert.strictEqual(imageElement.width, expectedWidth);
  assert.strictEqual(imageElement.height, expectedHeight);
});
