it("should allow resizing of SVG images in pdf.js", async () => {
  const { ImageManager } = await import("../../src/display/editor/tools.js");
  const imageManager = new ImageManager();
  const svgImage = new Blob([`
    <svg viewBox="0 0 100 100" width="100" height="100" xmlns="http://www.w3.org/2000/svg">
      <rect width="100" height="100" style="fill:red;"/>
    </svg>
  `], { type: "image/svg+xml" });
  const image = await imageManager.getFromFile(svgImage);
  const svgUrl = imageManager.getSvgUrl(image.id);
  const expectedSvgUrl = svgUrl.includes("#svgView(preserveAspectRatio(none))");
  expect(expectedSvgUrl).toBe(true);
});
