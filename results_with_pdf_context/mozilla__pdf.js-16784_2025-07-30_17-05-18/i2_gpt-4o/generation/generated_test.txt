it("should allow SVG images to be resized horizontallyvertically", async () => {
  const { shadow } = await import("../../src/shared/util.js");
  const { ImageManager } = await import("../../src/display/editor/tools.js");

  const svgData = `data:image/svg+xml;charset=UTF-8,<svg viewBox="0 0 1 1" width="1" height="1" xmlns="http://www.w3.org/2000/svg"><rect width="1" height="1" style="fill:red;"/></svg>`;
  const imageManager = new ImageManager();

  const dataBeforePatch = await imageManager.#get("testKey", svgData);
  const isSvgBeforePatch = dataBeforePatch.isSvg;

  shadow(ImageManager, "_isSVGFittingCanvas", Promise.resolve(true));

  const dataAfterPatch = await imageManager.#get("testKey", svgData);
  const isSvgAfterPatch = dataAfterPatch.isSvg;

  expect(isSvgBeforePatch).toBe(false);
  expect(isSvgAfterPatch).toBe(true);
});
