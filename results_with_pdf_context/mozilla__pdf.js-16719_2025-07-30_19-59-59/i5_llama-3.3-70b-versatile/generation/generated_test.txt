it("should delete the selected element when pressing the Delete key on a Mac", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { PDFDocument, Page } = await import("../../src/core/document.js");
  const { AnnotationLayer } = await import("../../src/display/annotation_layer.js");
  const { PDFPageProxy } = await import("../../src/display/api.js");

  const pdfDocument = await PDFDocument.load(await (await import("../../src/display/fetch_stream.js")).PDFFetchStream.fetch("path_to_pdf"));
  const pdfPage = pdfDocument.getPage(1);
  const pageProxy = new PDFPageProxy(pdfDocument, pdfPage);
  const annotationLayer = new AnnotationLayer({
    pageProxy: pageProxy,
    viewport: await pageProxy.getViewport(1),
  });
  const annotationEditorUIManager = new AnnotationEditorUIManager({
    annotationLayer: annotationLayer,
    pageProxy: pageProxy,
  });

  const deleteKey = "mac+Delete";
  const expected = true;

  annotationEditorUIManager.selectAnnotation(0);
  annotationEditorUIManager.handleKeyDown({ key: deleteKey });
  const actual = annotationEditorUIManager.getSelectedAnnotations().length === 0;

  expect(actual).toBe(expected);
});
