```javascript
it("should delete the selected element when pressing the Delete key on macOS", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const uiManager = new AnnotationEditorUIManager();
  const mockDeleteFunction = jest.fn();
  uiManager.delete = mockDeleteFunction;

  const event = new KeyboardEvent("keydown", { key: "Delete", metaKey: true });
  document.dispatchEvent(event);

  expect(mockDeleteFunction).toHaveBeenCalled();
});
```
