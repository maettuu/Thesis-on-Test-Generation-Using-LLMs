it("should delete the selected element when pressing the Delete key on macOS", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const eventBus = { _on: () => {} };
  const pdfDocument = { annotationStorage: {}, filterFactory: {} };
  const manager = new AnnotationEditorUIManager(null, eventBus, pdfDocument, null);
  const deleteSpy = jest.spyOn(manager, 'delete');

  manager.#selectedEditors.add("testElement");
  const event = new KeyboardEvent("keydown", { key: "Delete", metaKey: true });
  manager.keydown(event);

  expect(deleteSpy).toHaveBeenCalled();
});
