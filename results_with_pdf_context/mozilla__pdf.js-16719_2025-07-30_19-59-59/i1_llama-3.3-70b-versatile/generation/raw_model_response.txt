```javascript
it("should delete the selected element when pressing the Delete key", async () => {
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { Annotation } = await import("../../src/core/annotation.js");
  const { createIdFactory } = await import("./test_utils.js");
  const idFactory = createIdFactory();
  const annotation = new Annotation({
    subtype: "Highlight",
    rect: [10, 10, 20, 20],
    page: 1,
    id: idFactory.next(),
  });
  const annotationEditorUIManager = new AnnotationEditorUIManager({
    addEventListener: () => {},
    removeEventListener: () => {},
  }, {
    _on: () => {},
  }, {
    annotationStorage: {
      getAnnotations: () => [annotation],
    },
  }, {});
  annotationEditorUIManager.#selectedEditors.add(annotation);
  const deleteSpy = jest.spyOn(annotationEditorUIManager, "delete");
  annotationEditorUIManager.keydown({
    key: "Delete",
    preventDefault: () => {},
  });
  expect(deleteSpy).toHaveBeenCalledTimes(1);
});
```
