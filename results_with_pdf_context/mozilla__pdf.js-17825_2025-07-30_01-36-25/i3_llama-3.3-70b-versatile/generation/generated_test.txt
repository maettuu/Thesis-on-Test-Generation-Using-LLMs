it("should add a visually hidden accessible label to the highlight floating button", async () => {
  const button = await page.waitForSelector(".highlightButton");
  const span = await button.$(".visuallyHidden");
  if (!span) {
    throw new Error("The accessible label span is missing on the highlight button.");
  }
  const dataL10nId = await page.evaluate(el => el.getAttribute("data-l10n-id"), span);
  if (dataL10nId !== "pdfjs-highlight-floating-button-label") {
    throw new Error(`Expected span data-l10n-id to be "pdfjs-highlight-floating-button-label", but got "${dataL10nId}"`);
  }
});
