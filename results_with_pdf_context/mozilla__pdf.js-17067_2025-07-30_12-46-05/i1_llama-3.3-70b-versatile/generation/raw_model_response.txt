it("should render patterns correctly after applying a transform", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue17065.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const viewport = pdfPage.getViewport({ scale: 1 });
  const renderTask = pdfPage.render({
    viewport,
    canvasContext: null,
    renderInteractive: false,
    enableWebGL: false,
  });
  const renderResult = await renderTask.promise;
  const expectedRenderResult = true;
  expect(renderResult instanceof Object).toBe(expectedRenderResult);
});
