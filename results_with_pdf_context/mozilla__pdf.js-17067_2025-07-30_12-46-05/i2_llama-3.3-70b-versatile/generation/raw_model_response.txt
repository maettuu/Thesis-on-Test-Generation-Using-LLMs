it("should correctly render transformed rectangles with patterns", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue17065.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const canvas = document.createElement('canvas');
  const ctx = canvas.getContext('2d');
  const renderTask = pdfPage.render({
    canvasContext: ctx,
    viewport: pdfPage.getViewport({ scale: 1 }),
  });
  await renderTask.promise;
  const expectedImageData = await new Promise((resolve) => {
    ctx.getImageData(0, 0, canvas.width, canvas.height).then(resolve);
  });
  const actualImageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
  expect(actualImageData.data).toEqual(expectedImageData.data);
});
