it("should render the PDF with font fallback on nodejslegacy build", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue19234.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const image = await pdfPage.render({
    canvasContext: null,
    viewport: pdfPage.getViewport({ scale: 1 }),
  });
  const expectedText = "1040";
  const actualText = await pdfPage.getTextContent();
  const textItems = actualText.items;
  let found = false;
  for (let i = 0; i < textItems.length; i++) {
    if (textItems[i].str === expectedText) {
      found = true;
      break;
    }
  }
  expect(found).toBe(true);
});
