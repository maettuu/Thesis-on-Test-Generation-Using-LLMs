it("should save PDF annotations correctly", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1844572.pdf'));
  const pdfDoc = await loadingTask.promise;
  const pdfPages = await pdfDoc.getPages();
  const firstPage = pdfPages[0];
  const annotations = await firstPage.getAnnotations();
  const annotationStorage = new Map();
  for (const annotation of annotations) {
    annotationStorage.set(annotation.id, annotation);
  }
  const updatedPdf = await pdfDoc.saveDocument({ annotationStorage });
  const updatedLoadingTask = getDocument(buildGetDocumentParams(updatedPdf));
  const updatedPdfDoc = await updatedLoadingTask.promise;
  const updatedPdfPages = await updatedPdfDoc.getPages();
  const updatedFirstPage = updatedPdfPages[0];
  const updatedAnnotations = await updatedFirstPage.getAnnotations();
  expect(updatedAnnotations.length).toBe(annotations.length);
});
