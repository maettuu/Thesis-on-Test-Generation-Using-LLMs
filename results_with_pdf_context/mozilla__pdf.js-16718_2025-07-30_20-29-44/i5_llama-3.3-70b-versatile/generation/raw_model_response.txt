it("should save PDF annotations on certain PDFs", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1844572.pdf'));
  const pdfDoc = await loadingTask.promise;
  const firstPage = pdfDoc.getPage(1);
  const annotations = await firstPage.getAnnotations();
  const annotation = annotations[0];
  annotation.setFlags(2); // Set the hidden flag
  const pdfBytes = await pdfDoc.save();
  const loadedPdfDoc = await getDocument({ data: pdfBytes }).promise;
  const loadedFirstPage = await loadedPdfDoc.getPage(1);
  const loadedAnnotations = await loadedFirstPage.getAnnotations();
  const loadedAnnotation = loadedAnnotations[0];
  const expectedFlags = 2;
  const actualFlags = loadedAnnotation.flags;
  expect(actualFlags).toBe(expectedFlags);
});
