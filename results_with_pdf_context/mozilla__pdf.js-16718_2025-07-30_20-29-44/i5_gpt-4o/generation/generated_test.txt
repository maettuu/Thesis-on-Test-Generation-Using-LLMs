it("should save PDF annotations correctly on bugpdf", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1844572.pdf'));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const annotations = await page.getAnnotations();
  const expectedAnnotationCount = 1; // Assuming there's one annotation to check
  const actualAnnotationCount = annotations.length;
  expect(actualAnnotationCount).toBe(expectedAnnotationCount);
});
