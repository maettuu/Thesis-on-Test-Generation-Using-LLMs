it("should render PDF with noninteger DW correctly", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const { buildGetDocumentParams } = await import("./test_utils.js");

  const loadingTask = getDocument(buildGetDocumentParams("bug1903731.pdf"));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);

  const operatorList = await page.getOperatorList();
  const hasCorrectDefaultWidth = operatorList.argsArray.some(
    args => args[0] === OPS.setFont && args[1] === 1000
  );

  expect(hasCorrectDefaultWidth).toBe(true);
});
