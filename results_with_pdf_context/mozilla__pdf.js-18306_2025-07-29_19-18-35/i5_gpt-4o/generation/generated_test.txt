it("should render PDF with noninteger DW correctly", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1903731.pdf'));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const viewport = page.getViewport({ scale: 1.0 });
  const canvasFactory = new (await import('../../src/display/display_utils.js')).DOMCanvasFactory();
  const canvasAndContext = canvasFactory.create(viewport.width, viewport.height);
  const renderTask = page.render({
    canvasContext: canvasAndContext.context,
    viewport: viewport,
  });
  await renderTask.promise;
  const pixelData = canvasAndContext.context.getImageData(0, 0, viewport.width, viewport.height).data;
  const nonEmptyPixels = pixelData.some(value => value !== 0);
  expect(nonEmptyPixels).toBe(true);
});
