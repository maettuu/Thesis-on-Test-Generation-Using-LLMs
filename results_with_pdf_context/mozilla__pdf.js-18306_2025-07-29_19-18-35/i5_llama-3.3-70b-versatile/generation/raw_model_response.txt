it("should render PDF with non-integer DW correctly", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1903731.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const renderTask = pdfPage.render({
    viewport: pdfPage.getViewport({ scale: 1 }),
    canvasContext: null,
    renderInteractive: false,
    disableWebGL: true,
  });
  const renderedPage = await renderTask.promise;
  const expectedWidth = 1000;
  const actualWidth = renderedPage.width;
  expect(actualWidth).toBeGreaterThan(0);
  const { PartialEvaluator } = await import('../../src/core/evaluator.js');
  const evaluator = new PartialEvaluator();
  const dict = pdfPage.getAnnotations()[0].getDict();
  const dw = dict.get("DW");
  const defaultWidth = typeof dw === "number" ? Math.ceil(dw) : 1000;
  expect(defaultWidth).toBeGreaterThan(0);
});
