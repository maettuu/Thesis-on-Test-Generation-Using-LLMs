it("should render the signature page of a digitally signed PDF correctly", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue18973.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(45);
  const renderTask = pdfPage.render({
    viewport: pdfPage.getViewport({ scale: 1 }),
    canvasContext: null,
    renderInteractive: false,
    disableFontFace: true,
  });
  const renderedPage = await renderTask.promise;
  const expectedText = "Signature";
  const actualText = await pdfPage.getTextContent();
  const actualTextItems = actualText.items;
  const textFound = actualTextItems.some((item) => item.str.includes(expectedText));
  expect(textFound).toBe(true);
});
