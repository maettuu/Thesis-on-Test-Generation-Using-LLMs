it("should correctly decode the signature page of a digitally signed PDF", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue18973.pdf'));
  const pdf = await loadingTask.promise;
  const page = await pdf.getPage(45);
  const textContent = await page.getTextContent();
  const expectedText = "Signature";
  const actualText = textContent.items[0].str;
  expect(actualText).toContain(expectedText);
});
