```javascript
it("should generate a text layer for the given PDF", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue19954.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const textContent = await pdfPage.getTextContent();
  const expectedText = "Capital Bancorp Inc. Corporate Responsibility";
  const actualText = textContent.items[0].str;
  expect(actualText).toContain(expectedText);
});
```
