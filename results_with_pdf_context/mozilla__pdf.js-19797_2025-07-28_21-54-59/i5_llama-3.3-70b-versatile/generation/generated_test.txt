it("should not throw TypeError when drawing annotations", async () => {
  const { PDFDocument, PDFPageProxy } = await import("../../src/display/api.js");
  const { AnnotationEditorUIManager } = await import("../../src/display/editor/tools.js");
  const { TouchManager } = await import("../../src/display/touch_manager.js");

  const pdfDocument = await PDFDocument.load(await (await import("../../src/display/node_utils.js")).fetchData("https://github.com/user-attachments/files/19669325/blank.pdf"));
  const pdfPage = pdfDocument.getPage(1);
  const annotationEditor = new AnnotationEditorUIManager();
  const touchManager = new TouchManager();

  let error = null;
  try {
    annotationEditor.setMode(AnnotationEditorUIManager.AnnotationEditorType.DRAW);
    touchManager.beginTouch(10, 10);
    touchManager.moveTouch(20, 20);
    touchManager.endTouch();
  } catch (e) {
    error = e;
  }

  expect(error).toBeNull();
});
