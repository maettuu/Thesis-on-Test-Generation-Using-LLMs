it("should not throw TypeError when drawing annotations", async () => {
  const { PDFDocumentProxy } = await import("../../src/display/api.js");
  const { TouchManager } = await import("../../src/display/touch_manager.js");

  const pdfDocument = new PDFDocumentProxy();
  const touchManager = new TouchManager();

  let errorOccurred = false;
  try {
    touchManager.#touchMoveAC = null; // Simulate the condition where #touchMoveAC is null
    touchManager.#onTouchEnd({ touches: [{}] }); // Simulate a touch end event
  } catch (error) {
    if (error instanceof TypeError) {
      errorOccurred = true;
    }
  }

  expect(errorOccurred).toBe(false);
});
