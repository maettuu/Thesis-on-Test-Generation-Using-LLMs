it("should not throw TypeError when switching annotation modes", async () => {
  const { AnnotationEditorUIManager, TouchManager } = await import("../../src/display/editor/tools.js");
  const { PDFDocument, PDFPage } = await import("../../src/core/document.js");
  const pdfDocument = await PDFDocument.load(await fetchData("https://github.com/user-attachments/files/19669325/blank.pdf"));
  const container = { addEventListener: () => {}, removeEventListener: () => {} };
  const viewer = { getPages: () => [new PDFPage(pdfDocument)] };
  const annotationEditor = new AnnotationEditorUIManager(container, viewer, null, null, null, pdfDocument, null, null, false, false, false, null, null, true);
  const touchManager = new TouchManager({ container, isPinchingDisabled: null, isPinchingStopped: null, onPinchStart: null, onPinching: null, onPinchEnd: null, signal: new AbortController().signal });
  await annotationEditor.updateMode(AnnotationEditorType.INK);
  touchManager.#onTouchEnd({ touches: [] });
  await annotationEditor.updateMode(AnnotationEditorType.NONE);
  expect(() => touchManager.#onTouchEnd({ touches: [] })).not.toThrowError(TypeError);
});
