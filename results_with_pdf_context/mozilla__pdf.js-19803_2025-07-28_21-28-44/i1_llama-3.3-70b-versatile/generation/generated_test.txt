it("should prevent circular references in Form XObjects", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('evil_xobjects.pdf'));
  const pdf = await loadingTask.promise;
  const pages = await pdf.getPages();
  const page = pages[0];
  const renderedPage = await page.render({ intent: 'display' });
  expect(renderedPage).not.toBeNull();
});
