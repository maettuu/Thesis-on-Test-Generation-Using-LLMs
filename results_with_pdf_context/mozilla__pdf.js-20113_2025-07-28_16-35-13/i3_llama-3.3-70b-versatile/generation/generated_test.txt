it("should resolve the issue with HTML Spec page printed to PDF not being opened in Firefox", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1978317.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const pageViewport = pdfPage.getViewport({ scale: 1 });
  const renderTask = pdfPage.render({
    viewport: pageViewport,
    canvasContext: null,
    renderInteractive: false,
    disableWebGL: true,
  });
  await renderTask.promise;
  const expectedError = null;
  let actualError = null;
  try {
    await renderTask.promise;
  } catch (error) {
    actualError = error;
  }
  expect(actualError).toBe(expectedError);
});
