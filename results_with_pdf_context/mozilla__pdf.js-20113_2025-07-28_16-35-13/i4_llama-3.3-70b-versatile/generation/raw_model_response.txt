it("should resolve the issue with HTML Spec page printed to PDF not opening in Firefox", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1978317.pdf'));
  await loadingTask.promise;
  const pdfDocument = loadingTask.doc;
  const xref = pdfDocument.xref;
  const entries = xref.entries;
  const nums = [];
  const offsets = [];
  for (let i = 0; i < entries.length; i++) {
    const entry = entries[i];
    if (entry && entry.offset && !entry.free) {
      nums.push(i);
      offsets.push(entry.offset);
    }
  }
  for (let i = 0; i < nums.length; i++) {
    const num = nums[i];
    const offset = offsets[i];
    const entry = xref.getEntry(num);
    if (entry && entry.offset === offset && entry.gen !== i) {
      entry.gen = i;
    }
  }
  const root = xref.getCatalogObj();
  expect(root).not.toBeNull();
});
