it("should render vertical punctuation correctly in missing vertical fonts", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1905623.pdf'));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const textContent = await page.getTextContent();

  const expectedVerticalPunctuation = "︑︒"; // Example of expected vertical punctuation
  const actualText = textContent.items.map(item => item.str).join("");

  expect(actualText).toContain(expectedVerticalPunctuation);
});
