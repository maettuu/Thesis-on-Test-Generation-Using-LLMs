it("should render Japanese vertical punctuation correctly", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = await getDocument(buildGetDocumentParams('bug1905623.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const viewport = pdfPage.getViewport({ scale: 1 });
  const renderTask = pdfPage.render({
    viewport,
    canvasContext: null,
    renderInteractive: false,
    annotationMode: null,
    includeAnnotations: false,
    useFontSubsetting: true,
  });
  await renderTask.promise;
  const textContent = await pdfPage.getTextContent();
  const expectedText = "â€”"; // expected Japanese vertical punctuation
  const actualText = textContent.items.find(item => item.str === expectedText);
  expect(actualText).not.toBeNull();
});
