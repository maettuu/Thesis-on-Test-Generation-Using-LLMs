it("should render Japanese vertical punctuation correctly", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('bug1905623.pdf'));
  const pdfDocument = await loadingTask.promise;
  const pdfPage = await pdfDocument.getPage(1);
  const viewport = new PageViewport({ width: 1000, height: 1000 });
  const renderTask = pdfPage.render({ viewport, });
  const renderedPage = await renderTask.promise;
  const textContent = await pdfPage.getTextContent();
  const expectedText = "expected text with correct Japanese vertical punctuation";
  expect(textContent.items[0].str).toBe(expectedText);
});
