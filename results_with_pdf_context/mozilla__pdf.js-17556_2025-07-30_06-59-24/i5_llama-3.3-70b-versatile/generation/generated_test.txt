it("should ignore invalid operators like toString", async () => {
  const { getDocument } = await import('../../src/display/api.js');
  const { buildGetDocumentParams } = await import('./test_utils.js');
  const loadingTask = getDocument(buildGetDocumentParams('issue17554.pdf'));
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const operatorList = await page.getOperatorList();
  const validOperators = ['m', 'l', 'c', 'v', 'y', 'h', 're', 'S', 's', 'f', 'F', 'f*', 'B', 'B*', 'b', 'b*', 'n'];
  let isValid = true;
  for (const operator of operatorList) {
    if (!validOperators.includes(operator) && operator !== 'toString') {
      isValid = false;
      break;
    }
  }
  expect(isValid).toBe(true);
});
