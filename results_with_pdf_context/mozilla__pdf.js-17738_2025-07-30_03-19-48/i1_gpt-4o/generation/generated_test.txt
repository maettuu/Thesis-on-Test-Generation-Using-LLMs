it("should mark locked link annotations as clickable by ensuring datanoHTML is false", async () => {
  const { AnnotationFactory, AnnotationType } = await import("../../src/core/annotation.js");
  const dummyDict = {
    get(key) {
      if (key === "Subtype") {
        return "Link";
      }
      return undefined;
    }
  };
  const annotation = AnnotationFactory.create({
    dict: dummyDict,
    annotationGlobals: {},
    rectangle: [0, 0, 10, 10],
  });
  if (annotation.data.annotationType !== AnnotationType.LINK) {
    throw new Error("Annotation is not of type LINK");
  }
  expect(annotation.data.noHTML).toBe(false);
});
