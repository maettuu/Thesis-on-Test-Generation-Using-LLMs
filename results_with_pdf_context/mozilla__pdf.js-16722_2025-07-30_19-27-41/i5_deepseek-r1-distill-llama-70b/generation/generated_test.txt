it("should not reset filter when its already none", async () => {
  const { resetCtxToDefault } = await import("../../src/display/canvas.js");
  let setterCalled = false;
  const ctx = {
    filter: "none",
    get filter() {
      return this._filter;
    },
    set filter(value) {
      setterCalled = true;
      this._filter = value;
    },
  };
  ctx.filter = "none";
  resetCtxToDefault(ctx);
  expect(setterCalled).toBe(false);
});
