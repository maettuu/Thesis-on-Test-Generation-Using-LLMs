it("should render poster PDF quickly", async () => {
  const { PDFDocument, PDFPageProxy } = await import("../../src/display/api.js");
  const { createIdFactory } = await import("./test_utils.js");
  const idFactory = createIdFactory();
  const pdfDoc = await PDFDocument.load({ url: `${TEST_PDFS_PATH}/poster.pdf`, cMapUrl: CMAP_URL, cMapPacked: true, standardFontDataUrl: STANDARD_FONT_DATA_URL, fontExtraProperties: {}, isEvalSupported: true, useSystemFonts: false, useWorker: true, workerPort: new LoopbackPort(idFactory) });
  const page = await pdfDoc.getPage(1);
  const renderTask = page.render({ canvasContext: { filter: "blur(10px)" }, viewport: page.getViewport({ scale: 1 }) });
  const startTime = Date.now();
  await renderTask.promise;
  const endTime = Date.now();
  const renderTime = endTime - startTime;
  const expectedRenderTime = 1000; // 1 second
  if (renderTime > expectedRenderTime) {
    throw new Error(`Rendering took too long: ${renderTime}ms`);
  }
});
