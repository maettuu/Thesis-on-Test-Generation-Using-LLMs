it("should render poster PDF without being slow on Firefox", async () => {
  const pdfFile = `${TEST_PDFS_PATH}/poster.pdf`;
  const params = buildGetDocumentParams(pdfFile, null, null, DefaultFileReaderFactory);
  const pdfDocument = await getDocument(params);
  const pdfPage = await pdfDocument.getPage(1);
  const viewport = new PageViewport({ width: 100, height: 100 });
  const renderTask = pdfPage.render({
    viewport,
    canvasContext: new CanvasGraphics(
      new DefaultCanvasFactory().create(100, 100),
      {},
      {},
      new DefaultCanvasFactory(),
      null,
      {}
    ),
    continueCallback: null,
  });
  await renderTask.promise;
  expect(renderTask.status).toBe("finished");
});
