it("should correctly render a manually hinted TrueType font", async () => {
  const { getDocument } = await import("../../src/display/api.js");
  const { TEST_PDFS_PATH } = await import("./test_utils.js");
  // This file contains a manually hinted TrueType font that previously caused rasterisation errors.
  const loadingTask = getDocument(`${TEST_PDFS_PATH}/manual_hinted_trueType.pdf`);
  const pdfDocument = await loadingTask.promise;
  const page = await pdfDocument.getPage(1);
  const textContent = await page.getTextContent();
  const text = textContent.items.map(item => item.str).join("");
  // The expected text is embedded in the PDF when the font is correctly processed.
  expect(text).toContain("This is a hinted TrueType test");
});
